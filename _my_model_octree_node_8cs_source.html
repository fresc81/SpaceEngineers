456  // Calculate min/max coordinates of a child's bounding box
457  BoundingBox GetChildBoundingBox(BoundingBox parentBoundingBox, int childIndex)
458  {
459  // Get child offset
460  Vector3 offset;
461  switch (childIndex)
462  {
463  case 0: offset = new Vector3(0, 0, 0); break;
464  case 1: offset = new Vector3(1, 0, 0); break;
465  case 2: offset = new Vector3(1, 0, 1); break;
466  case 3: offset = new Vector3(0, 0, 1); break;
467  case 4: offset = new Vector3(0, 1, 0); break;
468  case 5: offset = new Vector3(1, 1, 0); break;
469  case 6: offset = new Vector3(1, 1, 1); break;
470  case 7: offset = new Vector3(0, 1, 1); break;
471  default: throw new InvalidBranchException(); break;
472  }
473 
474  // Calc size of child's bounding box
475  Vector3 size = (parentBoundingBox.Max - parentBoundingBox.Min) / 2.0f;
476 
477  BoundingBox ret = new BoundingBox();
478  ret.Min = parentBoundingBox.Min + offset * size;
479  ret.Max = ret.Min + size;
480 
481  ret.Min -= size * CHILD_BOUNDING_BOX_EXPAND;
482  ret.Max += size * CHILD_BOUNDING_BOX_EXPAND;
483 
484  // Make sure we stay in parent bounding box
485  ret.Min = Vector3.Max(ret.Min, parentBoundingBox.Min);
486  ret.Max = Vector3.Min(ret.Max, parentBoundingBox.Max);
487 
488  return ret;
489  }
