493  private void ApplyShakeForce(float standbyRotationRatio = 1.0f)
494  {
495  if (CubeGrid.Physics == null || PositionComp == null)
496  return;
497 
498  const float PeriodA = 13.35f;
499  const float PeriodB = 18.154f;
500  const float ShakeForceStrength = 240.0f;
501  int offset = GetHashCode(); // Different offset for each drill
502 
503  float strength = CubeGrid.GridSizeEnum == MyCubeSize.Small ? 1.0f : 5.0f;
504  var axisA = WorldMatrix.Up;
505  var axisB = WorldMatrix.Right;
506  var force = Vector3.Zero;
507  float timeMs = (float)MyPerformanceCounter.TicksToMs(MyPerformanceCounter.ElapsedTicks);
508  force += axisA * (float)Math.Sin(offset + timeMs * PeriodA / 5);
509  force += axisB * (float)Math.Sin(offset + timeMs * PeriodB / 5);
510  force *= standbyRotationRatio * strength * ShakeForceStrength * m_drillBase.AnimationMaxSpeedRatio * m_drillBase.AnimationMaxSpeedRatio; // Quadratic fade out looks better
511 
512  CubeGrid.Physics.AddForce(MyPhysicsForceType.APPLY_WORLD_IMPULSE_AND_WORLD_ANGULAR_IMPULSE, force, PositionComp.GetPosition(), null);
513  }
