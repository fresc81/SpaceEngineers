154  private static void MoveInternal(MyEntity entity)
155  {
156  if (entity.Parent != null && (entity.Flags & EntityFlags.IsGamePrunningStructureObject) == 0)
157  return;
158  ProfilerShort.Begin(string.Format("Move:{0}", (entity.GetTopMostParent() == entity ? "Topmost" : "Child")));
159  if (entity.TopMostPruningProxyId != MyVRageConstants.PRUNING_PROXY_ID_UNITIALIZED)
160  {
161  BoundingBoxD bbox = GetEntityAABB(entity);
162 
163  if (bbox.Size == Vector3D.Zero) // remove entities with zero bounding boxes
164  {
165  Remove(entity);
166  ProfilerShort.End();
167  return;
168  }
169 
170  var voxelMap = entity as MyVoxelBase;
171  if (voxelMap != null)
172  {
173  m_voxelMapsTree.MoveProxy(voxelMap.VoxelMapPruningProxyId, ref bbox, Vector3D.Zero);
174  }
175 
176  if (entity.TopMostPruningProxyId != MyVRageConstants.PRUNING_PROXY_ID_UNITIALIZED)
177  {
178  bool stat = IsEntityStatic(entity);
179 
180  // Swap trees if necessary.
181  if (stat != entity.StaticForPruningStructure)
182  {
183  if (entity.StaticForPruningStructure)
184  {
185  m_staticObjectsTree.RemoveProxy(entity.TopMostPruningProxyId);
186  entity.TopMostPruningProxyId = m_dynamicObjectsTree.AddProxy(ref bbox, entity, 0);
187  }
188  else
189  {
190  m_dynamicObjectsTree.RemoveProxy(entity.TopMostPruningProxyId);
191  entity.TopMostPruningProxyId = m_staticObjectsTree.AddProxy(ref bbox, entity, 0);
192  }
193  entity.StaticForPruningStructure = stat;
194  }
195  else
196  {
197  if (entity.StaticForPruningStructure)
198  m_staticObjectsTree.MoveProxy(entity.TopMostPruningProxyId, ref bbox, Vector3D.Zero);
199  else
200  m_dynamicObjectsTree.MoveProxy(entity.TopMostPruningProxyId, ref bbox, Vector3D.Zero);
201  }
202  }
203  }
204  ProfilerShort.End();
