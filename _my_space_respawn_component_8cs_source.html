572  {

574  MyRespawnComponent closestRespawn = null;

590  continue;
591

593  }
594  else

597  if (respawn.Entity.PositionComp == null) continue;
598 
599  distance = (float)Vector3D.Distance(position, respawn.Entity.PositionComp.GetPosition());
600  }
601 
602  if (distance < closestDistance)
603  {
604  closestRespawn = respawn;
605  closestDistance = distance;
606  }
607 
608  respawns.Add(respawn);
609  }
610

694  private void PutPlayerInRespawnGrid(MyPlayer player, List<MyCubeGrid> respawnGrids, MyBotDefinition botDefinition)
695  {
696  MyCharacter character = null;
697  MyCockpit cockpit = null;
698 
699  // Find cockpits
700  List<MyCockpit> shipCockpits = new List<MyCockpit>();
701  foreach (MyCubeGrid grid in respawnGrids)
702  {
703  foreach (MyCockpit gridCockpit in grid.GetFatBlocks<MyCockpit>())
704  {
705  // Ignore non-functional cockpits
706  if (!gridCockpit.IsFunctional)
707  continue;
708  shipCockpits.Add(gridCockpit);
709  }
710  }
711 
712  // First sort cockpits by order: Ship controlling cockpits set to main, then ship controlling cockpits not set to main, lastly whatever remains, e.g. CryoChambers and Passenger Seats
713  if (shipCockpits.Count > 1)
714  {
715  shipCockpits.Sort(delegate(MyCockpit cockpitA, MyCockpit cockpitB)
716  {
717  int controlCompare = cockpitB.EnableShipControl.CompareTo(cockpitA.EnableShipControl);
718  if (controlCompare != 0) return controlCompare;
719 
720  int mainCompare = cockpitB.IsMainCockpit.CompareTo(cockpitA.IsMainCockpit);
721  if (mainCompare != 0) return mainCompare;
722 
723  return 0;
724  });
725  }
726 
727  // Finally, select the most important cockpit
728  if (shipCockpits.Count > 0)
729  cockpit = shipCockpits[0];
730 
731  // No cockpit on grid. It's not good, but it's safe.
732  // Can be caused by modded respawn ship, wrongly loaded definition (not probably),
733  // or not spawned RS because of blocked place (because of grid spawn paralelization).
734  //System.Diagnostics.Debug.Assert(cockpit != null, "Character is spawning in ship without cockpit!");
735 
736  // Create character
737  MatrixD matrix = MatrixD.Identity;
738  if (cockpit != null)
739  {
740  matrix = cockpit.WorldMatrix;
741  matrix.Translation = cockpit.WorldMatrix.Translation - Vector3.Up - Vector3.Forward;
742  }
743  else if (respawnGrids.Count > 0)
744  {
745  matrix.Translation = respawnGrids[0].PositionComp.WorldAABB.Center + respawnGrids[0].PositionComp.WorldAABB.HalfExtents;
746  }
747 
748  character = MyCharacter.CreateCharacter(matrix, Vector3.Zero, player.Identity.DisplayName, player.Identity.Model, null, botDefinition, cockpit: cockpit, playerSteamId: player.Id.SteamId);
749 
750  CloseRespawnShip(player);
751  foreach (var respawnGrid in respawnGrids)
752  {
753  respawnGrid.ChangeGridOwnership(player.Identity.IdentityId, MyOwnershipShareModeEnum.None);
754  respawnGrid.IsRespawnGrid = true;
755  respawnGrid.m_playedTime = 0;
756  player.RespawnShip.Add(respawnGrid.EntityId);
757  }
758  //SaveRespawnShip(player);
759 
760  if (cockpit != null)
761  {
762  cockpit.AttachPilot(character, false);
763  MyMultiplayer.ReplicateImmediatelly(MyExternalReplicable.FindByObject(cockpit.CubeGrid), new EndpointId(player.Id.SteamId));
764  }
765 
766  if (cockpit == null)
767  {
768  Sync.Players.SetPlayerCharacter(player, character, null);
769  }
770  else
771  {
772  character.SetPlayer(player);
773  Sync.Players.SetControlledEntity(player.Id, cockpit);
774  // Sync.Players.SetPlayerToCockpit(player, cockpit);
775  }
776  Sync.Players.RevivePlayer(player);
777  }

867  private static void GetSpawnPositionNearPlanet(MyPlanet planet, float collisionRadius, ref Vector3D position, out Vector3D forward, out Vector3D up, float planetAtmosphereRatio = 0.3f, float randomRangeMin = 500, float randomRangeMax = 650)
868  {
869  // Position us at a desirable height on the planet
870  // Roughl halfway the planet's upper atmosphere and outer radius
871  float randomHeightAdjustment = MyUtils.GetRandomFloat(randomRangeMin, randomRangeMax);
872  float height = planet.MaximumRadius + planetAtmosphereRatio * (planet.AtmosphereRadius - planet.MaximumRadius) + randomHeightAdjustment;
873 
874  // Compute random position
875  position = planet.PositionComp.WorldVolume.Center + MyUtils.GetRandomVector3Normalized() * height;
876  Vector3D? freePlace = MyEntities.FindFreePlace(position, collisionRadius);
877  int numAttempts = 1;
878 
879  // While we have no valid spawn position, try again 3 more times
880  while (!freePlace.HasValue && numAttempts <= 3)
881  {
882  position = planet.PositionComp.WorldVolume.Center + MyUtils.GetRandomVector3Normalized() * height;
883  freePlace = MyEntities.FindFreePlace(position, collisionRadius);
884  numAttempts += 1;
885  }
886 
887  // If we have a valid position, use it
888  if (freePlace.HasValue)
889  position = freePlace.Value;
890 
891  // Return orientation perpendicular to planet surface
892  Vector3 gravityAtPosition = MyGravityProviderSystem.CalculateNaturalGravityInPoint(position);
893  if (Vector3.IsZero(gravityAtPosition)) // If we are somehow randomized outside of gravity range, make sure some orientation is used
894  gravityAtPosition = Vector3.Up;
895 
896  Vector3D normalizedGravityVector = Vector3D.Normalize(gravityAtPosition);
897  forward = Vector3.CalculatePerpendicularVector(-normalizedGravityVector);
898  up = -normalizedGravityVector;
899  }

911  public static void GetSpawnPosition(float collisionRadius, ref Vector3D position, out Vector3D forward, out Vector3D up, float planetSpawnHeightRatio = 0.3f, float randomRangeMin = 500, float randomRangeMax = 650)
912  {
913  // Are we spawning near a planet?
914  MyPlanet planet = MyGamePruningStructure.GetClosestPlanet(position);
915  if (planet != null)
916  {
917  GetSpawnPositionNearPlanet(planet, collisionRadius, ref position, out forward, out up, planetSpawnHeightRatio, randomRangeMin, randomRangeMax);
918  return;
919  }
920 
921  // Old logic, testing for asteroids and other objects
922  double distance = 0;
923  foreach (var entity in MyEntities.GetEntities())
924  {
925  MyVoxelMap voxelMap = entity as MyVoxelMap;
926 
927  // Only test against voxels
928  if (entity == null) continue;
929 
930  distance = MathHelper.Max(distance, entity.PositionComp.WorldVolume.Center.Length() + entity.PositionComp.WorldVolume.Radius);
931  }
932 
933  // Random range from last voxel
934  distance += MyUtils.GetRandomFloat(randomRangeMin, randomRangeMax);
935 
936  if (MyEntities.IsWorldLimited())
937  distance = Math.Min(distance, MyEntities.WorldSafeHalfExtent());
938  else
939  distance = Math.Min(distance, 20000); // limited spawn area in infinite worlds
940 
941  // Compute random position
942  forward = MyUtils.GetRandomVector3Normalized();
943  up = Vector3D.CalculatePerpendicularVector(forward);
944  Vector3D randomizedPosition = position + (forward * distance);
945 
946  // Test if we can spawn here
947  Vector3D? searchPosition = MyEntities.FindFreePlace(randomizedPosition, collisionRadius);
948  if (searchPosition.HasValue)
949  randomizedPosition = searchPosition.Value;
950 
951  position = randomizedPosition;
952  }
