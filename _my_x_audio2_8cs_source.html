910  {

912  {
913  m_timeFromTransitionStart += stepSizeinMS;

915  if (m_timeFromTransitionStart >= TRANSITION_TIME)

917  m_musicState = MyMusicState.Stopped;
918  if ((m_musicCue != null) && m_musicCue.IsPlaying)
919  {
920  m_musicCue.Stop(true);
921  m_musicCue = null;
922  }
923  }
924  // we decrease music volume (because transition effect)
925  else if ((m_musicCue != null) && m_musicCue.IsPlaying)
926  {
927  if ((m_musicAudioVoice.Volume > 0f) && m_musicOn)
928  m_musicAudioVoice.SetVolume((1f - (float)m_timeFromTransitionStart / TRANSITION_TIME) * m_volumeAtTransitionStart * m_globalVolumeLevel);
929  }
930  }
931 
932  if (m_musicState == MyMusicState.Stopped)
933  {

936  if ((m_currentTransition != null) && (m_nextTransitions.Count > 0) && (nextTransition != null) && (nextTransition.Value.Priority > m_currentTransition.Value.Priority))
937  m_nextTransitions[m_currentTransition.Value.Priority] = m_currentTransition.Value;
938 
939  m_currentTransition = nextTransition;
940  // it there is current transition to play, we play it and set state to playing
941  if (m_currentTransition != null)
942  {
943  m_musicAudioVoice.SetVolume(m_volumeMusic * m_globalVolumeLevel);
944  PlayMusicByTransition(m_currentTransition.Value);
945  m_nextTransitions.Remove(m_currentTransition.Value.Priority);
946  m_musicState = MyMusicState.Playing;
947  }
948  }

953  {
954  if (m_loopMusic && m_currentTransition != null)
955  {
956  // we play current transition in loop
957  Debug.Assert(m_currentTransition != null);
958  PlayMusicByTransition(m_currentTransition.Value);
959  }
960  else
961  {
962  // switches to another, random, track
963  m_currentTransition = null;

965  MyStringId? newTransitionEnum = MyStringId.GetOrCompute("Default");

990  public bool ApplyTransition(MyStringId transitionEnum, int priority = 0, MyStringId? category = null, bool loop = true)
991  {

993  return false;
994 
995  if (!m_musicAllowed)
996  return false;
997 
998  Debug.Assert(priority >= 0);
999  if (category.HasValue)
1000  {
1001  if (category.Value == MyStringId.NullOrEmpty)
1002  category = null;
1003  else if (!m_cueBank.IsValidTransitionCategory(transitionEnum, category.Value))
1004  {
1005  Debug.Fail("This category doesn't exist for this transition!");
1006  MyLog.Default.WriteLine(string.Format("Category {0} doesn't exist for this transition!", category));
1007  return false;
1008  }
1009  }
1010

1012  if ((m_currentTransition != null) &&
1013  (m_currentTransition.Value.Priority == priority) &&
1014  (m_currentTransition.Value.TransitionEnum == transitionEnum) &&
1015  ((category == null) || (m_currentTransition.Value.Category == category)))

1017  if ((m_musicState == MyMusicState.Transition) && !m_transitionForward)
1018  {
1019  m_musicState = MyMusicState.Playing;
1020  return true;
1021  }
1022  else
1023  return false;
1024  }
1025 
1026  // if category not set, we take random category from transition cues
1027  MyStringId transitionCategory = category ?? m_cueBank.GetRandomTransitionCategory(ref transitionEnum, ref NO_RANDOM);
1028  // we set this transition as next
1029  m_nextTransitions[priority] = new MyMusicTransition(priority, transitionEnum, transitionCategory);
1030 
1031  // if new transition has lower priority then current, we don't want apply new transition now
1032  if ((m_currentTransition != null) && (m_currentTransition.Value.Priority > priority))
1033  return false;
1034 
1035  m_loopMusic = loop;
1036 
1037  if (m_musicState == MyMusicState.Playing)
1038  StartTransition(true);
1039  else if (m_musicState == MyMusicState.Transition)
1040  {
1041  }
1042  else if (m_musicState == MyMusicState.Stopped)
1043  {
1044  }
1045  else
1046  throw new InvalidBranchException();
1047 
1048  return true;
1049  }
