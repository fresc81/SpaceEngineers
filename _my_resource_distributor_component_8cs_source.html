905  private float ComputeRemainingFuelTime(MyDefinitionId resourceTypeId)
906  {
907  ProfilerShort.Begin("MyResourceDistributor.ComputeRemainingFuelTime()");
908  try
909  {
910  var typeIndex = GetTypeIndex(ref resourceTypeId);
911  if (m_dataPerType[typeIndex].MaxAvailableResource == 0.0f)
912  return 0.0f;
913 
914  float powerInUse = 0.0f;
915  foreach (MySinkGroupData groupData in m_dataPerType[typeIndex].SinkDataByPriority)
916  {
917  if (groupData.RemainingAvailableResource >= groupData.RequiredInput)
918  powerInUse += groupData.RequiredInput;

920  powerInUse += groupData.RemainingAvailableResource;
921  else
922  break;
923  }

925  powerInUse += m_dataPerType[typeIndex].InputOutputData.Item1.RequiredInput;
926  else

928

930  bool hasAnyOtherProducer = false;
931  float remainingCapacity = 0.0f;
932  for (int i = 0; i < m_dataPerType[typeIndex].SourcesByPriority.Length; ++i)
933  {
934  MySourceGroupData groupData = m_dataPerType[typeIndex].SourceDataByPriority[i];
935  if (groupData.UsageRatio <= 0f)
936  continue;
937 
938  if (groupData.InfiniteCapacity)

944  }

947  foreach (MyResourceSourceComponent producer in group)
948  {
949  if (!producer.Enabled || !producer.ProductionEnabledByType(resourceTypeId))
950  continue;
951

953  remainingCapacity += producer.RemainingCapacityByType(resourceTypeId);
954  }
955  }

959  foreach (var sinkSourcePair in m_dataPerType[typeIndex].InputOutputList)

961  if (!sinkSourcePair.Item2.Enabled || !sinkSourcePair.Item2.ProductionEnabledByType(resourceTypeId))

965  remainingCapacity += sinkSourcePair.Item2.RemainingCapacityByType(resourceTypeId);
966  }
967  }
968 
969  if (hasInfiniteProducer && !hasAnyOtherProducer)
970  return float.PositiveInfinity;
971 
972  float remainingFuelTime = 0f;
973  if (powerInUse > 0f)
974  remainingFuelTime = remainingCapacity/powerInUse;
975 
976  return remainingFuelTime;
977  }

980  ProfilerShort.End();
981  }
