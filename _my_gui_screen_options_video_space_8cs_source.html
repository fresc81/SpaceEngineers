206  void OnVideoAdapterSelected()
207  {
208  int adapterIndex = (int)m_videoAdapterCombobox.GetSelectedKey();
209  { // AddDisplayModesToComboBox
210  m_resolutionCombobox.ClearItems();
211 
212  var duplicateFilter = new HashSet<Vector2I>(Vector2I.Comparer);
213  foreach (var displayMode in MyVideoSettingsManager.Adapters[adapterIndex].SupportedDisplayModes)
214  {
215  duplicateFilter.Add(new Vector2I(displayMode.Width, displayMode.Height));
216  }
217 
218  m_resolutionsForAdapter.Clear();
219  m_resolutionsForAdapter.AddHashset(duplicateFilter);
220  m_resolutionsForAdapter.Sort((a, b) =>
221  { // Sort by width, then height.
222  if (a.X != b.X)
223  return a.X.CompareTo(b.X);
224  return a.Y.CompareTo(b.Y);
225  });
226 
227  // Also show any extra display modes we have (there shouldn't be any on official builds).
228  foreach (var mode in MyVideoSettingsManager.DebugDisplayModes)
229  {
230  m_resolutionsForAdapter.Add(new Vector2I(mode.Width, mode.Height));
231  }
232 
233  var displayAspectRatio = MyVideoSettingsManager.GetRecommendedAspectRatio(adapterIndex);
234  int resolutionToSelect = 0;
235  int counter = 0;
236  for (int i = 0; i < m_resolutionsForAdapter.Count; ++i)
237  {
238  var resolution = m_resolutionsForAdapter[i];
239  float aspectRatio = (float)resolution.X / (float)resolution.Y;
240  var aspectId = MyVideoSettingsManager.GetClosestAspectRatio(aspectRatio);
241  var aspectDetails = MyVideoSettingsManager.GetAspectRatio(aspectId);
242  var aspectRatioText = aspectDetails.TextShort;
243  var starsMark = aspectDetails.IsSupported ? (aspectId == displayAspectRatio.AspectRatioEnum) ? " ***" // recommended
244  : "" // normal
245  : " *"; // unsupported
246  if (resolution.X == m_deviceSettingsOld.BackBufferWidth &&
247  resolution.Y == m_deviceSettingsOld.BackBufferHeight)
248  resolutionToSelect = counter;
249 
250  m_resolutionCombobox.AddItem(counter++, new StringBuilder(
251  string.Format("{0} x {1} - {2}{3}", resolution.X, resolution.Y, aspectRatioText, starsMark)));
252  }
253 
254  m_resolutionCombobox.SelectItemByKey(resolutionToSelect);
255  }
256 
257  { // UpdateRecommendecAspectRatioLabel
258  MyAspectRatio recommendedAspectRatio = MyVideoSettingsManager.GetRecommendedAspectRatio(adapterIndex);
259  StringBuilder sb = new StringBuilder();
260  sb.AppendFormat(MyTexts.GetString(MyCommonTexts.RecommendedAspectRatio), recommendedAspectRatio.TextShort);
261  m_recommendAspectRatioLabel.Text = string.Format("*** {0}", sb);

302 
303  int refreshRate = 0;

305  { // Pick the highest refresh rate available (although it might be better to add combobox for refresh rates as well)

308  refreshRate < displayMode.RefreshRate)
309  refreshRate = displayMode.RefreshRate;
310  }
311 
312  changed = changed || deviceSettings.RefreshRate != refreshRate;
313  deviceSettings.RefreshRate = refreshRate;
314 
315  changed = changed || deviceSettings.VSync != m_verticalSyncCheckbox.IsChecked;
316  deviceSettings.VSync = m_verticalSyncCheckbox.IsChecked;
317 
318  changed = changed || deviceSettings.WindowMode != (MyWindowModeEnum)m_windowModeCombobox.GetSelectedKey();
319  deviceSettings.WindowMode = (MyWindowModeEnum)m_windowModeCombobox.GetSelectedKey();
320 
321  var fov = MathHelper.ToRadians(m_fieldOfViewSlider.Value);
322  changed = changed || graphicsSettings.FieldOfView != fov;
323  graphicsSettings.FieldOfView = fov;
324 
325  changed = changed || graphicsSettings.HardwareCursor != m_hardwareCursorCheckbox.IsChecked;
326  graphicsSettings.HardwareCursor = m_hardwareCursorCheckbox.IsChecked;

328  changed = changed || graphicsSettings.EnableDamageEffects != m_enableDamageEffectsCheckbox.IsChecked;
