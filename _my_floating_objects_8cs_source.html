187  private void OptimizeCloseDistances()
188  {

197  float CLOSEST_DISTANCE = 4 * 4;
198  float CLOSE_DISTANCE = 16 * 16;
199  int HIGH_LIMIT = 32;
200  float NORMAL_DISTANCE = 64 * 64;
201  int NORMAL_LIMIT = 128;
202  float epsilonSq = 0.05f * 0.05f;
203  float lowerEpsilonSq = 0.0005f * 0.0005f;
204 
205  for (int i = 0; i < m_synchronizedFloatingObjects.Count; i++)
206  {
207  var syncObject = m_synchronizedFloatingObjects[i];
208 
209  m_needReupdateNewObjects |= syncObject.ClosestDistanceToAnyPlayerSquared == -1 || (syncObject.ClosestDistanceToAnyPlayerSquared < CLOSEST_DISTANCE && syncObject.SyncWaitCounter > 5);
210  var linearVelocitySq = syncObject.Physics.LinearVelocity.LengthSquared();
211  var angularVelocitySq = syncObject.Physics.AngularVelocity.LengthSquared();
212  if (syncObject.ClosestDistanceToAnyPlayerSquared == -1 ||
213  linearVelocitySq > epsilonSq ||
214  angularVelocitySq > epsilonSq)

216  if ((syncObject.ClosestDistanceToAnyPlayerSquared < CLOSE_DISTANCE) && (i < HIGH_LIMIT))
217  m_highPriority.Add(syncObject);
