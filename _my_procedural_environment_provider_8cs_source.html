119  public unsafe MyEnvironmentDataView GetItemView(int lod, ref Vector2I start, ref Vector2I end, ref Vector3D localOrigin)
120  {
121  var localLod = lod / LodFactor;
122  var logicalLod = lod % LodFactor;
123 
124  start >>= (localLod * LodFactor);
125  end >>= (localLod * LodFactor);
126 
127  MyProceduralDataView view = new MyProceduralDataView(this, lod, ref start, ref end);
128 
129  var lcount = (end - start + 1).Size();
130 
131  view.SectorOffsets = new List<int>(lcount);
132  view.LogicalSectors = new List<MyLogicalEnvironmentSectorBase>(lcount);
133  view.IntraSectorOffsets = new List<int>(lcount);
134 
135  // First round, calculate offsets and find any missing sectors.
136  int offset = 0;
137  for (int y = start.Y; y <= end.Y; y++)
138  for (int x = start.X; x <= end.X; x++)
139  {
140  var sector = GetLogicalSector(x, y, localLod);
141 
142  if (sector.MinimumScannedLod != logicalLod)
143  sector.ScanItems(logicalLod);
144 
145  sector.Viewers.Add(view);
146  sector.UpdateMinLod();
147 
148  view.SectorOffsets.Add(offset);
149  view.LogicalSectors.Add(sector);
150  offset += sector.ItemCountForLod[logicalLod];
151  view.IntraSectorOffsets.Add(0);
152  }
153 
154  // Allocate item list.
155  view.Items = new List<ItemInfo>(offset);
156 
157  int offsetIndex = 0;
158  for (int y = start.Y; y <= end.Y; y++)
159  for (int x = start.X; x <= end.X; x++)
160  {
161  MyProceduralLogicalSector sector = m_sectors[MyPlanetSectorId.MakeSectorId(x, y, ProviderId, localLod)];
162 
163  int itemCnt = sector.ItemCountForLod[logicalLod];
164 
165  offset = view.SectorOffsets[offsetIndex++];
166 
167  Vector3 centerOffset = sector.WorldPos - localOrigin;
168 
169  fixed (ItemInfo* viewItems = view.Items.GetInternalArray())
170  fixed (ItemInfo* sectorItems = sector.Items.GetInternalArray())
171  for (int i = 0; i < itemCnt; ++i)
172  {
173  int vi = i + offset; // view index
174 
175  viewItems[vi].Position = sectorItems[i].Position + centerOffset;
176 
177  // TODO: Memcpy?
178  viewItems[vi].DefinitionIndex = sectorItems[i].DefinitionIndex;
179  viewItems[vi].ModelIndex = sectorItems[i].ModelIndex;
180  viewItems[vi].Rotation = sectorItems[i].Rotation;
181  }
182  }
183 
184  view.Items.SetSize(view.Items.Capacity);
185 
186  if ((m_sectorsToRaise.Count > 0 || m_sectorsToRaise.Count > 0) && !m_sectorsQueued)
187  {
188  m_sectorsQueued = true;
189 
190  Parallel.ScheduleForThread(m_raiseCallback, null, MySandboxGame.Static.UpdateThread);
191  }
192  return view;
193  }
