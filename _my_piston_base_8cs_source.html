740  protected override bool CanPlaceTop(MyAttachableTopBlockBase topBlock, long builtBy)
741  {
742  // Compute the rough actual position for the head, this improves the detection if it can be placed
743  float topDistance = (Subpart3.Model.BoundingBoxSize.Y);
744  Vector3D topPosition = this.Subpart3.WorldMatrix.Translation + this.WorldMatrix.Up * topDistance;
745  float topRadius = topBlock.ModelCollision.HavokCollisionShapes[0].ConvexRadius * 0.9f;
746 
747  // First test if we intersect any blocks of our own grid
748  BoundingSphereD sphere = topBlock.Model.BoundingSphere;
749  sphere.Center = topPosition;
750  sphere.Radius = topRadius;
751  CubeGrid.GetBlocksInsideSphere(ref sphere, m_tmpSet);
752 
753  // If we intersect more than 1 block (because top sometimes intersects piston), don't add top
754  if (m_tmpSet.Count > 1)
755  {
756  m_tmpSet.Clear();
757  if (builtBy == MySession.Static.LocalPlayerId)
758  MyHud.Notifications.Add(MyNotificationSingletons.HeadNotPlaced);
759  return false;
760  }
761 
762  m_tmpSet.Clear();
763 
764  // Next test if we intersect any physics objects
765  HkSphereShape spShape = new HkSphereShape(topRadius);
766  Quaternion q = Quaternion.Identity;
767  MyPhysics.GetPenetrationsShape(topBlock.ModelCollision.HavokCollisionShapes[0], ref topPosition, ref q, m_penetrations, MyPhysics.CollisionLayers.DefaultCollisionLayer);
768 
769  // If we have any collisions with anything other than our own grid, don't add the head
770  // We already checked for inner-grid collisions in the previous case
771  for (int i = 0; i < m_penetrations.Count; i++)
772  {
773  MyCubeGrid grid = m_penetrations[i].GetCollisionEntity().GetTopMostParent() as MyCubeGrid;
774  if (grid == null || grid != CubeGrid)
775  {
776  m_penetrations.Clear();
777  if (builtBy == MySession.Static.LocalPlayerId)
778  MyHud.Notifications.Add(MyNotificationSingletons.HeadNotPlaced);
779  return false;
780  }
781  }
782 
783  m_penetrations.Clear();
784 
785  return true;
