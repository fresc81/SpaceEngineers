168  {
169  if (TextureState == FileTextureState.Loaded)
170  return;
171

173  Debug.Assert(m_srv == null, "Texture " + Name + " in invalid state");
174 
175  Image img = null;
176 
177  if (MyFileSystem.FileExists(m_path))
178  {
179  try
180  {
181  using (var s = MyFileSystem.OpenRead(m_path))
182  {
183  img = Image.Load(s);
184  ImageFormatInFile = img.Description.Format;
185  }
186  }
187  catch (Exception e)
188  {
189  MyRender11.Log.WriteLine("Error while loading texture: " + m_path + ", exception: " + e);
190  }
191  }
192 
193  bool loaded = false;
194  if (img != null)
195  {
196  int skipMipmaps = 0;
197  if (m_type != MyFileTextureEnum.GUI && m_type != MyFileTextureEnum.GPUPARTICLES && img.Description.MipLevels > 1)
198  skipMipmaps = MyRender11.Settings.User.TextureQuality.MipmapsToSkip(img.Description.Width, img.Description.Height);
199 
200  if (m_skipQualityReduction)
201  skipMipmaps = 0;
202 
203  int totalSize = 0;
204 
205  int targetMipmaps = img.Description.MipLevels - skipMipmaps;
206  var mipmapsData = new DataBox[(img.Description.MipLevels - skipMipmaps) * img.Description.ArraySize];
207  if (img.Description.MipLevels <= 1)
208  Debug.WriteLine(Name);
209 
210  int lastSize = 0;
211 
212  for (int z = 0; z < img.Description.ArraySize; z++)
213  {
214  for (int i = 0; i < targetMipmaps; i++)
215  {
216  var pixels = img.GetPixelBuffer(z, i + skipMipmaps);
217  mipmapsData[Resource.CalculateSubResourceIndex(i, z, targetMipmaps)] =
218  new DataBox { DataPointer = pixels.DataPointer, RowPitch = pixels.RowStride };
219 
220  void* data = pixels.DataPointer.ToPointer();
221 
222  lastSize = pixels.BufferStride;
223  totalSize += lastSize;
224  }
225  }
226 
227  var targetWidth = img.Description.Width >> skipMipmaps;
228  var targetHeight = img.Description.Height >> skipMipmaps;
229 
230 
231  bool overwriteFormatToSrgb = false;
232  if (MyCompilationSymbols.ReinterpretFormatsStoredInFiles)
233  overwriteFormatToSrgb = (m_type != MyFileTextureEnum.NORMALMAP_GLOSS) && !FormatHelper.IsSRgb(img.Description.Format);
234 
235  m_format = overwriteFormatToSrgb ? MyResourceUtils.MakeSrgb(img.Description.Format) : img.Description.Format;
236 
237  var desc = new Texture2DDescription
238  {
239  MipLevels = targetMipmaps,
240  Format = m_format,
241  Height = targetHeight,
242  Width = targetWidth,
243  ArraySize = img.Description.ArraySize,
244  BindFlags = BindFlags.ShaderResource,
245  CpuAccessFlags = CpuAccessFlags.None,
246  Usage = ResourceUsage.Immutable,
247  SampleDescription = new SampleDescription { Count = 1, Quality = 0 },
248  OptionFlags =
249  img.Description.Dimension == TextureDimension.TextureCube
250  ? ResourceOptionFlags.TextureCube
251  : ResourceOptionFlags.None
252  };
253 
254  try
255  {
256  m_resource = new Texture2D(MyRender11.Device, desc, mipmapsData);
257  m_size = new Vector2I(targetWidth, targetHeight);
258  //m_skippedMipmaps = skipMipmaps;
259  m_byteSize = totalSize;
260  m_ownsData = true;
261 
262  m_srv = new ShaderResourceView(MyRender11.Device, m_resource);
263  m_resource.DebugName = m_name;
264  m_srv.DebugName = m_name;
265 
266  img.Dispose();
267 
268  MipmapCount = targetMipmaps;
269 
270  loaded = true;
271  }
272  catch (SharpDXException)
273  {
274  img.Dispose();
275  }
276  }
277  if (!loaded)
278  {
279  ISrvBindable replacingTexture = MyGeneratedTextureManager.PinkTex;
280  switch (m_type)

282  case MyFileTextureEnum.NORMALMAP_GLOSS:

284  break;
285  case MyFileTextureEnum.EXTENSIONS:
286  replacingTexture = MyGeneratedTextureManager.MissingExtensionTex;
287  break;
288  case MyFileTextureEnum.ALPHAMASK:
289  replacingTexture = MyGeneratedTextureManager.MissingAlphamaskTex;
290  break;
291  case MyFileTextureEnum.CUBEMAP:
292  replacingTexture = MyGeneratedTextureManager.MissingCubeTex;
293  break;
294  }
295 
296  MyRender11.Log.WriteLine("Could not load texture: " + m_path);
297 
298  m_srv = replacingTexture.Srv;
299  m_resource = replacingTexture.Resource;
300  m_size = replacingTexture.Size;
301  m_ownsData = false;
302  m_byteSize = 0;
303  MyRender11.Log.WriteLine("Missing or invalid texture: " + Name);
304  }

396  public ITexture GetTexture(string name, MyFileTextureEnum type, bool waitTillLoaded = false, bool skipQualityReduction = false,
397  bool temporary = false)
398  {
399  if (name == null || name.Length <= 0)
400  return ReturnDefaultTexture(type);
401 
402  Uri uri;
403  if (!MyResourceUtils.NormalizeFileTextureName(ref name, out uri))
404  {
405  IGeneratedTexture texture;
406  if (m_generatedTextures.TryGetValue(name, out texture))
407  return texture;
408  else
409  {
410  MyRenderProxy.Assert(false, "Can't find generated texture with name \"" + name + "\"");
411  return ReturnDefaultTexture(type);
412  }
413  }
414 
415  MyFileTexture texOut;
416  if (!m_textures.TryGetValue(name, out texOut))
417  {
418  if (uri.Scheme != FILE_SCHEME)
419  {
420  Debug.Assert(false, "Cannot initialize a non file texture");
421  return ReturnDefaultTexture(type);
422  }
423 
424  m_texturesPool.AllocateOrCreate(out texOut);
425  texOut.Init(name, uri.LocalPath, type, waitTillLoaded, skipQualityReduction, temporary);
426  m_textures.Add(name, texOut);
427  }
428 
429  switch (texOut.TextureState)
430  {
431  case FileTextureState.Unloaded:
432  case FileTextureState.Requested:
433  if (waitTillLoaded)
434  LoadInternal(name);
435  else
436  {
437  texOut.TextureState = FileTextureState.Requested;
438  m_requestedTextures.Add(name);
439  }
440  break;
441  case FileTextureState.Loaded:
442  break;
443  }
444 
445  texOut.Temporary &= temporary;
446  m_temporaryTextureRequested |= temporary;
447 
448  return texOut;
