791  internal static ConstructorInfo ResolveTupleConstructor(Type type, out MemberInfo[] mappedMembers)
792  {
793  mappedMembers = null;
794  if(type == null) throw new ArgumentNullException("type");
795 #if WINRT
796  TypeInfo typeInfo = type.GetTypeInfo();
797  if (typeInfo.IsAbstract) return null; // as if!
798  ConstructorInfo[] ctors = Helpers.GetConstructors(typeInfo, false);
799 #else
800  if(type.IsAbstract) return null; // as if!
801  ConstructorInfo[] ctors = Helpers.GetConstructors(type, false);
802 #endif
803  // need to have an interesting constructor to bother even checking this stuff
804  if(ctors.Length == 0 || (ctors.Length == 1 && ctors[0].GetParameters().Length == 0)) return null;
805 
806  MemberInfo[] fieldsPropsUnfiltered = Helpers.GetInstanceFieldsAndProperties(type, true);
807  BasicList memberList = new BasicList();
808  for (int i = 0; i < fieldsPropsUnfiltered.Length; i++)
809  {
810  PropertyInfo prop = fieldsPropsUnfiltered[i] as PropertyInfo;
811  if (prop != null)
812  {
813  if (!prop.CanRead) return null; // no use if can't read
814  if (prop.CanWrite && Helpers.GetSetMethod(prop, false, false) != null) return null; // don't allow a public set (need to allow non-public to handle Mono's KeyValuePair<,>)
815  memberList.Add(prop);
816  }
817  else
818  {
819  FieldInfo field = fieldsPropsUnfiltered[i] as FieldInfo;
820  if (field != null)
821  {
822  if (!field.IsInitOnly) return null; // all public fields must be readonly to be counted a tuple
823  memberList.Add(field);
824  }
825  }
826  }
827  if (memberList.Count == 0)
828  {
829  return null;
830  }
831 
832  MemberInfo[] members = new MemberInfo[memberList.Count];
833  memberList.CopyTo(members, 0);
834 
835  int[] mapping = new int[members.Length];
836  int found = 0;
837  ConstructorInfo result = null;
838  mappedMembers = new MemberInfo[mapping.Length];
839  for(int i = 0 ; i < ctors.Length ; i++)
840  {
841  ParameterInfo[] parameters = ctors[i].GetParameters();
842 
843  if (parameters.Length != members.Length) continue;
844 
845  // reset the mappings to test
846  for (int j = 0; j < mapping.Length; j++) mapping[j] = -1;
847 
848  for(int j = 0 ; j < parameters.Length ; j++)
849  {
850  string lower = parameters[j].Name.ToLower();
851  for(int k = 0 ; k < members.Length ; k++)
852  {
853  if (members[k].Name.ToLower() != lower) continue;
854  Type memberType = Helpers.GetMemberType(members[k]);
855  if (memberType != parameters[j].ParameterType) continue;
856 
857  mapping[j] = k;
858  }
859  }
860  // did we map all?
861  bool notMapped = false;
862  for (int j = 0; j < mapping.Length; j++)
863  {
864  if (mapping[j] < 0)
865  {
866  notMapped = true;
867  break;
868  }
869  mappedMembers[j] = members[mapping[j]];
870  }
871 
872  if (notMapped) continue;
873  found++;
874  result = ctors[i];
875 
876  }
877  return found == 1 ? result : null;
878  }
