310  float GetAlphaForTime(float transitionTime, bool isStartLod)
311  {
312  if (transitionTime == 0 || Math.Abs(transitionTime) >= 1.0f)
313  return 0;
314 
315  // Value over 1 is interpreted by the shader to do dithering with an inversed mask
316  // This is done so that when blending between two lod levels, one pixel will be from current lod
317  // and the other from the next lod and there are no missing pixels.
318  // Could not use negative because that currently means hologram rendering.
319  // SL: Moved inversed dithering to values of 2 to 3 instead of 1 to 2 to make sure the values don't overlap
320  return isStartLod ? Math.Abs(transitionTime) : (3.0f - Math.Abs(transitionTime));
321  }
