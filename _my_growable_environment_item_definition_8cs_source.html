278  // We need to be able to find the last index of steps based on their next step index

282  }

284  // --------------
285  // Step 1: find the cycle:
286  HashSet<int> visited = new HashSet<int>();
287 
288  // Iterate over the elements and mark them untill we go around the cycle or reach the root.
289  var current = elements[0];
290  visited.Add(current);
291  while ((current = m_steps[current].NextStep) != -1)
292  {
293  if (visited.Contains(current)) break;
294  visited.Add(current);
295  }
296 
297  if (current != -1)

300  lineage.Add(new Step(current, m_steps[current].TransitionTime));
301  m_steps[current].SubtreeId = 0;
302

304  int next = current;
305  for (int i = 1; (next = m_steps[next].NextStep) != current; ++i)
306  {
307  lineage.Add(new Step(next, m_steps[next].TransitionTime));
308  m_steps[next].SubtreeId = i;
309  }
310 
311  Cycles = true;
312 
313  lineages.Add(lineage.ToArray());
314  lineage.Clear();
315  }
316 
317  // --------------
318  // Step 2: Determine the start and record each other lineage:
319

321

323  for (int i = 0; i < elements.Count; i++)
324  {

326  if (next != -1)

385  public void UpdateStep(ref int globalStepIndex, ref long enlapsedTime)
386  {
387  Step key;
388  int stepIndex;
389 
390  var originalTime = enlapsedTime;
391 
392  int subtreeId = m_parent.GrowthSteps[globalStepIndex].SubtreeId;
393 
394  int lineageIndex = (subtreeId >> 16) & 0xFFFF;
395  var lineage = Lineages[lineageIndex];
396 
397  int step = subtreeId & 0xFFFF;
398 
399  // do it as if from the start
400  if (step != 0)
401  enlapsedTime += lineage[step - 1].CumulativeTime;
402 
403  // Calculate position in the peripheric lineage
404  if (lineageIndex != 0 && Cycles)
405  {
406  key = new Step(-1, enlapsedTime);
407  stepIndex = lineage.BinaryIntervalSearch(key);
408 
409  if (stepIndex < lineage.Length)
410  {
411  // If we do not leave this lineage we are done.
412  globalStepIndex = lineage[stepIndex].Index;
413  goto calcNextTime;
414  }
415  else
416  {
417  var last = lineage[lineage.Length - 1];
418  // Remove any calculated time and move on to the cycle.
419  enlapsedTime -= last.CumulativeTime;
420 
421  // As if we always were in that lineage.
422  subtreeId = m_steps[m_steps[last.Index].NextStep].SubtreeId;
423 
424  lineage = Lineages[0];
425 
426  step = subtreeId & 0xFFFF;
427  if (step != 0)
428  enlapsedTime += lineage[step - 1].CumulativeTime;
429  }
430  }
431 
432  // Calculate position in the cycle.
433  if (Cycles)
434  enlapsedTime %= CycleTime;
435 
436  key = new Step(-1, enlapsedTime);
437  stepIndex = Math.Min(lineage.BinaryIntervalSearch(key), lineage.Length - 1);

441  if (m_steps[globalStepIndex].NextStep != -1)
442  {

444  enlapsedTime = originalTime - diff + m_steps[globalStepIndex].TransitionTime;

447  }
