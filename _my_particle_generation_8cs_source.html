800  private void UpdateParticlesCreation()
801  {
802  if (!Enabled.GetValue<bool>() || !m_show)
803  return;
804 
805  if (m_effect.CalculateDeltaMatrix == false)
806  {
807  float motionInheritance;
808  MotionInheritance.GetInterpolatedValue<float>(m_effect.GetElapsedTime(), out motionInheritance);
809  if (motionInheritance > 0)
810  m_effect.CalculateDeltaMatrix = true;
811  }
812 
813  //particles to create in this update
814  if (!m_effect.IsEmittingStopped)
815  {
816  float lodBirth = 1.0f;
817 
818  if (GetEffect().EnableLods && LODBirth.GetKeysCount() > 0)
819  {
820  LODBirth.GetInterpolatedValue<float>(GetEffect().Distance, out lodBirth);
821  }
822 
823  Birth.GetInterpolatedValue<float>(m_effect.GetElapsedTime(), out m_birthRate);
824  m_birthRate *=
825  VRage.Game.MyEngineConstants.UPDATE_STEP_SIZE_IN_SECONDS
826  * (EnableCustomBirth ? m_effect.UserBirthMultiplier : 1.0f)
827  * lodBirth;
828 
829  if (BirthPerFrame.GetKeysCount() > 0)
830  {
831  float keyTime, diff;
832  BirthPerFrame.GetNextValue(m_effect.GetElapsedTime() - VRage.Game.MyEngineConstants.UPDATE_STEP_SIZE_IN_SECONDS, out m_birthPerFrame, out keyTime, out diff);
833  if (
834  (keyTime >= (m_effect.GetElapsedTime() - VRage.Game.MyEngineConstants.UPDATE_STEP_SIZE_IN_SECONDS)) &&
835  (keyTime < m_effect.GetElapsedTime())
836  )
837  {
838  m_birthPerFrame *= (EnableCustomBirth ? m_effect.UserBirthMultiplier : 1.0f)
839  * lodBirth;
840  }
841  else
842  m_birthPerFrame = 0;
843  }
844 
845  m_particlesToCreate += m_birthRate;
846  }
847 
848  //If speed of effect is too high, there would be created bunches
849  //of particles each frame. By interpolating position, we create
850  //seamless particle creation.
851  Vector3 positionDelta = Vector3.Zero;
852 
853  if (!m_lastEffectPosition.HasValue)
854  m_lastEffectPosition = EffectMatrix.Translation;
855 
856  //Position delta interpolates particle position at fast flying objects, dont do that while motion inheritance
857  if (m_particlesToCreate > 1.0f && !m_effect.CalculateDeltaMatrix)
858  {
859  positionDelta = (EffectMatrix.Translation - m_lastEffectPosition.Value) / (int)m_particlesToCreate;
860  }
861 
862  //VRageRender.MyRenderProxy.GetRenderProfiler().StartNextBlock("CreateParticle");
863 
864 
865  int maxParticles = 40;
866  while (m_particlesToCreate >= 1.0f && maxParticles-- > 0)
867  {
868  if (m_effect.CalculateDeltaMatrix)
869  CreateParticle(EffectMatrix.Translation);
870  else
871  CreateParticle(m_lastEffectPosition.Value + positionDelta * (int)m_particlesToCreate);
872 
873  m_particlesToCreate -= 1.0f;
874  }
875 
876  while (m_birthPerFrame >= 1.0f && maxParticles-- > 0)
877  {
878  if (m_effect.CalculateDeltaMatrix)
879  CreateParticle(EffectMatrix.Translation);
880  else
881  CreateParticle(m_lastEffectPosition.Value + positionDelta * (int)m_birthPerFrame);
882 
883  m_birthPerFrame -= 1.0f;
884  }
885 
886  // VRageRender.MyRenderProxy.GetRenderProfiler().StartNextBlock("OnLife");
887 
888  if (OnLife.GetValue<int>() != -1)
889  {
890  MyParticleGeneration inheritedGeneration = GetInheritedGeneration(OnLife.GetValue<int>()) as MyParticleGeneration;
891 
892  if (inheritedGeneration == null)
893  {
894  OnLife.SetValue(-1);
895  }
896  else
897  {
898  inheritedGeneration.IsInherited = true;
899 
900  float particlesToCreate = inheritedGeneration.m_particlesToCreate;
901 
902  foreach (MyAnimatedParticle particle in m_particles)
903  {
904  inheritedGeneration.m_particlesToCreate = particlesToCreate;
905  inheritedGeneration.EffectMatrix = MatrixD.CreateWorld(particle.ActualPosition, (Vector3D)particle.Velocity, Vector3D.Cross(Vector3D.Left, particle.Velocity));
906  inheritedGeneration.UpdateParticlesCreation();
907  }
908  }
909  }
910 
911  // VRageRender.MyRenderProxy.GetRenderProfiler().EndProfilingBlock();
912 
913  m_lastEffectPosition = EffectMatrix.Translation;
914  }
