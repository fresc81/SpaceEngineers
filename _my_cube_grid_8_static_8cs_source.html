692  {
693  ProfilerShort.Begin("MyCubeBuilder.CheckMountPoints");
694  try
695  {
696  if (mountPoints == null)
697  return false;
698 
699  var center = def.Center;
700  var size = def.Size;
701  Vector3I rotatedSize;
702  Vector3I rotatedCenter;
703  Vector3I.Transform(ref center, ref rotation, out rotatedCenter);
704  Vector3I.Transform(ref size, ref rotation, out rotatedSize);
705 
706  for (int i = 0; i < mountPoints.Length; ++i)
707  {
708  var thisMountPoint = mountPoints[i];
709  Vector3 centeredStart = thisMountPoint.Start - center;
710  Vector3 centeredEnd = thisMountPoint.End - center;
711  if (MyFakes.ENABLE_TEST_BLOCK_CONNECTIVITY_CHECK)
712  {
713  // This code is used to avoid mixed start, end values, overlapped values on sides. So as result we need precise neighbour(s). Not neighbours touched on edges.
714 
715  // Start and end sometimes have exchanged values
716  Vector3 start = Vector3.Min(thisMountPoint.Start, thisMountPoint.End);
717  Vector3 end = Vector3.Max(thisMountPoint.Start, thisMountPoint.End);
718 
719  // Clamp only overlapped values on sides not thickness of aabb of mount point
720  Vector3I clampMask = Vector3I.One - Vector3I.Abs(thisMountPoint.Normal);
721  Vector3I invClampMask = Vector3I.One - clampMask;
722  Vector3 clampedStart = invClampMask * start + Vector3.Clamp(start, Vector3.Zero, size) * clampMask + 0.001f * clampMask;
723  Vector3 clampedEnd = invClampMask * end + Vector3.Clamp(end, Vector3.Zero, size) * clampMask - 0.001f * clampMask;
724 
725  centeredStart = clampedStart - center;
726  centeredEnd = clampedEnd - center;
727  }
728 
729  var centeredStartI = Vector3I.Floor(centeredStart);
730  var centeredEndI = Vector3I.Floor(centeredEnd);

741  // transforming floats, we can end up rotating eg. 0.5f to -0.5f which, after Floor operation, would be like rotation of 0 to -1 (ie. wrong).
742  // By rotating both floored and floating point versions, I can find out what change occured and handle it accordingly.
743  var rotatedStartI = Vector3I.Floor(rotatedStart);
744  var rotatedEndI = Vector3I.Floor(rotatedEnd);
745  var correctionStart = rotatedStartICorrect - rotatedStartI;
746  var correctionEnd = rotatedEndICorrect - rotatedEndI;

791  if (CheckNeighborMountPointsForCompound(currentMin, currentMax, thisMountPoint, ref transformedNormal, def,
