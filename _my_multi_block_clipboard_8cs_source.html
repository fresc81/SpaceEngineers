692  private MyCubeGrid DetectTouchingGrid(MySlimBlock block)
693  {

695  return null;
696 
697  if (block == null)
698  return null;
699 
700  if (block.FatBlock is MyCompoundCubeBlock)

713 
714  float gridSize = block.CubeGrid.GridSize;
715  BoundingBoxD aabb;
716  block.GetWorldBoundingBox(out aabb);
717  // Inflate by half cube, so it will intersect for sure when there's anything
718  aabb.Inflate(gridSize / 2);
719 
720  m_tmpNearEntities.Clear();
721  MyEntities.GetElementsInBox(ref aabb, m_tmpNearEntities);

724

728  {

730  if (grid != null && grid != block.CubeGrid && grid.Physics != null && grid.Physics.Enabled && grid.IsStatic && grid.GridSizeEnum == block.CubeGrid.GridSizeEnum)
731  {
732  Vector3I gridOffset = grid.WorldToGridInteger(m_pastePosition);
733  if (!grid.CanMergeCubes(block.CubeGrid, gridOffset))
734  continue;
735 
736  MatrixI transform = grid.CalculateMergeTransform(block.CubeGrid, gridOffset);
737  Base6Directions.Direction forward = transform.GetDirection(block.Orientation.Forward);
738  Base6Directions.Direction up = transform.GetDirection(block.Orientation.Up);
739  MyBlockOrientation newOrientation = new MyBlockOrientation(forward, up);

741  newOrientation.GetQuaternion(out newRotation);
742  Vector3I newPosition = Vector3I.Transform(block.Position, transform);
743 
744  if (!MyCubeGrid.CheckConnectivity(grid, block.BlockDefinition, mountPoints, ref newRotation, ref newPosition))
745  continue;
746 
747  return grid;
748  }
749  }
750  }
