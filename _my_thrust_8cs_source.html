505  continue;

574  public LineD GetDamageCapsuleLine(FlameInfo info)
575  {
576  var world = Matrix.CreateFromDir(Vector3.TransformNormal(info.Direction, WorldMatrix));
577 
578  var halfLenght = ThrustLengthRand * info.Radius / 2;
579  halfLenght *= BlockDefinition.FlameDamageLengthScale;
580  var position = Vector3D.Transform(info.Position, WorldMatrix);
581 
582  if (halfLenght > info.Radius)
583  return new LineD(position - world.Forward * (info.Radius), position + world.Forward * (2 * halfLenght - info.Radius));
584  else
