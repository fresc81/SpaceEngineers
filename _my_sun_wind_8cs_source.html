719  private static void ComputeMaxDistance(MySunWindBillboardSmall billboard)

722  var offset = (-m_directionFromSunNormalized * MySunWindConstants.RAY_CAST_DISTANCE);

724  // If intersection found, we will mark that place as small billboard's destination. It can't go further.

726  VRage.Game.Models.MyIntersectionResultLineTriangleEx? intersection = MyEntities.GetIntersectionWithLine(ref line, null, null);
727  if (intersection != null)
728  billboard.MaxDistance = (float)(intersection.Value.Triangle.Distance - billboard.Radius);
