179  // This formulas works even if FPS is low or high, or if step size is 1/10 or 1/10000
180  float amountOfMovement = VRage.Game.MyEngineConstants.UPDATE_STEP_SIZE_IN_SECONDS * 100;
181  float amountOfRotation = 0.0025f * m_speedModeAngular;

187  {
188  Vector3D r, u;
189  rollAmount = rollIndicator * m_speedModeAngular * 0.1f;
190  rollAmount = MathHelper.Clamp(rollAmount, -0.02f, 0.02f);
191  MyUtils.VectorPlaneRotation(m_orientation.Up, m_orientation.Right, out u, out r, rollAmount);
192  m_orientation.Right = r;
193  m_orientation.Up = u;
194  }
195 
196  Vector3 moveVector;
197 
198  {
199  if (AlignSpectatorToGravity)

201  rotationIndicator.Rotate(m_roll); // TODO: ROLL

208  m_pitch = MathHelper.Clamp(m_pitch, -Math.PI * 0.5f, Math.PI * 0.5f);

255  private void ComputeGravityAlignedOrientation(out MatrixD resultOrientationStorage)
256  {
257  // Y axis
258  bool inGravityField = true;
259  Vector3D upVector = -MyGravityProviderSystem.CalculateTotalGravityInPoint(Position);
260  if (upVector.LengthSquared() < MyMathConstants.EPSILON)
261  {
262  upVector = m_lastUpVec;
263  m_lastOrientationWeight = 1;
264  inGravityField = false;
265  }
266  else
267  {
268  m_lastUpVec = upVector;
269  }
270  upVector.Normalize();
271  // X axis
272  Vector3D rightVector = m_lastRightVec - Vector3D.Dot(m_lastRightVec, upVector) * upVector;
273  if (rightVector.LengthSquared() < MyMathConstants.EPSILON)
274  {
275  rightVector = m_orientation.Right - Vector3D.Dot(m_orientation.Right, upVector) * upVector;
276  // backup behavior if singularity happens
277  if (rightVector.LengthSquared() < MyMathConstants.EPSILON)
278  rightVector = m_orientation.Forward - Vector3D.Dot(m_orientation.Forward, upVector) * upVector;
279  // backup behavior if singularity happens
280  }
281  rightVector.Normalize();
282  m_lastRightVec = rightVector;
283  // Z axis
284  Vector3D forwardVector;
285  Vector3D.Cross(ref upVector, ref rightVector, out forwardVector);
286 
287  resultOrientationStorage = MatrixD.Identity;
288  resultOrientationStorage.Right = rightVector;
289  resultOrientationStorage.Up = upVector;
290  resultOrientationStorage.Forward = forwardVector;
291  resultOrientationStorage = MatrixD.CreateFromAxisAngle(Vector3D.Right, m_pitch) * resultOrientationStorage *
292  MatrixD.CreateFromAxisAngle(upVector, m_yaw);
293  upVector = resultOrientationStorage.Up;
294  rightVector = resultOrientationStorage.Right;
295  resultOrientationStorage.Right = Math.Cos(m_roll) * rightVector + Math.Sin(m_roll) * upVector;
296  resultOrientationStorage.Up = -Math.Sin(m_roll) * rightVector + Math.Cos(m_roll) * upVector;
297 
298  if (inGravityField && m_lastOrientationWeight > 0)
299  {
300  m_lastOrientationWeight = Math.Max(0, m_lastOrientationWeight - MyEngineConstants.UPDATE_STEP_SIZE_IN_SECONDS);
301 
302  resultOrientationStorage = MatrixD.Slerp(resultOrientationStorage, m_lastOrientation,
303  MathHelper.SmoothStepStable(m_lastOrientationWeight));
304  resultOrientationStorage = MatrixD.Orthogonalize(resultOrientationStorage);
305  resultOrientationStorage.Forward = Vector3D.Cross(resultOrientationStorage.Up, resultOrientationStorage.Right);
306  }
307  if (!inGravityField)
308  {
309  m_lastOrientation = resultOrientationStorage;
310  }
311  }
