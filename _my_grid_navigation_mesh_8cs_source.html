364  private bool IsFaceTriangle(MyNavigationTriangle triangle, Vector3I cubePosition, Vector3I direction)
365  {
366  var e = triangle.GetVertexEnumerator();
367 
368  // Quantize the point positions to 1/256ths inside each cube (but multiplied by 256 :-) )
369  e.MoveNext();
370  Vector3I positionA = Vector3I.Round(e.Current * 256);
371  e.MoveNext();
372  Vector3I positionB = Vector3I.Round(e.Current * 256);
373  e.MoveNext();
374  Vector3I positionC = Vector3I.Round(e.Current * 256);
375  cubePosition = cubePosition * 256;
376 
377  // Change the point coordinates to be relative to the face center
378  Vector3I faceCenter = cubePosition + direction * 128;
379  positionA -= faceCenter;
380  positionB -= faceCenter;
381  positionC -= faceCenter;
382 
383  // Discard all triangles whose points are not on the plane of the face
384  if (positionA * direction != Vector3I.Zero) return false;
385  if (positionB * direction != Vector3I.Zero) return false;
386  if (positionC * direction != Vector3I.Zero) return false;

389  return (positionA.AbsMax() <= 128 && positionB.AbsMax() <= 128 && positionC.AbsMax() <= 128);

562

576  vertices.MoveNext(); Vector3 v2 = vertices.Current;
577  vertices.MoveNext(); Vector3 v3 = vertices.Current;
578 
579  Vector3 center = (v1 + v2 + v3) / 3.0f;
580

582  Vector3 v23 = (v3 - v2);

589  if (dsq < (v12.LengthSquared() + v23.LengthSquared()))

592 
593  Vector3 triN = Vector3.Cross(v12, v23);
594  triN.Normalize();
595

597  v23 = Vector3.Cross(v23, triN);

601  float d2 = -Vector3.Dot(v23, v2);
602  float d3 = -Vector3.Dot(v31, v3);

605  float p2 = Vector3.Dot(v23, point) + d2;
606  float p3 = Vector3.Dot(v31, point) + d3;
607 
608  dsq = Vector3.Dot(triN, point) - Vector3.Dot(triN, center);
609  dsq = dsq * dsq;
610  if (p1 > 0)
611  {
612  if (p2 > 0)
613  {
614  if (p3 < 0)
615  dsq += p3 * p3;
616  }
617  else
618  {
619  if (p3 > 0)
620  dsq += p2 * p2;
621  else
622  dsq += Vector3.DistanceSquared(v3, point);
623  }
624  }
625  else
626  {
627  if (p2 > 0)
628  {
629  if (p3 > 0)
630  dsq += p1 * p1;
631  else
632  dsq += Vector3.DistanceSquared(v1, point);
633  }
634  else
635  {
636  if (p3 > 0)
637  dsq += Vector3.DistanceSquared(v2, point);
638  }
639  }
640  }
