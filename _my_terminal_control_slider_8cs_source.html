149  private static float DualLogDenormalizer(TBlock block, float value, float min, float max, float centerBand)
150  {
151  float shiftedVal = value * 2.0f - 1.0f;
152  if (Math.Abs(shiftedVal) < centerBand) return 0.0f;
153  float scaledVal = (Math.Abs(shiftedVal) - centerBand) / (1.0f - centerBand);
154  return MathHelper.Clamp(MathHelper.InterpLog(scaledVal, min, max), min, max) * Math.Sign(shiftedVal);
155  }

157  private static float DualLogNormalizer(TBlock block, float value, float min, float max, float centerBand)
158  {
159  if (Math.Abs(value) < min) return 0.5f;
160  float width = 0.5f - centerBand / 2.0f;
161  float result = MathHelper.Clamp(MathHelper.InterpLogInv(Math.Abs(value), min, max), 0, 1) * width;
162  if (value < 0.0f) result = width - result;
163  else result = result + width + centerBand;
164  return result;
165  }
