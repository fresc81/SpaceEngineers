811  public static void PrepareCharacterCollisionPoints(List<Vector3D> outList)
812  {
813  MyCharacter character = (MySession.Static.ControlledEntity as MyCharacter);
814  if (character == null) return;
815 
816  float height = character.Definition.CharacterCollisionHeight * 0.7f;
817  float width = character.Definition.CharacterCollisionWidth * 0.2f;
818 
819  if (character != null)
820  {
821  if (character.IsCrouching)
822  height = character.Definition.CharacterCollisionCrouchHeight;
823 
824  Vector3 upVec = character.PositionComp.LocalMatrix.Up * height;
825  Vector3 fwVec = character.PositionComp.LocalMatrix.Forward * width;
826  Vector3 rtVec = character.PositionComp.LocalMatrix.Right * width;
827  Vector3D pos = character.Entity.PositionComp.GetPosition() + character.PositionComp.LocalMatrix.Up * 0.2f;
828 
829  float angle = 0.0f;
830  for (int i = 0; i < 6; ++i)
831  {
832  float sin = (float)Math.Sin(angle);
833  float cos = (float)Math.Cos(angle);
834  Vector3D bottomPoint = pos + sin * rtVec + cos * fwVec;
835  outList.Add(bottomPoint);
836  outList.Add(bottomPoint + upVec);
837  angle += (float)Math.PI / 3.0f;
838  }
839  }
