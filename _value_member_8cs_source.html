308  private IProtoSerializer BuildSerializer()
309  {
310  int opaqueToken = 0;
311  try
312  {
313  model.TakeLock(ref opaqueToken);// check nobody is still adding this type
314  WireType wireType;
315  Type finalType = itemType == null ? memberType : itemType;
316  IProtoSerializer ser = TryGetCoreSerializer(model, dataFormat, finalType, out wireType, asReference, dynamicType, OverwriteList, true);
317  if (ser == null) throw new InvalidOperationException("No serializer defined for type: " + finalType.FullName);
318 
319  // apply tags
320  if (itemType != null && SupportNull)
321  {
322  if(IsPacked)
323  {
324  throw new NotSupportedException("Packed encodings cannot support null values");
325  }
326  ser = new TagDecorator(NullDecorator.Tag, wireType, IsStrict, ser);
327  ser = new NullDecorator(model, ser);
328  ser = new TagDecorator(fieldNumber, WireType.StartGroup, false, ser);
329  }
330  else
331  {
332  ser = new TagDecorator(fieldNumber, wireType, IsStrict, ser);
333  }
334  // apply lists if appropriate
335  if (itemType != null)
336  {
337 #if NO_GENERICS
338  Type underlyingItemType = itemType;
339 #else
340  Type underlyingItemType = SupportNull ? itemType : Helpers.GetUnderlyingType(itemType) ?? itemType;
341 #endif
342  Helpers.DebugAssert(underlyingItemType == ser.ExpectedType, "Wrong type in the tail; expected {0}, received {1}", ser.ExpectedType, underlyingItemType);
343  if (memberType.IsArray)
344  {
345  ser = new ArrayDecorator(model, ser, fieldNumber, IsPacked, wireType, memberType, OverwriteList, SupportNull);
346  }
347  else
348  {
349  ser = new ListDecorator(model, memberType, defaultType, ser, fieldNumber, IsPacked, wireType, member != null && PropertyDecorator.CanWrite(model, member), OverwriteList, SupportNull);
350  }
351  }
352  else if (defaultValue != null && !IsRequired && getSpecified == null)
353  { // note: "ShouldSerialize*" / "*Specified" / etc ^^^^ take precedence over defaultValue,
354  // as does "IsRequired"
355  ser = new DefaultValueDecorator(model, defaultValue, ser);
356  }
357  if (memberType == model.MapType(typeof(Uri)))
358  {
359  ser = new UriDecorator(model, ser);
360  }
361  if (member != null)
362  {
363  PropertyInfo prop = member as PropertyInfo;
364  if (prop != null)
365  {
366  ser = new PropertyDecorator(model, parentType, (PropertyInfo)member, ser);
367  }
368  else
369  {
370  FieldInfo fld = member as FieldInfo;
371  if (fld != null)
372  {
373  ser = new FieldDecorator(parentType, (FieldInfo)member, ser);
374  }
375  else
376  {
377  throw new InvalidOperationException();
378  }
379  }
380  if (getSpecified != null || setSpecified != null)
381  {
382  ser = new MemberSpecifiedDecorator(getSpecified, setSpecified, ser);
383  }
384  }
385  return ser;
386  }
387  finally
388  {
389  model.ReleaseLock(opaqueToken);
390  }
391  }

413  internal static IProtoSerializer TryGetCoreSerializer(RuntimeTypeModel model, DataFormat dataFormat, Type type, out WireType defaultWireType,
414  bool asReference, bool dynamicType, bool overwriteList, bool allowComplexTypes)
415  {
416 #if !NO_GENERICS
417  type = Helpers.GetUnderlyingType(type) ?? type;
418 #endif
419  if (Helpers.IsEnum(type))
420  {
421  if (allowComplexTypes && model != null)
422  {
423  // need to do this before checking the typecode; an int enum will report Int32 etc
424  defaultWireType = WireType.Variant;
425  return new EnumSerializer(type, model.GetEnumMap(type));
426  }
427  else
428  { // enum is fine for adding as a meta-type
429  defaultWireType = WireType.None;
430  return null;
431  }
432  }
433  ProtoTypeCode code = Helpers.GetTypeCode(type);
434  switch (code)
435  {
436  case ProtoTypeCode.Int32:
437  defaultWireType = GetIntWireType(dataFormat, 32);
438  return new Int32Serializer(model);
439  case ProtoTypeCode.UInt32:
440  defaultWireType = GetIntWireType(dataFormat, 32);
441  return new UInt32Serializer(model);
442  case ProtoTypeCode.Int64:
443  defaultWireType = GetIntWireType(dataFormat, 64);
444  return new Int64Serializer(model);
445  case ProtoTypeCode.UInt64:
446  defaultWireType = GetIntWireType(dataFormat, 64);
447  return new UInt64Serializer(model);
448  case ProtoTypeCode.String:
449  defaultWireType = WireType.String;
450  if (asReference)
451  {
452  return new NetObjectSerializer(model, model.MapType(typeof(string)), 0, BclHelpers.NetObjectOptions.AsReference);
453  }
454  return new StringSerializer(model);
455  case ProtoTypeCode.Single:
456  defaultWireType = WireType.Fixed32;
457  return new SingleSerializer(model);
458  case ProtoTypeCode.Double:
459  defaultWireType = WireType.Fixed64;
460  return new DoubleSerializer(model);
461  case ProtoTypeCode.Boolean:
462  defaultWireType = WireType.Variant;
463  return new BooleanSerializer(model);
464  case ProtoTypeCode.DateTime:
465  defaultWireType = GetDateTimeWireType(dataFormat);
466  return new DateTimeSerializer(model);
467  case ProtoTypeCode.Decimal:
468  defaultWireType = WireType.String;
469  return new DecimalSerializer(model);
470  case ProtoTypeCode.Byte:
471  defaultWireType = GetIntWireType(dataFormat, 32);
472  return new ByteSerializer(model);
473  case ProtoTypeCode.SByte:
474  defaultWireType = GetIntWireType(dataFormat, 32);
475  return new SByteSerializer(model);
476  case ProtoTypeCode.Char:
477  defaultWireType = WireType.Variant;
478  return new CharSerializer(model);
479  case ProtoTypeCode.Int16:
480  defaultWireType = GetIntWireType(dataFormat, 32);
481  return new Int16Serializer(model);
482  case ProtoTypeCode.UInt16:
483  defaultWireType = GetIntWireType(dataFormat, 32);
484  return new UInt16Serializer(model);
485  case ProtoTypeCode.TimeSpan:
486  defaultWireType = GetDateTimeWireType(dataFormat);
487  return new TimeSpanSerializer(model);
488  case ProtoTypeCode.Guid:
489  defaultWireType = WireType.String;
490  return new GuidSerializer(model);
491  case ProtoTypeCode.Uri:
492  defaultWireType = WireType.String;
493  return new StringSerializer(model); // treat as string; wrapped in decorator later
494  case ProtoTypeCode.ByteArray:
495  defaultWireType = WireType.String;
496  return new BlobSerializer(model, overwriteList);
497  case ProtoTypeCode.Type:
498  defaultWireType = WireType.String;
499  return new SystemTypeSerializer(model);
500  }
501  IProtoSerializer parseable = model.AllowParseableTypes ? ParseableSerializer.TryCreate(type, model) : null;
502  if (parseable != null)
503  {
504  defaultWireType = WireType.String;
505  return parseable;
506  }
507  if (allowComplexTypes && model != null)
508  {
509  int key = model.GetKey(type, false, true);
510  if (asReference || dynamicType)
511  {
512  defaultWireType = WireType.String;
513  BclHelpers.NetObjectOptions options = BclHelpers.NetObjectOptions.None;
514  if (asReference) options |= BclHelpers.NetObjectOptions.AsReference;
515  if (dynamicType) options |= BclHelpers.NetObjectOptions.DynamicType;
516  if (key >= 0)
517  { // exists
518  if (asReference && Helpers.IsValueType(type))
519  {
520  string message = "AsReference cannot be used with value-types";
521 
522  if (type.Name == "KeyValuePair`2")
523  {
524  message += "; please see http://stackoverflow.com/q/14436606/";
525  }
526  else
527  {
528  message += ": " + type.FullName;
529  }
530  throw new InvalidOperationException(message);
531  }
532  MetaType meta = model[type];
533  if (asReference && meta.IsAutoTuple) options |= BclHelpers.NetObjectOptions.LateSet;
534  if (meta.UseConstructor) options |= BclHelpers.NetObjectOptions.UseConstructor;
535  }
536  return new NetObjectSerializer(model, type, key, options);
537  }
538  if (key >= 0)
539  {
540  defaultWireType = dataFormat == DataFormat.Group ? WireType.StartGroup : WireType.String;
541  return new SubItemSerializer(type, key, model[type], true);
542  }
543  }
544  defaultWireType = WireType.None;
545  return null;
546  }
