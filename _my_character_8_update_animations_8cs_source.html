246  private Vector3 FilterLocalSpeed(Vector3 localSpeedWorldRotUnfiltered)
247  {
248  m_animationSpeedFilter[m_animationSpeedFilterCursor++] = localSpeedWorldRotUnfiltered;
249  if (m_animationSpeedFilterCursor >= m_animationSpeedFilter.Length)
