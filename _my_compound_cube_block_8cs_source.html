647  private static bool CompoundDummiesIntersect(ref Matrix thisRotation, ref Matrix otherRotation, List<MyModelDummy> thisDummies, List<MyModelDummy> otherDummies)
648  {
649  foreach (var dummy in thisDummies)
650  {
651  //TODO: thisBoxHalfExtends should be dummy.Matrix.Scale * 0.5 but Scale in Matrix is bad.
652  Vector3 thisBoxHalfExtends = new Vector3(dummy.Matrix.Right.Length(), dummy.Matrix.Up.Length(), dummy.Matrix.Forward.Length()) * 0.5f;
653  BoundingBox thisAABB = new BoundingBox(-thisBoxHalfExtends, thisBoxHalfExtends);
654 
655  // Normalize this dummy, use inverse matrix as temporary.
656  Matrix thisDummyMatrixInv = Matrix.Normalize(dummy.Matrix);
657 
658  // Rotate this dummy
659  Matrix thisDummyMatrix;
660  Matrix.Multiply(ref thisDummyMatrixInv, ref thisRotation, out thisDummyMatrix);
661  // Create trasform to this dummy (inverse).
662  Matrix.Invert(ref thisDummyMatrix, out thisDummyMatrixInv);
663 
664  //DebugDrawAABB(thisAABB, thisDummyMatrix);
665 
666  foreach (var otherDummy in otherDummies)
667  {
668  //TODO: otherBoxHalfExtends should be otherDummy.Matrix.Scale * 0.5 but Scale in Matrix is bad.
669  Vector3 otherBoxHalfExtends = new Vector3(otherDummy.Matrix.Right.Length(), otherDummy.Matrix.Up.Length(), otherDummy.Matrix.Forward.Length()) * 0.5f;
670  BoundingBox otherAABB = new BoundingBox(-otherBoxHalfExtends, otherBoxHalfExtends);
671 
672  // Store normalized dummy matrix as temporary
673  Matrix otherDummyMatrixInThis = Matrix.Normalize(otherDummy.Matrix);
674 
675  // Rotate other dummy
676  Matrix otherDummyMatrix;
677  Matrix.Multiply(ref otherDummyMatrixInThis, ref otherRotation, out otherDummyMatrix);
678  // Transform other dummy to this dummy
679  Matrix.Multiply(ref otherDummyMatrix, ref thisDummyMatrixInv, out otherDummyMatrixInThis);
680 
681  MyOrientedBoundingBox obb = MyOrientedBoundingBox.Create(otherAABB, otherDummyMatrixInThis);
682  //DebugDrawOBB(obb, thisDummyMatrix);
683  if (obb.Intersects(ref thisAABB))
684  return true;
685  }
686  }
687 
688  return false;
