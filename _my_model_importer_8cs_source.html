696  static void LinearKeyframeReduction(LinkedList<MyAnimationClip.Keyframe> keyframes, float translationThreshold, float rotationThreshold)
697  {
698  if (keyframes.Count < 3)
699  return;
700 
701  for (LinkedListNode<MyAnimationClip.Keyframe> node = keyframes.First.Next; ; )
702  {
703  LinkedListNode<MyAnimationClip.Keyframe> next = node.Next;
704  if (next == null)
705  break;
706 
707  // Determine nodes before and after the current node.
708  MyAnimationClip.Keyframe a = node.Previous.Value;
709  MyAnimationClip.Keyframe b = node.Value;
710  MyAnimationClip.Keyframe c = next.Value;
711 
712  float t = (float)((node.Value.Time - node.Previous.Value.Time) /
713  (next.Value.Time - node.Previous.Value.Time));
714 
715  Vector3 translation = Vector3.Lerp(a.Translation, c.Translation, t);
716  var rotation = VRageMath.Quaternion.Slerp(a.Rotation, c.Rotation, t);
717 
718  if ((translation - b.Translation).LengthSquared() < translationThreshold &&
719  VRageMath.Quaternion.Dot(rotation, b.Rotation) > rotationThreshold)
720  {
721  keyframes.Remove(node);
722  }
723 
724  node = next;
725  }

902  private void LoadTagData(BinaryReader reader, string[] tags)
903  {
904  //m_retTagData
905  string tagName = reader.ReadString();
906  string[] strArr = ReadArrayOfString(reader);
907  m_retTagData.Add(tagName, strArr);
908 
909  string versionTag = "Version:";
910  if (strArr.Length > 0 && strArr[0].Contains(versionTag))
911  {
912  string version = strArr[0].Replace(versionTag, "");
913  m_version = Convert.ToInt32(version);
914  }
915 
916  if (m_version >= 01066002)
917  {
918  var dict = ReadIndexDictionary(reader);
919 
920  if (tags == null)
921  tags = dict.Keys.ToArray();
922 
923  foreach (var tag in tags)
924  {
925  if (dict.ContainsKey(tag))
926  {
927  int index = dict[tag];
928  reader.BaseStream.Seek(index, SeekOrigin.Begin);
929 
930  string readTag = reader.ReadString();
931  System.Diagnostics.Debug.Assert(tag == readTag, "Wrong model data (version mismatch?)");
932 
933  if (TagReaders.ContainsKey(tag))
934  {
935  m_retTagData.Add(tag, TagReaders[tag].Read(reader, m_version));
936  }
937  }
938  }
939  }
