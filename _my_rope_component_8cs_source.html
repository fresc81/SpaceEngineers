1240  private static Vector3 ComputeTorqueFromRopeImpulse(WindingData windingData, InternalRopeData ropeData, Vector3 ropeDirectionVector, Vector3 centerDelta)
1241  {
1242  /*
1243  * /R = rope direction vector (unit length, world space)
1244  * /A = drum axis (unit length, world space)
1245  * /r = displacement vector for torque application (world space) from center of mass
1246  * /tau = torque applied
1247  * /C_d = center delta (center of winding (world space) - center of mass (world space))
1248  * /F = force
1249  * r = drum radius
1250  * I = impulse applied
1251  *
1252  * /r = |/R x /A| * r + C_d
1253  * /F = /R * -I
1254  * /tau = /r x /F
1255  */
1256 
1257  Vector3 A, r, F, tau;
1258  Vector3.Normalize(ref ropeDirectionVector, out ropeDirectionVector);
1259  A = windingData.CurrentDummyWorld.Backward;
1260  Vector3.Cross(ref ropeDirectionVector, ref A, out r);
1261  r = r * windingData.Radius + centerDelta;
1262  F = ropeDirectionVector * ropeData.ImpulseApplied;
1263  Vector3.Cross(ref r, ref F, out tau);
1264  return tau;
1265  }
