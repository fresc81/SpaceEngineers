165  static void Validate<T>(Type type, T list) where T : IList<string>
166  {
167  Array values = Enum.GetValues(type);
168  Type underlyingType = Enum.GetUnderlyingType(type);
169  if (underlyingType == typeof(System.Byte))
170  {
171  foreach (byte value in values)

173  MyDebug.AssertRelease(list[value] != null);
174  }
175  }
176  else if (underlyingType == typeof(System.Int16))
