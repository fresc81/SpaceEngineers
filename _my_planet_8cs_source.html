830  public bool CorrectSpawnLocation2(ref Vector3D position, double radius, bool resumeSearch = false)
831  {
832  Vector3D upVector = position - WorldMatrix.Translation;
833  upVector.Normalize();
834 
835  Vector3D offset = new Vector3D(radius, radius, radius);
836  ContainmentType cType;
837  Vector3D localPos;
838  BoundingBox testBBox;
839  Vector3D fixedPosition;
840  if (resumeSearch)
841  {
842  cType = ContainmentType.Intersects;
843  fixedPosition = position;
844  }
845  else
846  {
847  // Calculate if the position is not inside of the planet:
848  MyVoxelCoordSystems.WorldPositionToLocalPosition(PositionLeftBottomCorner, ref position, out localPos);
849 
850  // Setup safe bounding box for the drone.
851  testBBox = new BoundingBox(localPos - offset, localPos + offset);
852 
853  cType = Storage.Intersect(ref testBBox);
854  if (cType == ContainmentType.Disjoint)
855  return true;
856 
857  fixedPosition = GetClosestSurfacePointGlobal(ref position);
858  }
859 
860  int i = 0;
861  while (i < 10)
862  {
863  // Spawn it above the ground in the direction of up vector
864  fixedPosition += upVector * radius;
865 
866  MyVoxelCoordSystems.WorldPositionToLocalPosition(PositionLeftBottomCorner, ref fixedPosition, out localPos);
867  testBBox = new BoundingBox(localPos - offset, localPos + offset);
868  cType = Storage.Intersect(ref testBBox);
869 
870  if (cType == ContainmentType.Disjoint)
871  {
872  position = fixedPosition;
873  return true;
874  }
875 
876  i++;
877  }
878 
879  return false;
880  }
881 
882  // NOTE: This method has a problem: "closestPlanetPos" is always a point on the surface
883  // of the planet, so "position" is never more than up*radius far away than the surface.
884  // Also we don't really know if the given position was really fixed or not cause the
885  // modified position may still be colliding. CorrectSpawnLocation2() addresses
886  // all these problems
887  public void CorrectSpawnLocation(ref Vector3D position, double radius)
888  {
889  // Generate up vector according to the gravity of the planet.
890  Vector3D upVector = position - WorldMatrix.Translation;
891  upVector.Normalize();
892 
893  // Calculate if the position is not inside of the planet:
894  Vector3D localPos;
895  MyVoxelCoordSystems.WorldPositionToLocalPosition(PositionLeftBottomCorner, ref position, out localPos);
896 
897  // Setup safe bounding box for the drone.
898  Vector3D offset = new Vector3D(radius, radius, radius);
899  BoundingBox testBBox = new BoundingBox(localPos - offset, localPos + offset);
900 
901  ContainmentType cType = Storage.Intersect(ref testBBox);
902  int i = 0;
903  while (i < 10 && (cType == ContainmentType.Intersects || cType == ContainmentType.Contains))
904  {
905  Vector3D closestPlanetPos = GetClosestSurfacePointGlobal(ref position);
906  // Spawn it above the ground in the direction of up vector + 50% of spawn radius (just in case)
907  position = closestPlanetPos + upVector * radius;
908 
909  MyVoxelCoordSystems.WorldPositionToLocalPosition(PositionLeftBottomCorner, ref position, out localPos);
910  testBBox = new BoundingBox(localPos - offset, localPos + offset);
911  cType = Storage.Intersect(ref testBBox);
912 
913  i++;
914  }
915  }
