774  // Moves keyboard index to the next item, or previous item, or first item in the combobox.
775  // forwardMovement -> set to TRUE when you want forward movement, set to FALSE when you wasnt backward
776  void HandleItemMovement(bool forwardMovement, bool page = false, bool list = false)
777  {
778  m_preselectedKeyboardIndexPrevious = m_preselectedKeyboardIndex;
779 
780  int step = 0;
781  if (list && forwardMovement) // first item
782  {
783  m_preselectedKeyboardIndex = 0;
784  }
785  else if (list && !forwardMovement) // last item
786  {
787  m_preselectedKeyboardIndex = m_items.Count - 1;
788  }
789  else if (page && forwardMovement) // step + 1 page
790  {
791  if (m_openAreaItemsCount > m_items.Count)
792  step = m_items.Count - 1;
793  else
794  step = m_openAreaItemsCount - 1;
795  }
796  else if (page && !forwardMovement) // step - 1 page
797  {
798  if (m_openAreaItemsCount > m_items.Count)
799  step = -(m_items.Count - 1);
800  else
801  step = -m_openAreaItemsCount + 1;
802  }
803  else if (!page && !list && forwardMovement) // step 1 item
804  {
805  step = 1;
806  }
807  else // step -1 item
808  {
809  step = -1;
810  }
811 
812 
813  if (m_preselectedKeyboardIndex.HasValue == false)
814  {
815  // If this is first keypress in this combobox, we will set keyboard index to begining or end of the list
816  m_preselectedKeyboardIndex = (forwardMovement == true) ? 0 : m_items.Count - 1;
817  }
818  else
819  {
820  // Increase or decrease and than check ranges and do sort of overflow
821  m_preselectedKeyboardIndex += step;// sign;
822  if (m_preselectedKeyboardIndex > (m_items.Count - 1)) m_preselectedKeyboardIndex = (m_items.Count - 1);
823  if (m_preselectedKeyboardIndex < 0) m_preselectedKeyboardIndex = 0;
824  }
825 
826  if (m_preselectedKeyboardIndex != m_preselectedKeyboardIndexPrevious)
827  {
828  MyGuiSoundManager.PlaySound(GuiSounds.MouseOver);
829  }
830 
831  SetScrollBarPositionByIndex(m_preselectedKeyboardIndex.Value);
832  }
