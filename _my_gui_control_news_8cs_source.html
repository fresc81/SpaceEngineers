345  void DownloadNewsAsync()
346  {
347  try
348  {
349  var newsDownloadClient = new WebClient();
350  newsDownloadClient.Proxy = null;
351 
352  Uri changeLogUri;
353  if (MyFinalBuildConstants.IS_STABLE)
354  changeLogUri = new Uri(MyPerGameSettings.ChangeLogUrl);

356  changeLogUri = new Uri(MyPerGameSettings.ChangeLogUrlDevelop);
357 
358  var downloadedNews = newsDownloadClient.DownloadString(changeLogUri);
359 
360  using (StringReader stream = new StringReader(downloadedNews))
361  {
362  m_downloadedNews = (MyNews)m_newsSerializer.Deserialize(stream);
363 
364  StringBuilder text = new StringBuilder();
365  for (int i = 0; i < m_downloadedNews.Entry.Count; i++)
366  {
367  var newsItem = m_downloadedNews.Entry[i];
368 
369  string itemText = newsItem.Text.Trim(m_trimArray);
370  string[] lines = itemText.Split(m_splitArray);
371 
372  text.Clear();
373  foreach (string lineItem in lines)
374  {
375  string line = lineItem.Trim();
376  text.AppendLine(line);
377  }
378 
379  m_downloadedNews.Entry[i] = new MyNewsEntry()
380  {
381  Title = newsItem.Title,
382  Version = newsItem.Version,
383  Date = newsItem.Date,
384  Text = text.ToString(),
385  };
386  }
387 
388  if (MyFakes.TEST_NEWS)
389  {
390  var entry = m_downloadedNews.Entry[m_downloadedNews.Entry.Count - 1];
391  entry.Title = "Test";
392  entry.Text = "ASDF\nASDF\n[www.spaceengineersgame.com Space engineers web]\n[[File:Textures\\GUI\\MouseCursor.dds|64x64px]]\n";
393  m_downloadedNews.Entry.Add(entry);
394  }
395  m_downloadedNewsOK = true;
396  }
397  }
