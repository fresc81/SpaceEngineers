132  private static void DrawSelectedObjectHighlight(MyHudSelectedObject selection, MyHudObjectHighlightStyleData? data)
133  {

154  DrawSelectedObjectHighlightDummy(selection, data.Value.AtlasTexture, data.Value.TextureCoord);

158  DrawSelectedObjectHighlightOutline(selection);
159  }
160  break;
161  }
162  case MyHudObjectHighlightStyle.None:
163  {
164  return;
165  }

167  throw new Exception("Unknown highlight style");
