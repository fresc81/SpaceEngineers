301  void DownloadNewsAsync()
302  {
303  try
304  {
305  var newsDownloadClient = new WebClient();
306  newsDownloadClient.Proxy = null;
307  var downloadedNews = newsDownloadClient.DownloadString(new Uri(MyPerGameSettings.ChangeLogUrl));
308 
309  using (StringReader stream = new StringReader(downloadedNews))
310  {
311  m_news = (MyNews)m_newsSerializer.Deserialize(stream);
312 
313  if(MyFinalBuildConstants.IS_STABLE)
314  {
315  m_news.Entry.RemoveAll(entry => entry.Dev);
316  }
317 
318  StringBuilder text = new StringBuilder();
319  for (int i = 0; i < m_news.Entry.Count; i++)
320  {
321  var newsItem = m_news.Entry[i];
322 
323  string itemText = newsItem.Text.Trim(m_trimArray);
324  string[] lines = itemText.Split(m_splitArray);
325 
326  text.Clear();
327  foreach (string lineItem in lines)
328  {
329  string line = lineItem.Trim();
330  text.AppendLine(line);
331  }
332 
333  m_news.Entry[i] = new MyNewsEntry()
334  {
335  Title = newsItem.Title,
336  Version = newsItem.Version,
337  Date = newsItem.Date,
338  Text = text.ToString(),
339  };
340  }
341 
342  if (MyFakes.TEST_NEWS)
343  {
344  var entry = m_news.Entry[m_news.Entry.Count - 1];
345  entry.Title = "Test";
346  entry.Text = "ASDF\nASDF\n[www.spaceengineersgame.com Space engineers web]\n[[File:Textures\\GUI\\MouseCursor.dds|64x64px]]\n";
347  m_news.Entry.Add(entry);
348  }
349  m_downloadedNewsOK = true;
350  }
351  }
352  catch (Exception e)
353  {
354  MyLog.Default.WriteLine("Error while downloading news: " + e.ToString());
355  }
356  finally
357  {
358  m_downloadedNewsFinished = true;
359  }
360  }
