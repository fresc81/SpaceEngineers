199  public override void SetInputFromDistributor(MyDefinitionId resourceTypeId, float newResourceInput, bool isAdaptible, bool fireEvents = true)

222  bool currentInputChanged = newResourceInput != m_dataPerType[typeIndex].CurrentInput;
223  bool isPoweredChanged = (newIsPowered != m_dataPerType[typeIndex].IsPowered);
224 
225  m_dataPerType[typeIndex].IsPowered = newIsPowered;
226  m_dataPerType[typeIndex].SuppliedRatio = newSuppliedRatio;
227  m_dataPerType[typeIndex].CurrentInput = newResourceInput;
