640  //calculate ratio between each thruster type based on their current propulsion power
641  private void CalculateThrusterComposition()
642  {
643  if (m_shipThrusters == null)
644  {
645  m_thrusterVolumeTargets[(int)ShipThrusters.Ion] = 0f;
646  m_thrusterVolumeTargets[(int)ShipThrusters.Hydrogen] = 0f;
647  m_thrusterVolumeTargets[(int)ShipThrusters.Atmospheric] = 0f;
648  return;
649  }
650  float ion = 0f;
651  float hydro = 0f;
652  float atmo = 0f;
653 
654  bool hasIon = false;
655  bool hasAtmo = false;
656  bool hasHydro = false;
657 
658  //calculate total force for each thruster type
659  foreach (MyThrust thruster in m_shipGrid.GetFatBlocks<MyThrust>())
660  {
661  if (thruster != null)
662  {
663  if (thruster.BlockDefinition.ThrusterType == m_thrusterHydrogen)
664  {
665  hydro += thruster.CurrentStrength * (Math.Abs(thruster.ThrustForce.X) + Math.Abs(thruster.ThrustForce.Y) + Math.Abs(thruster.ThrustForce.Z));
666  hasHydro = hasHydro || (thruster.IsFunctional && thruster.Enabled);
667  }
668  else if (thruster.BlockDefinition.ThrusterType == m_thrusterAtmospheric)
669  {
670  atmo += thruster.CurrentStrength * (Math.Abs(thruster.ThrustForce.X) + Math.Abs(thruster.ThrustForce.Y) + Math.Abs(thruster.ThrustForce.Z));
671  hasAtmo = hasAtmo || (thruster.IsFunctional && thruster.Enabled);
672  }
673  else
674  {
675  ion += thruster.CurrentStrength * (Math.Abs(thruster.ThrustForce.X) + Math.Abs(thruster.ThrustForce.Y) + Math.Abs(thruster.ThrustForce.Z));
676  hasIon = hasIon || (thruster.IsFunctional && thruster.Enabled);
677  }
678  }
679  }
680 
681  ShipHasChanged = false;
682  m_singleThrusterTypeShip = !((hasIon && hasAtmo) || (hasIon && hasHydro) || (hasHydro && hasAtmo));
683 
684  //calculate volume modifiers for each thruster types based on ratio between them
685  if (m_singleThrusterTypeShip)
686  {
687  m_thrusterVolumeTargets[(int)ShipThrusters.Ion] = hasIon ? 1f : 0f;
688  m_thrusterVolumeTargets[(int)ShipThrusters.Hydrogen] = hasHydro ? 1f : 0f;
689  m_thrusterVolumeTargets[(int)ShipThrusters.Atmospheric] = hasAtmo ? 1f : 0f;
690  if (!hasIon && !hasHydro && !hasAtmo)
691  ShipHasChanged = true;
692  }
693  else if (ion + hydro + atmo > 0f)//at least one thruster is thrusting
694  {
695  float sum = hydro + ion + atmo;
696  ion = ion > 0f ? (m_groupData.ThrusterCompositionMinVolume_c + (ion / sum)) / (1f + m_groupData.ThrusterCompositionMinVolume_c) : 0f;
697  hydro = hydro > 0f ? (m_groupData.ThrusterCompositionMinVolume_c + (hydro / sum)) / (1f + m_groupData.ThrusterCompositionMinVolume_c) : 0f;
698  atmo = atmo > 0f ? (m_groupData.ThrusterCompositionMinVolume_c + (atmo / sum)) / (1f + m_groupData.ThrusterCompositionMinVolume_c) : 0f;
699  m_thrusterVolumeTargets[(int)ShipThrusters.Ion] = ion;
700  m_thrusterVolumeTargets[(int)ShipThrusters.Hydrogen] = hydro;
701  m_thrusterVolumeTargets[(int)ShipThrusters.Atmospheric] = atmo;
702  }
703 
704  //stop obsolete sounds
705  if (m_thrusterVolumes[(int)ShipThrusters.Ion] <= 0f && m_emitters[(int)ShipEmitters.IonThrusters].IsPlaying)
706  {
707  m_emitters[(int)ShipEmitters.IonThrusters].StopSound(false);
708  m_emitters[(int)ShipEmitters.IonThrustersIdle].StopSound(false);
709  }
710  if (m_thrusterVolumes[(int)ShipThrusters.Hydrogen] <= 0f && m_emitters[(int)ShipEmitters.HydrogenThrusters].IsPlaying)
711  {
712  m_emitters[(int)ShipEmitters.HydrogenThrusters].StopSound(false);
713  m_emitters[(int)ShipEmitters.HydrogenThrustersIdle].StopSound(false);
714  }
715  if (m_thrusterVolumes[(int)ShipThrusters.Atmospheric] <= 0f && m_emitters[(int)ShipEmitters.AtmosphericThrusters].IsPlaying)
716  {
717  m_emitters[(int)ShipEmitters.AtmosphericThrusters].StopSound(false);
718  m_emitters[(int)ShipEmitters.AtmosphericThrustersIdle].StopSound(false);
719  }
720 
721  //check if we need to play new sound
722  if ((m_thrusterVolumeTargets[(int)ShipThrusters.Ion] > 0f && !m_emitters[(int)ShipEmitters.IonThrusters].IsPlaying)
723  || (m_thrusterVolumeTargets[(int)ShipThrusters.Hydrogen] > 0f && !m_emitters[(int)ShipEmitters.HydrogenThrusters].IsPlaying)
724  || (m_thrusterVolumeTargets[(int)ShipThrusters.Atmospheric] > 0f && !m_emitters[(int)ShipEmitters.AtmosphericThrusters].IsPlaying))
725  m_forceSoundCheck = true;
726  }
