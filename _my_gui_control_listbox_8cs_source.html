593  private void HandleNewMousePress(ref MyGuiControlBase captureInput)
594  {
595  var mousePos = MyGuiManager.MouseCursorPosition - GetPositionAbsoluteTopLeft();
596  bool cursorInItems = m_itemsRectangle.Contains(mousePos);
597 
598  if (MyInput.Static.IsAnyNewMouseOrJoystickPressed() && cursorInItems)
599  {
600  int row = ComputeIndexFromPosition(mousePos);
601  if (IsValidIndex(row) && Items[row].Visible)
602  {
603  if (MultiSelect && MyInput.Static.IsAnyCtrlKeyPressed())
604  {
605  if (SelectedItems.Contains(Items[row]))
606  SelectedItems.Remove(Items[row]);
607  else
608  SelectedItems.Add(Items[row]);
609  }
610  else if (MultiSelect && MyInput.Static.IsAnyShiftKeyPressed())
611  {
612  int index = 0;
613  if (SelectedItems.Count > 0)
614  index = Items.IndexOf(SelectedItems[SelectedItems.Count - 1]);
615 
616  do
617  {
618  index += index > row ? -1 : 1;
619  if (!IsValidIndex(index))
620  break;
621  if (!Items[index].Visible)
622  continue;
623  if (SelectedItems.Contains(Items[index]))
624  SelectedItems.Remove(Items[index]);
625  else
626  SelectedItems.Add(Items[index]);
627  } while (index != row);
628  }
629  else
630  {
631  SelectedItems.Clear();
632  SelectedItems.Add(Items[row]);
633  }
634  if (ItemsSelected != null)
635  ItemsSelected(this);
636 
637  captureInput = this;
638  if (ItemClicked != null)
639  {
640  ItemClicked(this);
641  MyGuiSoundManager.PlaySound(GuiSounds.MouseClick);
642  }
643  }
644  }
645 
646  if (MyInput.Static.IsNewPrimaryButtonPressed() && cursorInItems)
647  {
648  if (!m_doubleClickStarted.HasValue)
649  {
650  int row = ComputeIndexFromPosition(mousePos);
651  if (IsValidIndex(row) && Items[row].Visible)
652  {
653  m_doubleClickStarted = MyGuiManager.TotalTimeInMilliseconds;
654  m_doubleClickFirstPosition = MyGuiManager.MouseCursorPosition;
655  }
656  }
657  else if ((MyGuiManager.TotalTimeInMilliseconds - m_doubleClickStarted.Value) <= MyGuiConstants.DOUBLE_CLICK_DELAY &&
658  (m_doubleClickFirstPosition - MyGuiManager.MouseCursorPosition).Length() <= 0.005f)
659  {
660  if (ItemDoubleClicked != null)
661  ItemDoubleClicked(this);
662 
663  m_doubleClickStarted = null;
664  captureInput = this;
665  }
666  }
