219  private void CalculateMergeArea(out Vector3I minI, out Vector3I maxI)
220  {

398  private void CalculateMergeData(ref MergeData data)
399  {
400  var mergeBlockDefinition = this.BlockDefinition as MyMergeBlockDefinition;
401  float maxStrength = mergeBlockDefinition != null ? mergeBlockDefinition.Strength : 0.1f;
402  data.Distance = (float)(WorldMatrix.Translation - m_other.WorldMatrix.Translation).Length() - CubeGrid.GridSize;
403 
404  data.StrengthFactor = (float)Math.Exp(-data.Distance / CubeGrid.GridSize);
405  // Debug.Assert(x <= 1.0f); // This is not so important, but testers kept reporting it, so let's leave it commented out :-)
406  float strength = MathHelper.Lerp(0.0f, maxStrength * (CubeGrid.GridSizeEnum == MyCubeSize.Large ? 0.005f : 0.1f), data.StrengthFactor); // 0.005 for large grid, 0.1 for small grid!?
407 
408  Vector3 thisVelocity = CubeGrid.Physics.GetVelocityAtPoint(PositionComp.GetPosition());
409  Vector3 otherVelocity = m_other.CubeGrid.Physics.GetVelocityAtPoint(m_other.PositionComp.GetPosition());
410  data.RelativeVelocity = otherVelocity - thisVelocity;
411  float velocityFactor = 1.0f;
412 
413  // The quicker the ships move towards each other, the weaker the constraint strength
414  float rvLength = data.RelativeVelocity.Length();
415  velocityFactor = Math.Max(3.6f / (rvLength > 0.1f ? rvLength : 0.1f), 1.0f);
416  data.ConstraintStrength = strength / velocityFactor;
417 
418  Vector3 toOther = m_other.PositionComp.GetPosition() - PositionComp.GetPosition();
419  Vector3 forward = WorldMatrix.GetDirectionVector(m_forward);
420 
421  data.Distance = (toOther).Length();
422  data.PositionOk = data.Distance < CubeGrid.GridSize + 0.17f; // 17 cm is tested working value. 15 cm was too few
423 
424  data.AxisDelta = (float)(forward + m_other.WorldMatrix.GetDirectionVector(m_forward)).Length();
425  data.AxisOk = data.AxisDelta < 0.1f;
426 
427  data.RotationDelta = (float)(WorldMatrix.GetDirectionVector(m_right) - m_other.WorldMatrix.GetDirectionVector(m_other.m_otherRight)).Length();
