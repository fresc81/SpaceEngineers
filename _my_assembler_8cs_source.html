844  {
845  ClearQueue(sendEvent: false);

847  var items = OutputInventory.GetItems();
848  var toAdd = new List<Tuple<MyBlueprintDefinitionBase, MyFixedPoint>>();
849  bool add = true;

854  if (blueprint != null)
855  {
856  var tuple = Tuple.Create(blueprint, item.Amount);
857  toAdd.Add(tuple);
858  }
859  else
860  {
861  add = false;
862  toAdd.Clear();
863  break;
864  }
865  }
866  if (add)
867  {
868  foreach (var bp in toAdd)
869  {
870  InsertQueueItemRequest(-1, bp.Item1, bp.Item2);
871  }
872  return;

875  InitializeInventoryCounts(inputInventory: false);
876 
877  MyFixedPoint disassembleAmount, remainingAmount;
878  for (int i = 0; i < m_assemblerDef.BlueprintClasses.Count; ++i)
879  {
880  foreach (var blueprint in m_assemblerDef.BlueprintClasses[i])
881  {
882  disassembleAmount = MyFixedPoint.MaxValue;
883  foreach (var result in blueprint.Results)
884  {
885  remainingAmount = 0;
886  m_tmpInventoryCounts.TryGetValue(result.Id, out remainingAmount);
887  if (remainingAmount == 0)
888  {
889  disassembleAmount = 0;
890  break;
891  }
892  disassembleAmount = MyFixedPoint.Min((MyFixedPoint)((double)remainingAmount / (double)result.Amount), disassembleAmount);
893  }
894 
895  if (blueprint.Atomic) disassembleAmount = MyFixedPoint.Floor(disassembleAmount);
896 
897  if (disassembleAmount > 0)
898  {
899  InsertQueueItemRequest(-1, blueprint, disassembleAmount);
900  foreach (var result in blueprint.Results)
901  {
902  m_tmpInventoryCounts.TryGetValue(result.Id, out remainingAmount);
903  remainingAmount -= result.Amount * disassembleAmount;
904  Debug.Assert(remainingAmount >= 0);
905  if (remainingAmount == 0)
906  m_tmpInventoryCounts.Remove(result.Id);
907  else
908  m_tmpInventoryCounts[result.Id] = remainingAmount;
909  }
