503  // add vertex to mesh, fix position to be on unit sphere, return index
504  private int addVertex(Vector3 p)
505  {
506  double length = Math.Sqrt(p.X * p.X + p.Y * p.Y + p.Z * p.Z);
507  this.positions.Add(new Vector3(p.X / length, p.Y / length, p.Z / length));

511  // return index of point in the middle of p1 and p2
512  private int getMiddlePoint(int p1, int p2)
513  {
514  // first check if we have it already
515  bool firstIsSmaller = p1 < p2;
516  Int64 smallerIndex = firstIsSmaller ? p1 : p2;
517  Int64 greaterIndex = firstIsSmaller ? p2 : p1;
518  Int64 key = (smallerIndex << 32) + greaterIndex;
519 
520  int ret;
521  if (this.middlePointIndexCache.TryGetValue(key, out ret))
522  {
523  return ret;
524  }
525 
526  // not in cache, calculate it
527  Vector3 point1 = this.positions[p1];
528  Vector3 point2 = this.positions[p2];
529  Vector3 middle = new Vector3(
530  (point1.X + point2.X) / 2.0,
531  (point1.Y + point2.Y) / 2.0,
532  (point1.Z + point2.Z) / 2.0);
533 
534  // add vertex makes sure point is on unit sphere
535  int i = addVertex(middle);
536 
537  // store it, return index
538  this.middlePointIndexCache.Add(key, i);
539  return i;
540  }

662

664  int N = 8; // 16;
665  float radius = 1;

673  private void Create()
674  {
675  int i, j;
676  int i1, i2, i3, i4;
677  double theta, phi;
678 
679  for (j = 0; j <= N / 4; j++)
680  { // top cap
681  for (i = 0; i <= N; i++)
682  {
683  Vector3 vert = new Vector3();
684  theta = i * PIm2 / N;
685  phi = -PId2 + Math.PI * j / (N / 2);
686  vert.X = radius * (float)(Math.Cos(phi) * Math.Cos(theta));
687  vert.Y = radius * (float)(Math.Cos(phi) * Math.Sin(theta));
688  vert.Z = radius * (float)(Math.Sin(phi)) - height / 2;
689  m_verticeList.Add(vert);
690  }
691  }
692 
693  for (j = N / 4; j <= N / 2; j++)
694  { // bottom cap
695  for (i = 0; i <= N; i++)
696  {
697  Vector3 vert = new Vector3();
698  theta = i * PIm2 / N;
699  phi = -PId2 + Math.PI * j / (N / 2);
700  vert.X = radius * (float)(Math.Cos(phi) * Math.Cos(theta));
701  vert.Y = radius * (float)(Math.Cos(phi) * Math.Sin(theta));
702  vert.Z = radius * (float)(Math.Sin(phi)) + height / 2;
703  m_verticeList.Add(vert);
704  }
705  }
706 
707  for (j = 0; j <= N / 2; j++)
708  {
709  for (i = 0; i < N; i++)
710  {
711  i1 = j * (N + 1) + i;
712  i2 = j * (N + 1) + (i + 1);
713  i3 = (j + 1) * (N + 1) + (i + 1);
714  i4 = (j + 1) * (N + 1) + i;
715 
716  m_triangleList.Add(i1);
717  m_triangleList.Add(i2);
718  m_triangleList.Add(i3);
719 
720  m_triangleList.Add(i1);
721  m_triangleList.Add(i3);
722  m_triangleList.Add(i4);
723  }
724  }
