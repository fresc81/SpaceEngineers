330  double closestDistanceSquared = double.PositiveInfinity;
331  bool positionChanged = false;
332 
333  // ray cast - very close objects
334  MyPhysics.CastRay(raycastOrigin, raycastOrigin + CAMERA_RADIUS * rayDirection, m_raycastList);

336  {

343  continue;
344

346  if (distSq < closestDistanceSquared)
347  {
348  closestDistanceSquared = distSq;
349  double dist = Math.Sqrt(distSq) - CAMERA_RADIUS;
350  outSafePosition = raycastOrigin + rayDirection * dist;
351  positionChanged = true;
352  }

364  HkShape shapeSphere = new HkSphereShape(CAMERA_RADIUS);
365  MatrixD raycastOriginTransform = MatrixD.Identity;
366  raycastOriginTransform.Translation = raycastOrigin + CAMERA_RADIUS * rayDirection;
367  MyPhysics.CastShapeReturnContactBodyDatas(raycastEnd, shapeSphere, ref raycastOriginTransform, 0,

370  float closestFraction = 1;

372  {
373  IMyEntity hitEntity = rb.HkHitInfo.GetHitEntity();
374  if (rb.HkHitInfo.Body == null
375  || rb.HkHitInfo.Body.UserObject == null
376  || hitEntity == controlledEntity

378  continue;
379  if (hitEntity is IMyHandheldGunObject<Game.Weapons.MyDeviceBase>)
380  // ignore player weapons
381  continue;
382

384  Math.Max(rb.HkHitInfo.HitFraction, 0.0001));
385  double distSq = Vector3D.DistanceSquared(raycastOrigin, outSafePosition);
386  if (rb.HkHitInfo.HitFraction < closestFraction && distSq < closestDistanceSquared)
387  {
388  outSafePosition = safePos;
389  closestDistanceSquared = distSq;
390  positionChanged = true;
391  closestFraction = rb.HkHitInfo.HitFraction;
392  }
393  }
394  shapeSphere.RemoveReference();
395  }
396 
397  if (m_debugDraw)
398  foreach (var raycastResult in m_raycastList)

400  VRageRender.MyRenderProxy.DebugDrawSphere(raycastResult.Position, CAMERA_RADIUS, Color.Red, 1, false);
