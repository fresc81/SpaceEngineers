364  void ResizeInternal<TBuffer>(TBuffer buffer, int newElements, int newByteStride, IntPtr? newData, Action<TBuffer> initializer = null)
365  where TBuffer : MyBufferInternal, new()
366  {
367  Debug.Assert(m_isDeviceInit, "Cannot modify buffer resources when the device has not been initialized!");
368  Debug.Assert(buffer != null, "Invalid buffer to dispose. It is likely a buffer not created through this manager. Use the CreateX methods to create buffers before resizing them.");
369 
370  var pool = GetPool<TBuffer>();
371  BufferDescription description = buffer.Description;
372 
373  if (newByteStride < 0)
374  newByteStride = description.StructureByteStride;
375 
376  description.SizeInBytes = newElements * newByteStride;
377  description.StructureByteStride = newByteStride;
378 
379  if (buffer.ByteSize == description.SizeInBytes
380  && buffer.Description.StructureByteStride == description.StructureByteStride
381  && newData == null)
382  {
383  return;
384  }
385 
386  pool.LogResize(buffer, ref description); // We have to log before Init
387 
388  string name = buffer.Name;
389  buffer.DisposeInternal();
390 
391  if (initializer != null)
392  initializer(buffer);
393 
394  buffer.Init(name, ref description, newData);
395  }
