104  public bool CollideRay(ref IndexedVector3 vorigin, ref IndexedVector3 vdir)
105  {
106  IndexedVector3 extents, center;
107  GetCenterExtend(out center, out extents);
108 
109  float Dx = vorigin.X - center.X;
110  if (BoxCollision.BT_GREATER(Dx, extents.X) && Dx * vdir.X >= 0.0f) return false;
111  float Dy = vorigin.Y - center.Y;
112  if (BoxCollision.BT_GREATER(Dy, extents.Y) && Dy * vdir.Y >= 0.0f) return false;
113  float Dz = vorigin.Z - center.Z;
114  if (BoxCollision.BT_GREATER(Dz, extents.Z) && Dz * vdir.Z >= 0.0f) return false;
115 
116 
117  float f = vdir.Y * Dz - vdir.Z * Dy;
118  if (Math.Abs(f) > extents.Y * Math.Abs(vdir.Z) + extents.Z * Math.Abs(vdir.Y)) return false;
119  f = vdir.Z * Dx - vdir.X * Dz;
120  if (Math.Abs(f) > extents.X * Math.Abs(vdir.Z) + extents.Z * Math.Abs(vdir.X)) return false;
121  f = vdir.X * Dy - vdir.Y * Dx;
122  if (Math.Abs(f) > extents.X * Math.Abs(vdir.Y) + extents.Y * Math.Abs(vdir.X)) return false;
123  return true;
124  }

126  public float? CollideRayDistance(ref IndexedVector3 origin, ref IndexedVector3 direction)
127  {
128  // r.dir is unit direction vector of ray
129  IndexedVector3 dirfrac = new IndexedVector3(1.0f / direction.X, 1.0f / direction.Y, 1.0f / direction.Z);
130 
131  // m_min is the corner of AABB with minimal coordinates - left bottom, m_max is maximal corner
132  // r.org is origin of ray
133  float t1 = (m_min.X - origin.X) * dirfrac.X;
134  float t2 = (m_max.X - origin.X) * dirfrac.X;
135  float t3 = (m_min.Y - origin.Y) * dirfrac.Y;
136  float t4 = (m_max.Y - origin.Y) * dirfrac.Y;
137  float t5 = (m_min.Z - origin.Z) * dirfrac.Z;
138  float t6 = (m_max.Z - origin.Z) * dirfrac.Z;
139 
140  float tmin = Math.Max(Math.Max(Math.Min(t1, t2), Math.Min(t3, t4)), Math.Min(t5, t6));
141  float tmax = Math.Min(Math.Min(Math.Max(t1, t2), Math.Max(t3, t4)), Math.Max(t5, t6));
142 
143  float t;
144 
145  // if tmax < 0, ray (line) is intersecting AABB, but whole AABB is behing us
146  if (tmax < 0)
147  {
148  t = tmax;
149  return null;
150  }
151 
152  // if tmin > tmax, ray doesn't intersect AABB
153  if (tmin > tmax)
154  {
155  t = tmax;
156  return null;
157  }
158 
159  t = tmin;
160  return t;
161  }
