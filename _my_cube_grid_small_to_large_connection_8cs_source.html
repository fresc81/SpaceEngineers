749  private bool SmallBlockConnectsToLarge(MySlimBlock smallBlock, ref BoundingBoxD smallBlockWorldAabb, MySlimBlock largeBlock, ref BoundingBoxD largeBlockWorldAabb)
750  {
751  Debug.Assert(GetCubeSize(smallBlock) == MyCubeSize.Small);
752  Debug.Assert(GetCubeSize(largeBlock) == MyCubeSize.Large);
753  Debug.Assert(!(smallBlock.FatBlock is MyCompoundCubeBlock));
754  Debug.Assert(!(largeBlock.FatBlock is MyCompoundCubeBlock));
755 
756  BoundingBoxD smallBlockWorldAabbReduced = smallBlockWorldAabb;
757  smallBlockWorldAabbReduced.Inflate(-smallBlock.CubeGrid.GridSize / 4);
758 
759  // Small block aabb penetrates large block aabb (large timbers).
760  bool penetratesAabbs = largeBlockWorldAabb.Intersects(smallBlockWorldAabbReduced);
761  if (!penetratesAabbs)
762  {
763  Vector3I addDir = GetSmallBlockAddDirection(ref smallBlockWorldAabb, ref smallBlockWorldAabbReduced, ref largeBlockWorldAabb);
764  // Check small grid mount points
765  Quaternion smallBlockRotation;
766  smallBlock.Orientation.GetQuaternion(out smallBlockRotation);
767  smallBlockRotation = Quaternion.CreateFromRotationMatrix(smallBlock.CubeGrid.WorldMatrix) * smallBlockRotation;
768  if (!MyCubeGrid.CheckConnectivitySmallBlockToLargeGrid(largeBlock.CubeGrid, smallBlock.BlockDefinition, ref smallBlockRotation, ref addDir))
769  return false;
770  }
771 
772  BoundingBoxD smallBlockWorldAabbInflated = smallBlockWorldAabb;
773  smallBlockWorldAabbInflated.Inflate(2 * smallBlock.CubeGrid.GridSize / 3);
774 
775  // Trim small block aabb with large block aabb.
776  BoundingBoxD intersectedBox = smallBlockWorldAabbInflated.Intersect(largeBlockWorldAabb);
777  Vector3D intersectedBoxCenter = intersectedBox.Center;
778  HkShape shape = new HkBoxShape((Vector3)intersectedBox.HalfExtents);
779 
780  Quaternion largeRotation;
781  largeBlock.Orientation.GetQuaternion(out largeRotation);
782  largeRotation = Quaternion.CreateFromRotationMatrix(largeBlock.CubeGrid.WorldMatrix) * largeRotation;
783  Vector3D largeTranslation;
784  largeBlock.ComputeWorldCenter(out largeTranslation);
785 
786  bool result = false;
787 
788  try
789  {
790  if (largeBlock.FatBlock != null)
791  {
792  MyModel model = largeBlock.FatBlock.Model;
793  if (model != null && model.HavokCollisionShapes != null)
794  {
795  HkShape[] shapes = model.HavokCollisionShapes;
796  for (int i = 0; i < shapes.Length; ++i)
797  {
798  result = MyPhysics.IsPenetratingShapeShape(shape, ref intersectedBoxCenter, ref Quaternion.Identity, shapes[i], ref largeTranslation, ref largeRotation);
799  if (result)
800  break;
801  }
802  }
803  else
804  {
805  HkShape shapeLarge = new HkBoxShape(largeBlock.BlockDefinition.Size * largeBlock.CubeGrid.GridSize / 2);
806 
807  result = MyPhysics.IsPenetratingShapeShape(shape, ref intersectedBoxCenter, ref Quaternion.Identity, shapeLarge, ref largeTranslation, ref largeRotation);
808 
809  shapeLarge.RemoveReference();
810  }
811  }
812  else
813  {
814  HkShape shapeLarge = new HkBoxShape(largeBlock.BlockDefinition.Size * largeBlock.CubeGrid.GridSize / 2);
815 
816  result = MyPhysics.IsPenetratingShapeShape(shape, ref intersectedBoxCenter, ref Quaternion.Identity, shapeLarge, ref largeTranslation, ref largeRotation);
817 
818  shapeLarge.RemoveReference();
819  }
820  }
821  finally
822  {
823  shape.RemoveReference();
824  }
825 
826  return result;
827  }
