306  private static long MultiplySafeInternal(long a, long b)
307  {
308  long result = a * b;
309  if (b == 0 || result / b == a) return result; // Not counting the zero cases, if and only if a does not cause b to overflow when multiplied, then result / b = a
310 
311  return (Math.Sign(a) * Math.Sign(b) == 1) ? long.MaxValue : long.MinValue;
312  }

314  private static long AddSafeInternal(long a, long b)
315  {
316  int sa = Math.Sign(a);
317  if (sa * Math.Sign(b) != 1) return a + b; // Different signs or at least one zero => no overflow
318 
319  long result = a + b;
320  if (Math.Sign(result) == sa) return result; // Same signs, but the result is also the same => no overflow
321 
322  return sa < 0 ? long.MinValue : long.MaxValue; // Overflow => return the correct max value
323  }
