735  private MyConveyorConnectedGroup TrySplitGroup(IMyConveyorEndpointBlock conveyorEndpointBlock, MyConveyorConnectedGroup groupOverride = null)
736  {

739 
740  var endpointEntity = conveyorEndpointBlock as MyEntity;
741  returnGroup = groupOverride ?? FindEntityGroup(endpointEntity);
742  if (returnGroup == null)
743  return null;
744 
745  if (conveyorEndpointBlock != null && conveyorEndpointBlock.ConveyorEndpoint == returnGroup.FirstEndpoint)
746  {
747  if (returnGroup.ThrustCount == 1)
748  return returnGroup;
749 
750  // Move the reference endpoint and the sink to another entity in this group
751  foreach (var fuelData in returnGroup.DataByFuelType)
752  {
753  bool endpointMoved = false;
754  foreach (var thrustSet in fuelData.ThrustsByDirection.Values)
755  {
756  foreach (var thrustEntity in thrustSet)
757  {
758  if (thrustEntity == endpointEntity)
759  continue;

761  returnGroup.FirstEndpoint = (thrustEntity as IMyConveyorEndpointBlock).ConveyorEndpoint;
762  endpointEntity.Components.Remove<MyResourceSinkComponent>();
763  thrustEntity.Components.Add(returnGroup.ResourceSink);
764  endpointMoved = true;
765  }
766  if (endpointMoved)
767  break;
768  }
769  if (endpointMoved)
770  break;
771  }
772  }
773 
774  // Find all the newly disconnected blocks
775  foreach (var direction in Base6Directions.IntDirections)
776  {
777  for (int typeIndex = 0; typeIndex < returnGroup.FuelTypes.Count; ++typeIndex)
778  {
779  var fuelData = returnGroup.DataByFuelType[typeIndex];
780  foreach (var thrustEntity in fuelData.ThrustsByDirection[direction])
781  {
782  Debug.Assert(thrustEntity is IMyConveyorEndpointBlock);
783  var endpoint = (thrustEntity as IMyConveyorEndpointBlock).ConveyorEndpoint;
784  if (endpointEntity == thrustEntity) // Leave the splitting block in the original group
785  continue;
786 
787  if (!MyGridConveyorSystem.Reachable(endpoint, returnGroup.FirstEndpoint))
788  {
789  var fuelType = FuelType(thrustEntity);
790  returnGroup.ResourceSink.SetMaxRequiredInputByType(fuelType, returnGroup.ResourceSink.MaxRequiredInputByType(fuelType) - PowerAmountToFuel(ref fuelType, MaxPowerConsumption(thrustEntity), returnGroup));
791  --fuelData.ThrustCount;
792  --returnGroup.ThrustCount;
793  m_tmpEntitiesWithDirections.Add(new MyTuple<MyEntity, Vector3I>(thrustEntity, direction));
794  }
795  }

798  {
799  fuelData.ThrustsByDirection[entityAndDirection.Item2].Remove(entityAndDirection.Item1);
800  RemoveFromGroup(entityAndDirection.Item1, returnGroup);
801  }
802  }
803  }
804 
805  foreach (var entityAndDirection in m_tmpEntitiesWithDirections)
806  {
807  MyEntity entity = entityAndDirection.Item1;
808  Vector3I direction = entityAndDirection.Item2;
809  var fuelType = FuelType(entity);
810  bool foundGroup = false;
811  foreach (var newGroup in m_tmpGroups)
812  {
813  // A group connected to this entity was already created
814  if (MyGridConveyorSystem.Reachable((entity as IMyConveyorEndpointBlock).ConveyorEndpoint, newGroup.FirstEndpoint))
815  {
816  int typeIndex;
817  // Initialize a new type for this group if necessary
818  if (!newGroup.TryGetTypeIndex(ref fuelType, out typeIndex))
819  typeIndex = InitializeType(fuelType, newGroup.DataByFuelType, newGroup.FuelTypes, newGroup.FuelTypeToIndex, newGroup.ResourceSink);
820 
821  var fuelData = newGroup.DataByFuelType[typeIndex];
822  fuelData.ThrustsByDirection[direction].Add(entity);
823  AddToGroup(entity, newGroup);
824  ++fuelData.ThrustCount;
825  ++newGroup.ThrustCount;
826  newGroup.ResourceSink.SetMaxRequiredInputByType(fuelType, newGroup.ResourceSink.MaxRequiredInputByType(fuelType) + PowerAmountToFuel(ref fuelType, MaxPowerConsumption(entity), newGroup));
827  foundGroup = true;
828  break;
829  }
830  }
831 
832  if (foundGroup)
833  continue;
834 
835  var createdGroup = new MyConveyorConnectedGroup(entity as IMyConveyorEndpointBlock);
836  m_tmpGroups.Add(createdGroup);
837  m_connectedGroups.Add(createdGroup);
838 
839  var createdTypeIndex = InitializeType(fuelType, createdGroup.DataByFuelType, createdGroup.FuelTypes, createdGroup.FuelTypeToIndex, createdGroup.ResourceSink);
840  entity.Components.Add(createdGroup.ResourceSink);
841 
842  var createdFuelData = createdGroup.DataByFuelType[createdTypeIndex];
843  var otherTypeIndex = returnGroup.GetTypeIndex(ref fuelType);
844  createdFuelData.Efficiency = returnGroup.DataByFuelType[otherTypeIndex].Efficiency;
845  createdFuelData.EnergyDensity = returnGroup.DataByFuelType[otherTypeIndex].EnergyDensity;
846  createdFuelData.ThrustsByDirection[direction].Add(entity);
847  AddToGroup(entity, createdGroup);
848  ++createdFuelData.ThrustCount;
849  ++createdGroup.ThrustCount;
850  createdGroup.ResourceSink.SetMaxRequiredInputByType(fuelType, createdGroup.ResourceSink.MaxRequiredInputByType(fuelType) + PowerAmountToFuel(ref fuelType, MaxPowerConsumption(entity), createdGroup));
851  }
852  m_tmpGroups.Clear();
853  m_tmpEntitiesWithDirections.Clear();
854 
855  Debug.Assert(returnGroup != null, "TrySplitGroup should always return a valid group!");
856  return returnGroup;
