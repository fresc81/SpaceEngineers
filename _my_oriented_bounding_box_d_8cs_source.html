726  // length of the projection (sometime called the "radius") is equal to
727  // h(u) * abs(n dot b(u))) + h(v) * abs(n dot b(v)) + h(w) * abs(n dot b(w))
728  // where h(i) are extents of the box, n is the plane normal, and b(i) are the
729  // axes of the box.
730  double r = Math.Abs(HalfExtent.X * localNormal.X)
731  + Math.Abs(HalfExtent.Y * localNormal.Y)
732  + Math.Abs(HalfExtent.Z * localNormal.Z);
733 
734  if (dist > r)
735  {
736  return PlaneIntersectionType.Front;
737  }
738  else if (dist < -r)
739  {
740  return PlaneIntersectionType.Back;
741  }
742  else
743  {
744  return PlaneIntersectionType.Intersecting;
745  }
746  }

798  // Determine whether the box described by half-extents hA, axis-aligned and centered at the origin, contains
799  // the box described by half-extents hB, whose position and orientation are given by the transform matrix mB.
800  // The matrix is assumed to contain only rigid motion; if it contains scaling or perpsective the result of
801  // this method will be incorrect.
802  public static ContainmentType ContainsRelativeBox(ref Vector3D hA, ref Vector3D hB, ref MatrixD mB)
803  {
804  Vector3D mB_T = mB.Translation;
805  Vector3D mB_TA = new Vector3D(Math.Abs(mB_T.X), Math.Abs(mB_T.Y), Math.Abs(mB_T.Z));
806 
807  // Transform the extents of B
808  Vector3D bX = mB.Right; // x-axis of box B
809  Vector3D bY = mB.Up; // y-axis of box B
810  Vector3D bZ = mB.Backward; // z-axis of box B
811  Vector3D hx_B = bX * hB.X; // x extent of box B
812  Vector3D hy_B = bY * hB.Y; // y extent of box B
813  Vector3D hz_B = bZ * hB.Z; // z extent of box B
814 
815  // Check for containment first.
816  double projx_B = Math.Abs(hx_B.X) + Math.Abs(hy_B.X) + Math.Abs(hz_B.X);
817  double projy_B = Math.Abs(hx_B.Y) + Math.Abs(hy_B.Y) + Math.Abs(hz_B.Y);
818  double projz_B = Math.Abs(hx_B.Z) + Math.Abs(hy_B.Z) + Math.Abs(hz_B.Z);
819  if (mB_TA.X + projx_B <= hA.X && mB_TA.Y + projy_B <= hA.Y && mB_TA.Z + projz_B <= hA.Z)
820  return ContainmentType.Contains;
821 
822  // Check for separation along the faces of the other box,
823  // by projecting each local axis onto the other boxes' axes
824  // http://www.cs.unc.edu/~geom/theses/gottschalk/main.pdf
825  //
826  // The general test form, given a choice of separating axis, is:
827  // sizeA = abs(dot(A.e1,axis)) + abs(dot(A.e2,axis)) + abs(dot(A.e3,axis))
828  // sizeB = abs(dot(B.e1,axis)) + abs(dot(B.e2,axis)) + abs(dot(B.e3,axis))
829  // distance = abs(dot(B.center - A.center),axis))
830  // if distance >= sizeA+sizeB, the boxes are disjoint
831  //
832  // We need to do this test on 15 axes:
833  // x, y, z axis of box A
834  // x, y, z axis of box B
835  // (v1 cross v2) for each v1 in A's axes, for each v2 in B's axes
836  //
837  // Since we're working in a space where A is axis-aligned and A.center=0, many
838  // of the tests and products simplify away.
839 
840  // Check for separation along the axes of box A
841  if (mB_TA.X > hA.X + Math.Abs(hx_B.X) + Math.Abs(hy_B.X) + Math.Abs(hz_B.X))
842  return ContainmentType.Disjoint;
843 
844  if (mB_TA.Y > hA.Y + Math.Abs(hx_B.Y) + Math.Abs(hy_B.Y) + Math.Abs(hz_B.Y))
845  return ContainmentType.Disjoint;
846 
847  if (mB_TA.Z > hA.Z + Math.Abs(hx_B.Z) + Math.Abs(hy_B.Z) + Math.Abs(hz_B.Z))
848  return ContainmentType.Disjoint;
849 
850  // Check for separation along the axes box B, hx_B/hy_B/hz_B
851  if (Math.Abs(Vector3.Dot(mB_T, bX)) > Math.Abs(hA.X * bX.X) + Math.Abs(hA.Y * bX.Y) + Math.Abs(hA.Z * bX.Z) + hB.X)
852  return ContainmentType.Disjoint;
853 
854  if (Math.Abs(Vector3.Dot(mB_T, bY)) > Math.Abs(hA.X * bY.X) + Math.Abs(hA.Y * bY.Y) + Math.Abs(hA.Z * bY.Z) + hB.Y)
855  return ContainmentType.Disjoint;
856 
857  if (Math.Abs(Vector3.Dot(mB_T, bZ)) > Math.Abs(hA.X * bZ.X) + Math.Abs(hA.Y * bZ.Y) + Math.Abs(hA.Z * bZ.Z) + hB.Z)
858  return ContainmentType.Disjoint;
859 
860  // Check for separation in plane containing an axis of box A and and axis of box B
861  //
862  // We need to compute all 9 cross products to find them, but a lot of terms drop out
863  // since we're working in A's local space. Also, since each such plane is parallel
864  // to the defining axis in each box, we know those dot products will be 0 and can
865  // omit them. Note that axis can be zero vector!
866  Vector3 axis;
867 
868  // a.X ^ b.X = (1,0,0) ^ bX
869  axis = new Vector3D(0, -bX.Z, bX.Y);
870  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.Y * axis.Y) + Math.Abs(hA.Z * axis.Z) + Math.Abs(Vector3.Dot(axis, hy_B)) + Math.Abs(Vector3.Dot(axis, hz_B)))
871  return ContainmentType.Disjoint;
872 
873  // a.X ^ b.Y = (1,0,0) ^ bY
874  axis = new Vector3D(0, -bY.Z, bY.Y);
875  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.Y * axis.Y) + Math.Abs(hA.Z * axis.Z) + Math.Abs(Vector3.Dot(axis, hz_B)) + Math.Abs(Vector3.Dot(axis, hx_B)))
876  return ContainmentType.Disjoint;
877 
878  // a.X ^ b.Z = (1,0,0) ^ bZ
879  axis = new Vector3D(0, -bZ.Z, bZ.Y);
880  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.Y * axis.Y) + Math.Abs(hA.Z * axis.Z) + Math.Abs(Vector3.Dot(axis, hx_B)) + Math.Abs(Vector3.Dot(axis, hy_B)))
881  return ContainmentType.Disjoint;
882 
883  // a.Y ^ b.X = (0,1,0) ^ bX
884  axis = new Vector3D(bX.Z, 0, -bX.X);
885  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.Z * axis.Z) + Math.Abs(hA.X * axis.X) + Math.Abs(Vector3.Dot(axis, hy_B)) + Math.Abs(Vector3.Dot(axis, hz_B)))
886  return ContainmentType.Disjoint;
887 
888  // a.Y ^ b.Y = (0,1,0) ^ bY
889  axis = new Vector3D(bY.Z, 0, -bY.X);
890  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.Z * axis.Z) + Math.Abs(hA.X * axis.X) + Math.Abs(Vector3.Dot(axis, hz_B)) + Math.Abs(Vector3.Dot(axis, hx_B)))
891  return ContainmentType.Disjoint;
892 
893  // a.Y ^ b.Z = (0,1,0) ^ bZ
894  axis = new Vector3D(bZ.Z, 0, -bZ.X);
895  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.Z * axis.Z) + Math.Abs(hA.X * axis.X) + Math.Abs(Vector3.Dot(axis, hx_B)) + Math.Abs(Vector3.Dot(axis, hy_B)))
896  return ContainmentType.Disjoint;
897 
898  // a.Z ^ b.X = (0,0,1) ^ bX
899  axis = new Vector3D(-bX.Y, bX.X, 0);
900  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.X * axis.X) + Math.Abs(hA.Y * axis.Y) + Math.Abs(Vector3.Dot(axis, hy_B)) + Math.Abs(Vector3.Dot(axis, hz_B)))
901  return ContainmentType.Disjoint;
902 
903  // a.Z ^ b.Y = (0,0,1) ^ bY
904  axis = new Vector3D(-bY.Y, bY.X, 0);
905  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.X * axis.X) + Math.Abs(hA.Y * axis.Y) + Math.Abs(Vector3.Dot(axis, hz_B)) + Math.Abs(Vector3.Dot(axis, hx_B)))
906  return ContainmentType.Disjoint;
907 
908  // a.Z ^ b.Z = (0,0,1) ^ bZ
909  axis = new Vector3D(-bZ.Y, bZ.X, 0);
910  if (Math.Abs(Vector3.Dot(mB_T, axis)) > Math.Abs(hA.X * axis.X) + Math.Abs(hA.Y * axis.Y) + Math.Abs(Vector3.Dot(axis, hx_B)) + Math.Abs(Vector3.Dot(axis, hy_B)))
911  return ContainmentType.Disjoint;
912 
913  return ContainmentType.Intersects;
914  }

916  // Convert this BoundingOrientedBox to a BoundingFrustum describing the same volume.
917  //
918  // A BoundingFrustum is defined by the matrix that carries its volume to the
919  // box from (-1,-1,0) to (1,1,1), so we just need a matrix that carries our box there.
920  public BoundingFrustumD ConvertToFrustum()
921  {
922  Quaternion invOrientation;
923  Quaternion.Conjugate(ref Orientation, out invOrientation);
924  double sx = 1.0 / HalfExtent.X;
925  double sy = 1.0 / HalfExtent.Y;
926  double sz = 0.5 / HalfExtent.Z;
927  MatrixD temp;
928  MatrixD.CreateFromQuaternion(ref invOrientation, out temp);
929  temp.M11 *= sx; temp.M21 *= sx; temp.M31 *= sx;
930  temp.M12 *= sy; temp.M22 *= sy; temp.M32 *= sy;
931  temp.M13 *= sz; temp.M23 *= sz; temp.M33 *= sz;
932  temp.Translation = Vector3.UnitZ * 0.5f + Vector3D.TransformNormal(-Center, temp);
933 
934  return new BoundingFrustumD(temp);
935  }
