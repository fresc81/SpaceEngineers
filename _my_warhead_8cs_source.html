375  void MarkForExplosion()
376  {
377  m_marked = true;
378  //Large grid = 12.5m radius of block
379  //Small grid = 2.5m radius
380  float radiusMultiplier = 4; //reduced by 20%
381  float warheadBlockRadius = CubeGrid.GridSize * radiusMultiplier;
382 
383  float shrink = 0.85f;
384  m_explosionShrinkenSphere = new BoundingSphereD(PositionComp.GetPosition(), (double)warheadBlockRadius * shrink);
385 
386  m_explosionParticleSphere = new BoundingSphereD(PositionComp.GetPosition(), double.MinValue);
387 
388  MyGamePruningStructure.GetAllEntitiesInSphere(ref m_explosionShrinkenSphere, m_entitiesInShrinkenSphere);
389  m_warheadsInsideCount = 0;
390  foreach (var entity in m_entitiesInShrinkenSphere)
391  {
392  if (entity as MyCubeBlock != null && (entity as MyCubeBlock).CubeGrid.Projector != null)
393  continue;
394  if (Vector3D.DistanceSquared(PositionComp.GetPosition(), entity.PositionComp.GetPosition()) < warheadBlockRadius * shrink * warheadBlockRadius * shrink)
395  {
396  MyWarhead warhead = entity as MyWarhead;
397  if (warhead != null)
398  {
399  m_warheadsInsideCount++;
400 
401  if (!warhead.MarkedToExplode)
402  m_explosionParticleSphere = m_explosionParticleSphere.Include(new BoundingSphereD(warhead.PositionComp.GetPosition(), CubeGrid.GridSize * radiusMultiplier + warhead.CubeGrid.GridSize));
403  }
404  var block = entity as MyCubeBlock;
405  if (block != null)
406  {
407  block.MarkedToExplode = true;
408  }
409  }
410  }
411  m_entitiesInShrinkenSphere.Clear();
412 
413  //m_radius += m_warheadsInsideCount * 0.1f;
414  //Explosion radius is based on linear function where 1 warhead has explosion radius :
415  //Large: 22.4415f
416  // Small: 4.4883f
417  //each warhead contribute 0.26 % of radius
418  //explosion is clamped to maxExplosionRadius
419  float fullExplosionRadius = Math.Min(m_maxExplosionRadius,(1 + 0.024f * m_warheadsInsideCount) * m_warheadDefinition.ExplosionRadius);
420  //fullExplosionRadius = fullExplosionRadius;
421  m_explosionFullSphere = new BoundingSphere(m_explosionParticleSphere.Center, (float)Math.Max(fullExplosionRadius, m_explosionParticleSphere.Radius));
422 
423  if (MyExplosion.DEBUG_EXPLOSIONS)
424  {
425  MyWarheads.DebugWarheadShrinks.Add(m_explosionShrinkenSphere);
