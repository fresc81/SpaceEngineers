168  public async Task<Assembly> Compile(MyApiTarget target, string assemblyName, IEnumerable<Script> scripts, List<Message> messages, bool enableDebugInformation = false)
169  {
170  messages.Clear();
171  CompilationWithAnalyzers analyticCompilation;
172  CSharpCompilation compilation;
173  switch (target)
174  {
175  case MyApiTarget.None:
176  compilation = CreateCompilation(assemblyName, scripts, EnableDebugInformation || enableDebugInformation);
177  await WriteDiagnostics(target, assemblyName, compilation.SyntaxTrees).ConfigureAwait(false);
178  analyticCompilation = null;
179  break;
180 
181  case MyApiTarget.Mod:
182  analyticCompilation = CreateCompilation(assemblyName, scripts, EnableDebugInformation || enableDebugInformation)
183  .WithAnalyzers(ImmutableArray.Create<DiagnosticAnalyzer>(m_modApiWhitelistDiagnosticAnalyzer));
184  await WriteDiagnostics(target, assemblyName, analyticCompilation.Compilation.SyntaxTrees).ConfigureAwait(false);
185  compilation = (CSharpCompilation)analyticCompilation.Compilation;
186  break;
187 
188  case MyApiTarget.Ingame:
189  compilation = CreateCompilation(assemblyName, scripts, EnableDebugInformation || enableDebugInformation);
190  await WriteDiagnostics(target, assemblyName, compilation.SyntaxTrees).ConfigureAwait(false);
191  var newSyntaxTrees = await Task.WhenAll(compilation.SyntaxTrees.Select(tree => InjectInstructionCounter(compilation, tree))).ConfigureAwait(false);
192  await WriteDiagnostics(target, assemblyName, newSyntaxTrees, ".injected").ConfigureAwait(false);
193 
194  analyticCompilation = compilation.RemoveAllSyntaxTrees().AddSyntaxTrees(newSyntaxTrees).WithAnalyzers(ImmutableArray.Create<DiagnosticAnalyzer>(m_ingameWhitelistDiagnosticAnalyzer));
195  compilation = (CSharpCompilation)analyticCompilation.Compilation;
196  break;
197 
198  default:
199  throw new ArgumentOutOfRangeException("target", target, "Invalid compilation target");
200  }
201 
202  using (var assemblyStream = new MemoryStream())
203  {
204  var result = compilation.Emit(assemblyStream);
205  var success = result.Success;
206 
207  //GR: Check for finalizers after sucessfull compilation
208  if (success)
209  {
210  foreach (var script in scripts)
211  {
212  success &= HasFinalizers(script.Code, messages);
213  }
214  }
215 
216  AnalyzeDiagnostics(result.Diagnostics, messages, ref success);
217  if (analyticCompilation != null)
218  {
219  AnalyzeDiagnostics(await analyticCompilation.GetAllDiagnosticsAsync().ConfigureAwait(false), messages, ref success);
220  }
221 
222  await WriteDiagnostics(target, assemblyName, messages, success).ConfigureAwait(false);
223 
224  if (success)
225  {
226  assemblyStream.Seek(0, SeekOrigin.Begin);
227  return Assembly.Load(assemblyStream.ToArray());
228  }
229 
230  return null;
231  }

514  private bool HasFinalizers(string code, List<Message> messages)
515  {
516  var nextIndx = 0;
517  var indx = code.IndexOf('~', nextIndx);
518  while (indx != -1)
519  {
520  nextIndx = NextNonSpaceCharIndx(code, indx + 1);
521  if (code[nextIndx] == '@' || code[nextIndx] == '_' || Char.IsLetter(code[nextIndx]))
522  {
523  while (!WhiteSpaceRelated(code[nextIndx]) && code[nextIndx] != '(') { ++nextIndx; if (nextIndx > code.Length) break; }
524  nextIndx = NextNonSpaceCharIndx(code, nextIndx);
525  if (code[nextIndx] == '(')
526  {
527  nextIndx = NextNonSpaceCharIndx(code, ++nextIndx);
528  if (code[nextIndx] == ')')
529  {
530  nextIndx = NextNonSpaceCharIndx(code, ++nextIndx);
531  if (code[nextIndx] == '{')
532  {
533  messages.Add(new Message(TErrorSeverity.Error, "Error at : " + code.Substring(indx, (code.IndexOf('\n', indx) - indx)) + ". Finalizers are not allowed! Remove any destructors and try again!"));
534  return false;
535  }
536  }
537  }
538  }
539  else

541  indx = code.IndexOf('~', nextIndx);
542  }
