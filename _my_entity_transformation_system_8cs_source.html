466  // Rotation operation performed per update
467  private void PerformRotation()
468  {
469  var ray = CreateRayFromCursorPosition();
470  // Calculate the intersection point
471  var planeIntersectionPoint = m_rotationPlane.Intersection(ref ray.From, ref ray.Direction);
472  // Prevent too small steps
473  if (Vector3D.DistanceSquared(m_rotationStartingPoint, planeIntersectionPoint) < double.Epsilon*20)
474  return;
475 
476  var fromRotationVector = m_rotationStartingPoint - m_gizmoMatrix.Translation;
477  var toRotationVector = planeIntersectionPoint - m_gizmoMatrix.Translation;
478 
479  var rotationMat = MatrixD.CreateFromQuaternion(QuaternionD.CreateFromTwoVectors(fromRotationVector, toRotationVector));
480  var resultMat = m_storedOrientation*rotationMat*m_storedScale;
481  resultMat.Translation = m_storedTranslation;
482 
483  SetWorldMatrix(ref resultMat);
484  }

495  // Drag operation performed per update
496  private void PerformDragg(bool lockToAxis = true)
497  {
498  var ray = CreateRayFromCursorPosition();
499 
500  // Calculate the intersection point
501  var planeIntersectionPoint = m_dragPlane.Intersection(ref ray.From, ref ray.Direction);
502  // Calculate the offset
503  var dragDelta = planeIntersectionPoint - m_dragStartingPoint;
504 
505  if (lockToAxis)
506  {
507  // axis must be unit vector in order to project right
508  var deltaProjection = dragDelta.Dot(ref m_dragAxis);

511  return;

515  worldMat.Translation = m_dragStartingPosition + m_dragAxis*deltaProjection;

523  return;

692  private MyEntity PickEntity()
693  {
694  var hitInfo = MyPhysics.CastRay(m_lastRay.From, m_lastRay.To);
695 
696  m_rayCastResultList.Clear();
697  MyGamePruningStructure.GetAllEntitiesInRay(ref m_lastRay, m_rayCastResultList);
698 
699  // Reorder the entities, that the grids are in the back
700  var frontCursor = 0;
701  for (var index = 0; index < m_rayCastResultList.Count; index++)
702  {
703  var currentElement = m_rayCastResultList[index];
704  if (!(currentElement.Element is MyCubeGrid))
705  {
706  m_rayCastResultList.Swap(frontCursor, index);
707  frontCursor++;
708  }
709  }
710 
711  // No results nothing to do
712  if (m_rayCastResultList.Count == 0 && hitInfo == null)
713  return null;
714 
715  MyEntity foundEntity = null;
716  double overlapDistance = Double.MaxValue;
717  foreach (var overlapResult in m_rayCastResultList)
718  {
719  if (overlapResult.Element.PositionComp.WorldAABB.Intersects(ref m_lastRay, out overlapDistance)
720  && (
721  overlapResult.Element is MyCubeGrid
722  || overlapResult.Element is MyFloatingObject
723  || overlapResult.Element.GetType() == typeof(MyEntity))
724  )
725  {
726  foundEntity = overlapResult.Element;
727  break;
728  }
729  }
730 
731  if (hitInfo.HasValue && Vector3D.Distance(hitInfo.Value.Position, m_lastRay.From) < overlapDistance)
732  {
733  // hit info is better option
734  var hitEntity = hitInfo.Value.HkHitInfo.GetHitEntity();
735  if (hitEntity is MyCubeGrid
736  || hitEntity is MyFloatingObject)
737  {
738  return (MyEntity)hitEntity;
739  }
740  }
741 
742  if (foundEntity == null)
743  {
744  // Just to try to traverse it children again

746  }
747
