408  internal static byte[] Compile(string filepath, ShaderMacro[] macros, MyShaderProfile profile, string sourceDescriptor, bool optimize, bool invalidateCache, out bool wasCached, out string compileLog)
409  {
410  ProfilerShort.Begin("MyShaders.Compile");
411 
412  var globalMacros = GlobalShaderMacros;
413  if (globalMacros.Length != 0)
414  {
415  var macroList = new List<ShaderMacro>();
416  macroList.AddRange(globalMacros);
417  macroList.AddRange(macros);
418  macros = macroList.ToArray();
419  }
420 
421  string function = ProfileEntryPoint(profile);
422  string profileName = ProfileToString(profile);
423 
424  wasCached = false;
425  compileLog = null;
426 
427  ProfilerShort.Begin("MyShaders.Preprocess");
428  string errors;
429  string preprocessedSource = PreprocessShader(filepath, macros, out errors);
430  if (preprocessedSource == null)
431  {
432  compileLog = errors;
433  return null;
434  }
435 
436  // modify preprocessor to be readable for NSight
437  if (MyCompilationSymbols.EnableShaderDebuggingInNSight)
438  {
439  preprocessedSource = Regex.Replace(preprocessedSource, "#line [^\n]*\n", "");
440  }
441 
442  MyShaderIdentity identity = null;
443  if (!invalidateCache && !MyCompilationSymbols.EnableShaderDebuggingInNSight)
444  {
445  identity = MyShaderCache.ComputeShaderIdentity(preprocessedSource, profile);
446  byte[] cached;
447  if (MyShaderCache.TryFetch(identity, out cached))
448  {
449  wasCached = true;
450  ProfilerShort.End();
451  ProfilerShort.End();
452  return cached;
453  }
454  }
455  ProfilerShort.End();
456 
457  try
458  {
459  string descriptor = sourceDescriptor + " " + profile + " " + macros.GetString();
460  CompilationResult compilationResult;
461  if (MyCompilationSymbols.EnableShaderDebuggingInNSight)
462  {
463  if (MyCompilationSymbols.EnableShaderPreprocessorInNSight)
464  compilationResult = ShaderBytecode.Compile(preprocessedSource, function, profileName, 0, 0, macros, new MyIncludeProcessor(filepath));
465  else
466  {
467  compilationResult = ShaderBytecode.CompileFromFile(filepath, function, profileName, 0, 0, macros, new MyIncludeProcessor(filepath));
468  }
469  }
470  else
471  compilationResult = ShaderBytecode.Compile(preprocessedSource, function, profileName,
472  optimize ? ShaderFlags.OptimizationLevel3 : ShaderFlags.None,
473  EffectFlags.None, filepath);
474 
475  if (DUMP_CODE)
476  {
477  var disassembly = compilationResult.Bytecode.Disassemble(DisassemblyFlags.EnableColorCode |
478  DisassemblyFlags.EnableInstructionNumbering);
479  string asmPath;
480  if (MyRender11.DebugMode)
481  {
482  asmPath = Path.GetFileName(descriptor + "__DEBUG.html");
483  }
484  else
485  {
486  asmPath = Path.GetFileName(descriptor + "__O3.html");
487  }
488 
489  using (var writer = new StreamWriter(Path.Combine(MyFileSystem.ContentPath, "ShaderOutput", asmPath)))
490  {
491  writer.Write(disassembly);
492  }
493  }
494 
495  if (compilationResult.Message != null)
496  compileLog = compilationResult.Message;
497 
498  if (!MyCompilationSymbols.EnableShaderDebuggingInNSight && compilationResult.Bytecode != null
499  && compilationResult.Bytecode.Data.Length > 0)
500  MyShaderCache.Store(identity, compilationResult.Bytecode.Data);
501 
502  return compilationResult.Bytecode != null ? compilationResult.Bytecode.Data : null;
503  }
504  catch (Exception e)
505  {
506  Debug.WriteLine(preprocessedSource);
507  compileLog = e.Message;
508  }
509  finally
510  {
511  ProfilerShort.End();
512  }
513  return null;
514  }
