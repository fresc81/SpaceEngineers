48  private void SectorOnContactPoint(int itemId, MyEntity other, ref MyPhysics.MyContactPointEvent e)
49  {
50  // if item is already disabled: puff
51  // We get multiple contact points so this is for that
52  if (m_sector.DataView.Items[itemId].ModelIndex < 0) return;
53 
54  var vel = Math.Abs(e.ContactPointEvent.SeparatingVelocity);
55 
56  if (other == null || other.Physics == null || other is MyFloatingObject) return;
57 
58  if (other is IMyHandheldGunObject<MyDeviceBase>) return;
59 
60  // Prevent debris from breaking trees.
61  // Debris flies in unpredictable ways and this could cause out of sync tree destruction which would is bad.
62  if (other.Physics.RigidBody != null && other.Physics.RigidBody.Layer == MyPhysics.CollisionLayers.DebrisCollisionLayer) return;
63 
64  // On objects held in manipulation tool, Havok returns high velocities, after this contact is fired by contraint solver.
65  // Therefore we disable damage from objects connected by constraint to character
66  if (MyManipulationTool.IsEntityManipulated(other))
67  return;
68 
69  float otherMass = MyDestructionHelper.MassFromHavok(other.Physics.Mass);
70 
71  double impactEnergy = vel * vel * otherMass;
72 
73  // TODO: per item max impact energy
74  if (impactEnergy > ItemResilience(itemId))
75  {
76  BreakAt(itemId, e.Position, e.ContactPointEvent.ContactPoint.Normal, impactEnergy);
77  }
78 
79  // Meteor destroy always
80  if (other is MyMeteor)
81  m_sector.EnableItem(itemId, false);
82  }

116  private void DisableItemAndCreateDebris(ref Impact imp, int itemId)
117  {
118  if (m_sector.DataView.Items[itemId].ModelIndex < 0) return;
119 
120  // Create particle
121  MyParticleEffect effect;
122  if (MyParticlesManager.TryCreateParticleEffect((int)MyParticleEffectsIDEnum.DestructionTree, out effect))
123  {
124  effect.WorldMatrix = MatrixD.CreateTranslation(imp.Position);//, (Vector3D)normal, Vector3D.CalculatePerpendicularVector(normal));
125  }
126 
127  // Spawn no debris for stuff far away
128  if (m_sector.LodLevel <= 1)
129  {
130  // Spawn debris
131  var debris = CreateDebris(itemId);
132  if (debris != null)
133  {
134  var treeMass = debris.Physics.Mass;
135 
136  const float ENERGY_PRESERVATION = .8f;
137 
138  // Tree final velocity
139  float velTree = (float)Math.Sqrt(imp.Energy / treeMass);
140  float accell = velTree / (MyEngineConstants.UPDATE_STEP_SIZE_IN_SECONDS * MyFakes.SIMULATION_SPEED) * ENERGY_PRESERVATION;
141  var force = accell * -imp.Normal;
142 
143  var pos = debris.Physics.CenterOfMassWorld + 0.5f * Vector3D.Dot(imp.Position - debris.Physics.CenterOfMassWorld, debris.WorldMatrix.Up) * debris.WorldMatrix.Up;
144  debris.Physics.AddForce(MyPhysicsForceType.APPLY_WORLD_IMPULSE_AND_WORLD_ANGULAR_IMPULSE, force, pos, null);
145  Debug.Assert(debris.GetPhysicsBody().HavokWorld.ActiveRigidBodies.Contains(debris.Physics.RigidBody));
