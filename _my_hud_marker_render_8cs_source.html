1207  public static bool TryComputeScreenPoint(Vector3D worldPosition, out Vector2 projectedPoint2D, out bool isBehind)
1208  {
1209  // Transform point to camera space, so Z = -1 is always forward and then do projective transformation
1210  Vector3D transformedPoint = Vector3D.Transform(worldPosition, MySector.MainCamera.ViewMatrix);
1211  Vector4D projectedPoint = Vector4D.Transform(transformedPoint, MySector.MainCamera.ProjectionMatrix);
1212 
1213  // If point is behind camera we swap X and Y (this is mirror-like transformation)
1214  if (transformedPoint.Z > 0)
1215  {
1216  projectedPoint.X *= -1;
1217  projectedPoint.Y *= -1;
1218  }
1219 
1220  if (projectedPoint.W == 0)
1221  {
1222  projectedPoint2D = Vector2.Zero;
1223  isBehind = false;
1224  return false;
1225  }
1226 
1227  // Calculate centered coordinates in range <0..1>
1228  projectedPoint2D = new Vector2((float)(projectedPoint.X / projectedPoint.W / 2.0f) + 0.5f, (float)(-projectedPoint.Y / projectedPoint.W) / 2.0f + 0.5f);
1229  if (MyVideoSettingsManager.IsTripleHead())
1230  {
1231  projectedPoint2D.X = (projectedPoint2D.X - (1.0f / 3.0f)) / (1.0f / 3.0f);
1232  }
1233 
1234  // Compute isBehind
1235  Vector3D viewDirection = worldPosition - MyHudMarkerRender.ActiveWorldMatrix.Translation;
1236  viewDirection.Normalize();
1237  double dotProduct = Vector3D.Dot(MySector.MainCamera.ForwardVector, viewDirection);
1238  isBehind = (dotProduct < 0);
1239 
1240  return true;
1241  }
