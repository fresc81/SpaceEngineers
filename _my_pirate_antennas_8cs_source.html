452  private bool SpawnDrone(MyRadioAntenna antenna, long ownerId, Vector3D position, MySpawnGroupDefinition spawnGroup, Vector3? spawnUp = null, Vector3? spawnForward = null) {
453  long antennaEntityId = antenna.EntityId;
454  Vector3D antennaPos = antenna.PositionComp.GetPosition();
455 
456  Vector3D upVector;
457 
458  var planet = MyGamePruningStructure.GetClosestPlanet(position);
459  if (planet != null)
460  {
461  if (!MyGravityProviderSystem.IsPositionInNaturalGravity(antennaPos))
462  {
463  MySandboxGame.Log.WriteLine("Couldn't spawn drone; antenna is not in natural gravity but spawn location is.");
464  return false;
465  }
466  planet.CorrectSpawnLocation(ref position, spawnGroup.SpawnRadius * 2.0);
467  upVector = position - planet.PositionComp.GetPosition();
468  upVector.Normalize();
469  }
470  else
471  {
472  var totalGravityInPoint = MyGravityProviderSystem.CalculateTotalGravityInPoint(position);
473  if (totalGravityInPoint != Vector3.Zero)
474  {
475  upVector = -totalGravityInPoint;
476  upVector.Normalize();
477  }
478  else if (spawnUp != null)
479  upVector = spawnUp.Value;
480  else
481  upVector = MyUtils.GetRandomVector3Normalized();
482  }
483 
484  Vector3D direction = MyUtils.GetRandomPerpendicularVector(ref upVector);
485  if (spawnForward != null)
486  {
487  // Align forward to up vector.
488  Vector3 forward = spawnForward.Value;
489  if (Math.Abs(Vector3.Dot(forward, upVector)) >= 0.98f)
490  {
491  forward = Vector3.CalculatePerpendicularVector(upVector);
492  }
493  else
494  {
495  Vector3 right = Vector3.Cross(forward, upVector);
496  right.Normalize();
497  forward = Vector3.Cross(upVector, right);
498  forward.Normalize();
499  }
500 
501  direction = forward;
502  }
503 
504  MatrixD originMatrix = MatrixD.CreateWorld(position, direction, upVector);
505 
506  foreach (var shipPrefab in spawnGroup.Prefabs)
507  {
508  Vector3D shipPosition = Vector3D.Transform((Vector3D)shipPrefab.Position, originMatrix);
509 
510  List<MyCubeGrid> tmpGridList = new List<MyCubeGrid>();
511 
512  Stack<Action> callback = new Stack<Action>();
513  callback.Push(delegate() { ChangeDroneOwnership(tmpGridList, ownerId, antennaEntityId); });
514 
515  MyPrefabManager.Static.SpawnPrefab(
516  resultList: tmpGridList,
517  prefabName: shipPrefab.SubtypeId,
518  position: shipPosition,
519  forward: direction,
520  up: upVector,
521  initialLinearVelocity: default(Vector3),
522  beaconName: null,
523  spawningOptions: VRage.Game.ModAPI.SpawningOptions.None,
524  ownerId: ownerId,

526  }
