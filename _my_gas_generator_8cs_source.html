230  }
231  }
232 
233  float toProduce = 0f;
234 
235  foreach (var item in items)
236  {
237  if (gasProductionAmount <= 0f)
238  return;
239 
240  var gasContainer = item.Content as MyObjectBuilder_GasContainerObject;
241  if (gasContainer == null || gasContainer.GasLevel >= 1f)
242  continue;

244  var physicalItem = MyDefinitionManager.Static.GetPhysicalItemDefinition(gasContainer) as MyOxygenContainerDefinition;
245  if (physicalItem.StoredGasId != gasId)
246  continue;
247 
248  Debug.Assert(physicalItem != null);
249  float bottleGasAmount = gasContainer.GasLevel * physicalItem.Capacity;
250 
251  float transferredAmount = Math.Min(physicalItem.Capacity - bottleGasAmount, gasProductionAmount);
252  gasContainer.GasLevel = Math.Min((bottleGasAmount + transferredAmount) / physicalItem.Capacity, 1f);
253 
254  toProduce += transferredAmount;
255  gasProductionAmount -= transferredAmount;
256  }
257 
258  if (toProduce > 0f)
