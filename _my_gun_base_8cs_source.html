363  protected void CreateEffects(MyWeaponDefinition.WeaponEffectAction action)
364  {
365  if (dummies != null && dummies.Count > 0 && WeaponProperties.WeaponDefinition.WeaponEffects.Length > 0)
366  {
367  for (int i = 0; i < WeaponProperties.WeaponDefinition.WeaponEffects.Length; i++)
368  {
369  if (WeaponProperties.WeaponDefinition.WeaponEffects[i].Action == action)
370  {
371  MyModelDummy dummy;
372  if (dummies.TryGetValue(WeaponProperties.WeaponDefinition.WeaponEffects[i].Dummy, out dummy))
373  {
374  MyParticleEffect effect;
375  bool add = true;
376  int effectId = -1;
377  MyParticlesLibrary.GetParticleEffectsID(WeaponProperties.WeaponDefinition.WeaponEffects[i].Particle, out effectId);
378  if (WeaponProperties.WeaponDefinition.WeaponEffects[i].Loop)
379  {
380  for (int j = 0; j < m_activeEffects.Count; j++)
381  {
382  if (m_activeEffects[j].Name == dummy.Name && m_activeEffects[j].EffectId == effectId)
383  {
384  add = false;
385  break;
386  }
387  }
388  }
389  if (add && MyParticlesManager.TryCreateParticleEffect(effectId, out effect))
390  {
391  var dummyNormalized = MatrixD.Normalize(dummy.Matrix);
392  effect.WorldMatrix = MatrixD.Multiply(dummyNormalized, WorldMatrix);
393 
394  if (WeaponProperties.WeaponDefinition.WeaponEffects[i].Loop)
395  {
396  m_activeEffects.Add(new WeaponEffect(dummy.Name, dummyNormalized, effectId, action, effect));
397  }
398  }
399  }
400  }
401  }
402  }
