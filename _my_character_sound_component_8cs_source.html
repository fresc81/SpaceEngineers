762  private MyStringHash RayCastGround()
763  {
764  MyStringHash walkSurfaceMaterial = new MyStringHash();
765 
766  float maxDistValue = MyConstants.DEFAULT_GROUND_SEARCH_DISTANCE;
767  var from = m_character.PositionComp.GetPosition() + m_character.PositionComp.WorldMatrix.Up * 0.5; //(needs some small distance from the bottom or the following call to HavokWorld.CastRay will find no hits)
768  var to = from + m_character.PositionComp.WorldMatrix.Down * maxDistValue;
769 
770  MyPhysics.CastRay(from, to, m_hits, MyPhysics.CollisionLayers.CharacterCollisionLayer);
771 
772  // Skips invalid hits (null body, self character)
773  int index = 0;
774  while ((index < m_hits.Count) && ((m_hits[index].HkHitInfo.Body == null) || (m_hits[index].HkHitInfo.GetHitEntity() == Entity.Components)))
775  {
776  index++;
777  }
778 
779  if (m_hits.Count == 0)
780  {
781  if ((m_standingOnGrid != null || m_standingOnVoxel != null) && ShouldUpdateSoundEmitters)
782  {
783  m_standingOnGrid = null;
784  m_standingOnVoxel = null;
785  MyEntity3DSoundEmitter.UpdateEntityEmitters(true, true, false);
786  }
787  else
788  {
789  m_standingOnGrid = null;
790  m_standingOnVoxel = null;
791  }
792  }
793 
794  if (index < m_hits.Count)
795  {
796  // We must take only closest hit (others are hidden behind)
797  var h = m_hits[index];
798  var entity = h.HkHitInfo.GetHitEntity();
799 
800  var sqDist = Vector3D.DistanceSquared((Vector3D)h.Position, from);
801  if (sqDist < maxDistValue * maxDistValue)
802  {
803  var cubeGrid = entity as MyCubeGrid;
804  var voxelBase = entity as MyVoxelBase;
805 
806  if (((cubeGrid != null && m_standingOnGrid != cubeGrid) || (voxelBase != null && m_standingOnVoxel != voxelBase)) && ShouldUpdateSoundEmitters)
807  {
808  m_standingOnGrid = cubeGrid;
809  m_standingOnVoxel = voxelBase;
810  MyEntity3DSoundEmitter.UpdateEntityEmitters(true, true, true);
811  }
812  else
813  {
814  m_standingOnGrid = cubeGrid;
815  m_standingOnVoxel = voxelBase;
816  }
817  if(cubeGrid != null || voxelBase != null)
818  m_jumpReady = true;
819 
820  if (cubeGrid != null)
821  walkSurfaceMaterial = cubeGrid.Physics.GetMaterialAt(h.Position + m_character.PositionComp.WorldMatrix.Down * 0.1f);
822  else if (voxelBase != null && voxelBase.Storage != null && voxelBase.Storage.DataProvider != null)
823  {
824  var materialDefinition = voxelBase.GetMaterialAt(ref h.Position);
825  if (materialDefinition != null)
826  walkSurfaceMaterial = MyStringHash.GetOrCompute(materialDefinition.MaterialTypeName);
827  }
828  if (walkSurfaceMaterial.ToString().Length == 0)
829  walkSurfaceMaterial = MyMaterialType.ROCK;
830  }
831  }
832 
833  m_hits.Clear();
834 
835  return walkSurfaceMaterial;
836  }
