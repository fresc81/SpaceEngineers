410  if (itemCount > 0)
411  {
412  int itemsCreated = TryCreatingItemsByMerge(group, j, itemCount);
413  requiredItems[j] = itemCount - itemsCreated;
414  m_solvedItemCounter += itemsCreated;
415  }
416  }
417 
418  // Now all of the required items are bigger than available items (otherwise, they'd be created in matching or splitting phase)

435  private int TryCreatingItemsBySplit(MyComponentGroupDefinition group, int itemValue, int itemCount)
436  {
437  int createdCount = 0;
438  // Avoid making too much "mess" by trying to find the best-fitting items first
439  for (int k = itemValue + 1; k <= group.GetComponentNumber(); k++)
440  {
441  int fitNumber = k / itemValue; // How many items fit into k-item
442 
443  int wholeCount = itemCount / fitNumber; // How many k-items will be fully split
444 
445  int partialFitNumber = itemCount % fitNumber; // How many items will be created from the last k-item
446  int partialCount = partialFitNumber == 0 ? 0 : 1; // How many k-items will be partially split (either 0 or 1 that is)
447 
448  int removedCount = TryRemovePresentItems(k, wholeCount + partialCount);
449  if (removedCount > 0)
450  {
451  int removedWholeCount = Math.Min(removedCount, wholeCount);
452  if (removedWholeCount != 0)
453  {
454  int created = SplitHelper(group, k, itemValue, removedWholeCount, fitNumber);
455  createdCount += created;
456  itemCount -= created;
457  }
458 
459  if (removedCount - wholeCount > 0)

462  int created = SplitHelper(group, k, itemValue, 1, partialFitNumber);

464  itemCount -= created;
465  }

501

503  for (int i = 0; i < itemCount; ++i)
504  {
505  // What remains to be found to create this item
506  int remainder = itemValue;
507
