423  public ScriptTerminationReason RunSandboxedProgramAction(Action<ModAPI.IMyGridProgram> action, out string response)
424  {
425  if (m_isRunning) {
426  response = MyTexts.GetString(MySpaceTexts.ProgrammableBlock_Exception_AllreadyRunning);
427  return ScriptTerminationReason.AlreadyRunning;
428  }
429  if (m_terminationReason != ScriptTerminationReason.None) {
430  response = DetailedInfo.ToString();
431  return m_terminationReason;
432  }
433  DetailedInfo.Clear();
434  m_echoOutput.Clear();
435  if (m_assembly == null) {
436  response = MyTexts.GetString(MySpaceTexts.ProgrammableBlock_Exception_NoAssembly);
437  return ScriptTerminationReason.NoScript;
438  }
439  var gridGroup = MyCubeGridGroups.Static.Logical.GetGroup(CubeGrid);
440  var terminalSystem = gridGroup.GroupData.TerminalSystem;
441  terminalSystem.UpdateGridBlocksOwnership(this.OwnerId);
442  m_instance.GridTerminalSystem = terminalSystem;
443 
444  m_isRunning = true;
445  response = "";
446 #if !XB1 // XB1_NOILINJECTOR
447  try {
448  using (var handle = IlInjector.BeginRunBlock(MAX_NUM_EXECUTED_INSTRUCTIONS, MAX_NUM_METHOD_CALLS)) {
449  m_runtime.InjectorHandle = handle;
450  action(m_instance);
451  }
452  if (m_echoOutput.Length > 0) {
453  response = m_echoOutput.ToString();
454  }
455  return m_terminationReason;
456  } catch (Exception ex) {
457  // Unwrap the exception if necessary
458  if (ex is TargetInvocationException) {
459  ex = ex.InnerException;
460  }
461 
462  // Since we just had an exception I'm not fussed about using old
463  // fashioned string concatenation here. We'll still want the echo
464  // output, since its primary purpose is debugging.
465  if (m_echoOutput.Length > 0) {
466  response = m_echoOutput.ToString();
467  }
468  if (ex is ScriptOutOfRangeException) {
469  if (IlInjector.IsWithinRunBlock()) {
470  // If we're within a nested run, we don't reset the program, we just pass the error
471  response += MyTexts.GetString(MySpaceTexts.ProgrammableBlock_Exception_NestedTooComplex);
472  return ScriptTerminationReason.InstructionOverflow;
473  } else {
474  response += MyTexts.GetString(MySpaceTexts.ProgrammableBlock_Exception_TooComplex);
475  OnProgramTermination(ScriptTerminationReason.InstructionOverflow);
476  }
477  } else {
478  response += MyTexts.GetString(MySpaceTexts.ProgrammableBlock_Exception_ExceptionCaught) + ex.Message;
479  OnProgramTermination(ScriptTerminationReason.RuntimeException);
480  }
481  return m_terminationReason;
482  } finally {
483  m_runtime.InjectorHandle = null;
484  m_isRunning = false;
485  }
486 #else // XB1
487  System.Diagnostics.Debug.Assert(false, "No scripts on XB1!");
488  return m_terminationReason;
489 #endif // XB1
490  }

719  private void EchoTextToDetailInfo(string line)
720  {
721  line = line ?? string.Empty;
722  var lineLength = line.Length + 1; // line length + lineshift
723  if (lineLength > MAX_ECHO_LENGTH)
724  {
725  // If the input line is already longer than the maximum allowed length,
726  // we clear the current output and add only allowed portion of the string
727  // to the output. Obviously this is unlikely to happen but it could.
728  m_echoOutput.Clear();
729  line = line.Substring(0, MAX_ECHO_LENGTH);
730  }
731 
732  // Now we make sure the addition of this new line does not overshoot the
733  // maximum size by removing any excess amount of characters from the beginning
734  // of the stream.
735  var newLength = m_echoOutput.Length + lineLength;
736  if (newLength > MAX_ECHO_LENGTH)
737  {
738  m_echoOutput.Remove(0, newLength - MAX_ECHO_LENGTH);
739  }
740 
741  // Append the new line.
742  m_echoOutput.Append(line);
743  m_echoOutput.Append('\n');
