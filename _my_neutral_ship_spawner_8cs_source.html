165  {
166  double tolerance = 10.0f;
167  BoundingSphereD sphere = new BoundingSphereD(start, tolerance);
168 
169  var players = MySession.Static.Players.GetOnlinePlayers();
170  bool tryIncludeOtherPlayers = true;
171 
172  // We have to try adding other players until the bounding sphere stays the same
173  while (tryIncludeOtherPlayers)
174  {
175  tryIncludeOtherPlayers = false;
176  foreach (var player in players)
177  {
178  Vector3D playerPosition = player.GetPosition();
179  double distanceFromSphere = (sphere.Center - playerPosition).Length() - sphere.Radius;
180 
181  if (distanceFromSphere <= 0.0) continue;
182  if (distanceFromSphere > spawnDistance * 2.0f) continue;
183 
184  sphere.Include(new BoundingSphereD(playerPosition, tolerance));
185  tryIncludeOtherPlayers = true;
186  }
187  }
188 
189  sphere.Radius += spawnDistance;
190  output = new BoundingBoxD(sphere.Center - new Vector3D(sphere.Radius), sphere.Center + new Vector3D(sphere.Radius));
191 
192  var entities = MyEntities.GetEntitiesInAABB(ref output);
193  foreach (var entity in entities)
194  {
195  if (entity is MyCubeGrid)
196  {
197  var cubeGrid = entity as MyCubeGrid;
198  if (cubeGrid.IsStatic)
199  {
200  Vector3D gridPosition = cubeGrid.PositionComp.GetPosition();
201 
202  // If grid is close to picked player we need to include it's "safe" bounding box for spawning ships,
203  // so cargo ships don't spawn near it.
204 
205  output.Include(new BoundingBoxD(new Vector3D(gridPosition - spawnDistance), new Vector3D(gridPosition + spawnDistance)));
206  }
