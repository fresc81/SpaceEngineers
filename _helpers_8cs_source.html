104  public static void Sort(int[] keys, object[] values)
105  {
106  // bubble-sort; it'll work on MF, has small code,
107  // and works well-enough for our sizes. This approach
108  // also allows us to do `int` compares without having
109  // to go via IComparable etc, so win:win
110  bool swapped;
111  do {
112  swapped = false;
113  for (int i = 1; i < keys.Length; i++) {
114  if (keys[i - 1] > keys[i]) {
115  int tmpKey = keys[i];
116  keys[i] = keys[i - 1];
117  keys[i - 1] = tmpKey;
118  object tmpValue = values[i];
119  values[i] = values[i - 1];
120  values[i - 1] = tmpValue;
121  swapped = true;
122  }
123  }
124  } while (swapped);
125  }
