126  private void PlaceAreaMarker()
127  {
128  Vector3D cameraPos, cameraDir;
129 
130  if (MySession.Static.GetCameraControllerEnum() == MyCameraControllerEnum.ThirdPersonSpectator || MySession.Static.GetCameraControllerEnum() == MyCameraControllerEnum.Entity)
131  {
132  var headMatrix = MySession.Static.ControlledEntity.GetHeadMatrix(true, true);
133  cameraPos = headMatrix.Translation;
134  cameraDir = headMatrix.Forward;
135  }
136  else
137  {
138  cameraPos = MySector.MainCamera.Position;
139  cameraDir = MySector.MainCamera.WorldMatrix.Forward;
140  }
141 
142  List<MyPhysics.HitInfo> hitInfos = new List<MyPhysics.HitInfo>();
143 
144  MyPhysics.CastRay(cameraPos, cameraPos + cameraDir * 100, hitInfos, MyPhysics.CollisionLayers.ObjectDetectionCollisionLayer);
145  if (hitInfos.Count == 0)
146  return;
147 
148  MyPhysics.HitInfo? closestValidHit = null;
149  foreach (var hitInfo in hitInfos)
150  {
151  var ent = hitInfo.HkHitInfo.GetHitEntity();
152  if (ent is MyCubeGrid)
153  {
154  closestValidHit = hitInfo;
155  break;
156  }
157  else if (ent is MyVoxelMap)
158  {
159  closestValidHit = hitInfo;
160  break;
161  }
162  }
163 
164  if (closestValidHit.HasValue)
165  {
166  MyAreaMarkerDefinition definition = AreaMarkerDefinition;
167  Debug.Assert(definition != null, "Area marker definition cannot be null!");
168  if (definition == null) return;
169 
170  Vector3D position = closestValidHit.Value.Position;
171 
172  var forward = Vector3D.Reject(cameraDir, Vector3D.Up);
173 
174  if (Vector3D.IsZero(forward))
175  forward = Vector3D.Forward;
176 
177  var positionAndOrientation = new MyPositionAndOrientation(position, Vector3D.Normalize(forward), Vector3D.Up);
178 
179  MyObjectBuilder_AreaMarker objectBuilder = (MyObjectBuilder_AreaMarker)MyObjectBuilderSerializer.CreateNewObject(definition.Id);
180  objectBuilder.PersistentFlags = MyPersistentEntityFlags2.Enabled | MyPersistentEntityFlags2.InScene;
181  objectBuilder.PositionAndOrientation = positionAndOrientation;
182 
183  if (objectBuilder.IsSynced)
184  {
185  SerializableDefinitionId id = definition.Id;
186  MyMultiplayer.RaiseStaticEvent(x => CreateNewPlaceArea, id,positionAndOrientation);
187  }
188  else
189  {
190  MyAreaMarker flag = MyEntityFactory.CreateEntity<MyAreaMarker>(objectBuilder);
191  flag.Init(objectBuilder);
192 
193  MyEntities.Add(flag);
194  }
195  }
