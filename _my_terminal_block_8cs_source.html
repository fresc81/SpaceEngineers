406  protected void FixSingleInventory()
407  {
408  MyInventoryBase inventoryBase;
409  if (!Components.TryGet<MyInventoryBase>(out inventoryBase))
410  return;
411  MyInventoryAggregate aggregate = inventoryBase as MyInventoryAggregate;
412  MyInventory bestInventory = null;
413  if (aggregate != null)
414  {
415  foreach (var inventory in aggregate.ChildList.Reader)
416  {
417  var myInventory = inventory as MyInventory;
418  if (myInventory == null)
419  continue;
420  if (bestInventory == null)
421  {
422  bestInventory = myInventory;
423  }
424  else if (bestInventory.GetItemsCount() < myInventory.GetItemsCount())
425  {
426  bestInventory = myInventory;
427  }
428  }
429  }
430  if (bestInventory != null)
431  {
432  Components.Remove<MyInventoryBase>();
433  Components.Add<MyInventoryBase>(bestInventory);
434  }
435  }
