242  private void ConsumeFuel(int timeDelta)
243  {
244  if (!SourceComp.HasCapacityRemaining)
245  return;
246  if (SourceComp.CurrentOutput == 0.0f)
247  return;
248 
249  float consumptionPerMillisecond = SourceComp.CurrentOutput / (SourceComp.ProductionToCapacityMultiplier * 1000);
250  consumptionPerMillisecond /= m_reactorDefinition.FuelDefinition.Mass; // Convert weight to number of items
251 
252  MyFixedPoint consumedFuel = (MyFixedPoint)(timeDelta * consumptionPerMillisecond);
253  if (consumedFuel == 0)
254  {
255  consumedFuel = MyFixedPoint.SmallestPossibleValue;
256  }
257 
258  if (this.GetInventory().ContainItems(consumedFuel, m_reactorDefinition.FuelId))
259  {
260  this.GetInventory().RemoveItemsOfType(consumedFuel, m_reactorDefinition.FuelId);
261  }
262  else if (MyFakes.ENABLE_INFINITE_REACTOR_FUEL)
263  {
264  this.GetInventory().AddItems((MyFixedPoint)(200 / m_reactorDefinition.FuelDefinition.Mass), m_reactorDefinition.FuelItem);
265  }
266  else
267  {
268  var amountAvailable = this.GetInventory().GetItemAmount(m_reactorDefinition.FuelId);
269  this.GetInventory().RemoveItemsOfType(amountAvailable, m_reactorDefinition.FuelId);
270  }
