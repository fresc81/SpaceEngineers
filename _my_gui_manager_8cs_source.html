666  // This method scales picture to bounding area according to bounding area's height. We don't care about width, so sometimes if picture has wide
667  // aspect ratio, borders of image may be outisde of the bounding area. This method is used when we want to have picture covering whole screen
668  // and don't care if part of picture is invisible. Also, aspect ration is unchanged too.
669  static void GetSafeHeightPictureSize(Vector2I originalSize, Rectangle boundingArea, out Rectangle outRect)
670  {
671  outRect.Height = boundingArea.Height;
672  outRect.Width = (int)(((float)outRect.Height / (float)originalSize.Y) * originalSize.X);
673  outRect.X = boundingArea.Left + (boundingArea.Width - outRect.Width) / 2;
674  outRect.Y = boundingArea.Top + (boundingArea.Height - outRect.Height) / 2;
675  }
676 
677  // Return picture size that is safe for displaying in bounding area and doesn't distort the aspect ratio of original picture or bounding area.
678  // Example: picture of video frame is scaled to size of screen, so it fits it as much as possible, but still maintains aspect ration of
679  // original video frame or screen. So if screen's heigh is not enouch, video frame is scaled down to fit height, but also width is scaled
680  // according to original video frame.
681  // It's used whenever we need to scale picture/texture to some area, usually to screen size.
682  // Also this method calculated left/top coordinates, so it's always centered.
683  static void GetSafeAspectRatioPictureSize(Vector2I originalSize, Rectangle boundingArea, out Rectangle outRect)
684  {
685  outRect.Width = boundingArea.Width;
686  outRect.Height = (int)(((float)outRect.Width / (float)originalSize.X) * originalSize.Y);
687 
688  if (outRect.Height > boundingArea.Height)
689  {
690  outRect.Height = boundingArea.Height;
691  outRect.Width = (int)(outRect.Height * ((float)originalSize.X / (float)originalSize.Y));
692  }
693 
694  outRect.X = boundingArea.Left + (boundingArea.Width - outRect.Width) / 2;
695  outRect.Y = boundingArea.Top + (boundingArea.Height - outRect.Height) / 2;
696  }
