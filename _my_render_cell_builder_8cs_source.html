342  private void ProcessLowMesh(MyIsoMesh highResMesh, MyIsoMesh lowResMesh)
343  {
344  ProfilerShort.Begin("ProcessLowMesh");
345  if (lowResMesh != null)
346  {
347  /*
348  * Derived transformation of normalized coordinates from low res mesh to high res mesh.
349  * _l is for low res, _h for high res, x is vertex position
350  x = x_l * scale_l + offset_l
351  x_h = (x - offset_h) / scale_h
352  x_h = ((x_l * scale_l + offset_l) - offset_h) / scale_h
353  x_h = (x_l * scale_l + offset_l - offset_h) / scale_h
354  x_h = x_l * (scale_l / scale_h) + ((offset_l - offset_h) / scale_h)
355  */
356  Vector3 morphOffset, morphScale;
357  morphScale = lowResMesh.PositionScale / highResMesh.PositionScale;
358  morphOffset = (Vector3)(lowResMesh.PositionOffset - highResMesh.PositionOffset) / highResMesh.PositionScale;
359  for (int i = 0; i < lowResMesh.VerticesCount; ++i)
360  {
361  var vertex = new Vertex
362  {
363  Target = new MorphData
364  {
365  Position = lowResMesh.Positions[i] * morphScale + morphOffset,
366  Normal = lowResMesh.Normals[i],
367  Material = lowResMesh.Materials[i],
368  Ambient = lowResMesh.Ambient[i],
369  },
370  Cell = lowResMesh.Cells[i],
371  };
372 
373  m_lowVertices.Add(vertex);
374  m_morphMap[vertex.Cell] = vertex;
375  }
376  }
377  ProfilerShort.End();
