99  protected override void EmitRead(ProtoBuf.Compiler.CompilerContext ctx, ProtoBuf.Compiler.Local valueFrom)
100  {
101  /* This looks more complex than it is. Look at the non-compiled Read to
102  * see what it is trying to do, but note that it needs to cope with a
103  * few more scenarios. Note that it picks the **most specific** Add,
104  * unlike the runtime version that uses IList when possible. The core
105  * is just a "do {list.Add(readValue())} while {thereIsMore}"
106  *
107  * The complexity is due to:
108  * - value types vs reference types (boxing etc)
109  * - initialization if we need to pass in a value to the tail
110  * - handling whether or not the tail *returns* the value vs updates the input
111  */
112  bool returnList = ReturnList;
113  using (Compiler.Local list = AppendToCollection ? ctx.GetLocalWithValue(ExpectedType, valueFrom) : new Compiler.Local(ctx, declaredType))
114  using (Compiler.Local origlist = (returnList && AppendToCollection) ? new Compiler.Local(ctx, ExpectedType) : null)
115  {
116  if (!AppendToCollection)
117  { // always new
118  ctx.LoadNullRef();
119  ctx.StoreValue(list);
120  }
121  else if (returnList)
122  { // need a copy
123  ctx.LoadValue(list);
124  ctx.StoreValue(origlist);
125  }
126  if (concreteType != null)
127  {
128  ctx.LoadValue(list);
129  Compiler.CodeLabel notNull = ctx.DefineLabel();
130  ctx.BranchIfTrue(notNull, true);
131  ctx.EmitCtor(concreteType);
132  ctx.StoreValue(list);
133  ctx.MarkLabel(notNull);
134  }
135 
136  EmitReadList(ctx, list, Tail, add, packedWireType);
137 
138  if (returnList)
139  {
140  if (AppendToCollection)
141  {
142  // remember ^^^^ we had a spare copy of the list on the stack; now we'll compare
143  ctx.LoadValue(origlist);
144  ctx.LoadValue(list); // [orig] [new-value]
145  Compiler.CodeLabel sameList = ctx.DefineLabel(), allDone = ctx.DefineLabel();
146  ctx.BranchIfEqual(sameList, true);
147  ctx.LoadValue(list);
148  ctx.Branch(allDone, true);
149  ctx.MarkLabel(sameList);
150  ctx.LoadNullRef();

152  }

217  private static void EmitReadAndAddItem(Compiler.CompilerContext ctx, Compiler.Local list, IProtoSerializer tail, MethodInfo add)
218  {
219  ctx.LoadValue(list);
220  Type itemType = tail.ExpectedType;
221  if (tail.RequiresOldValue)
222  {
223  if (itemType.IsValueType || !tail.ReturnsValue)
224  {
225  // going to need a variable
226  using (Compiler.Local item = new Compiler.Local(ctx, itemType))
227  {
228  if (itemType.IsValueType)
229  { // initialise the struct
230  ctx.LoadAddress(item, itemType);
231  ctx.EmitCtor(itemType);
232  }
233  else
234  { // assign null
235  ctx.LoadNullRef();
236  ctx.StoreValue(item);
237  }
238  tail.EmitRead(ctx, item);
239  if (!tail.ReturnsValue) { ctx.LoadValue(item); }
240  }
241  }
242  else
243  { // no variable; pass the null on the stack and take the value *off* the stack
244  ctx.LoadNullRef();
245  tail.EmitRead(ctx, null);
246  }
247  }
248  else
249  {
250  if (tail.ReturnsValue)
251  { // out only (on the stack); just emit it
252  tail.EmitRead(ctx, null);
253  }
254  else
255  { // doesn't take anything in nor return anything! WTF?
256  throw new InvalidOperationException();
257  }
258  }
259  // our "Add" is chosen either to take the correct type, or to take "object";
260  // we may need to box the value
261 
262  Type addParamType = add.GetParameters()[0].ParameterType;
263  if(addParamType != itemType) {
264  if (addParamType == ctx.MapType(typeof(object)))
265  {
266  ctx.CastToObject(itemType);
267  }
268 #if !NO_GENERICS
269  else if(Helpers.GetUnderlyingType(addParamType) == itemType)
270  { // list is nullable
271  ConstructorInfo ctor = Helpers.GetConstructor(addParamType, new Type[] {itemType}, false);
272  ctx.EmitCtor(ctor); // the itemType on the stack is now a Nullable<ItemType>
273  }
274 #endif
275  else
276  {
277  throw new InvalidOperationException("Conflicting item/add type");
278  }
279  }
280  ctx.EmitCall(add);
281  if (add.ReturnType != ctx.MapType(typeof(void)))
282  {
283  ctx.DiscardValue();
284  }
285  }
286 #endif
