332  private void CalculateNextCue()
333  {
334  if (MySession.Static == null || MySession.Static.LocalCharacter == null)
335  return;
336  m_noMusicTimer = m_random.Next(DEFAULT_NO_MUSIC_TIME_MIN, DEFAULT_NO_MUSIC_TIME_MAX);
337 
338  //planet or space
339  Vector3D pos = MySession.Static.LocalCharacter.PositionComp.GetPosition();
340  MyPlanet planet = MyGamePruningStructure.GetClosestPlanet(pos);
341  var grav = planet!= null ? planet.Components.Get<MyGravityProviderComponent>() as MySphericalNaturalGravityComponent : null;
342  if (planet != null && (grav != null) && Vector3D.Distance(pos, planet.PositionComp.GetPosition()) <= grav.GravityLimit * 0.65f)
343  {
344  if (planet != m_lastVisitedPlanet)
345  {
346  m_lastVisitedPlanet = planet;
347  if (planet.Generator.MusicCategories != null && planet.Generator.MusicCategories.Count > 0)
348  {
349  m_actualMusicOptions.Clear();
350  foreach (var option in planet.Generator.MusicCategories)
351  m_actualMusicOptions.Add(new MusicOption(option.Category, option.Frequency));
352  }
353  else
354  m_actualMusicOptions = m_defaultPlanetCategory;
355  }
356  }
357  else
358  {
359  m_lastVisitedPlanet = null;
360  m_actualMusicOptions = m_defaultSpaceCategories;
361  }
362 
363  //choose category based on frequency
364  float sum = 0f;
365  foreach (var option in m_actualMusicOptions)
366  {
367  sum += Math.Max(option.Frequency, 0f);
368  }
369  float r = (float)m_random.NextDouble() * sum;
370  MyStringId selected = m_actualMusicOptions[0].Category;
371  for (int i = 0; i < m_actualMusicOptions.Count; i++)
372  {
373  if (r <= m_actualMusicOptions[i].Frequency)
374  {
375  selected = m_actualMusicOptions[i].Category;
376  break;
377  }
378  else
379  r -= m_actualMusicOptions[i].Frequency;
380  }
381 
382  //pick from cue list and play
383  CueIdPlaying = SelectCueFromCategory(selected);
384  CategoryPlaying = selected;
385 
386  if (CueIdPlaying == m_cueEmpty)
387  return;
388 
389  PlayMusic(CueIdPlaying, MyStringHash.NullOrEmpty);
390  m_currentMusicCategory = MusicCategory.location;
391  }
