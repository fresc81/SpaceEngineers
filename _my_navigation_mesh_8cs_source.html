181  private void GetEdgeVerticesSafe(MyNavigationTriangle triangle, int edgeIndex, out Vector3 left, out Vector3 right)
182  {
183  triangle.GetEdgeVertices(edgeIndex, out left, out right);
184 
185  float d = (left - right).LengthSquared();
186 
187  bool leftDangerous = triangle.IsEdgeVertexDangerous(edgeIndex, predVertex: true);
188  bool rightDangerous = triangle.IsEdgeVertexDangerous(edgeIndex, predVertex: false);
189 
190  m_segmentDangerous |= leftDangerous | rightDangerous;
191 
192  if (leftDangerous)
193  {
194  if (rightDangerous)
195  {
196  if (SAFE_DISTANCE2_SQ > d)
197  {
198  left = (left + right) * 0.5f;
199  right = left;
200  }
201  else
202  {
203  float t = SAFE_DISTANCE / (float)Math.Sqrt(d);
204  Vector3 newLeft = right * t + left * (1.0f - t);
205  right = left * t + right * (1.0f - t);
206  left = newLeft;
207  }
208  }
209  else
210  {
211  if (SAFE_DISTANCE_SQ > d)
212  {
213  left = right;
214  }
215  else
216  {
217  float t = SAFE_DISTANCE / (float)Math.Sqrt(d);
218  left = right * t + left * (1.0f - t);
219  }
220  }
221  }
222  else

225  {
226  if (SAFE_DISTANCE_SQ > d)

228  right = left;

360  // Adds a triangle and connects it to the other triangles in the mesh by the given edges.
361  // Because connecting by vertices would produce non-manifold meshes, we connect triangles by their edges.
362  // When a triangle is added that connects to another triangle only by a vertex, the two touching vertices will be regarded
363  // as two separate vertices and they will only be merged when another triangle is added that shares two edges with the two
364  // original triangles.
365  //
366  // When the method returns, edgeAB, edgeBC and edgeCA will contain the indices for the new edges.
367  // The positions of vertices A, B and C will be unmodified and might not correspond to the real positions of the resulting
368  // triangle's vertices (because the positions of the original edge vertices will sometimes be used due to edge merging).
369  //
370  // Note: The triangle's vertices and edges must be ordered clockwise:
371  // B
372  // / \
373  // / \
374  // AB BC
375  // / \
376  // / \
377  // A -- CA --- C
378  protected MyNavigationTriangle AddTriangle(ref Vector3 A, ref Vector3 B, ref Vector3 C, ref int edgeAB, ref int edgeBC, ref int edgeCA)
379  {
380  MyNavigationTriangle newTri = m_triPool.Allocate();
381 
382  // There are several cases that need to be handled and they can be distinguished by the number of
383  // existing edges and their shared vertices.
384 
385  int newEdgeCount = 0;
386  newEdgeCount += edgeAB == -1 ? 1 : 0;
387  newEdgeCount += edgeBC == -1 ? 1 : 0;
388  newEdgeCount += edgeCA == -1 ? 1 : 0;
389 
390  int newTriIndex = -1;
391  if (newEdgeCount == 3)
392  {
393  newTriIndex = m_mesh.MakeNewTriangle(newTri, ref A, ref B, ref C, out edgeAB, out edgeBC, out edgeCA);
394  }
395  else if (newEdgeCount == 2)
396  {
397  if (edgeAB != -1)
398  newTriIndex = m_mesh.ExtrudeTriangleFromEdge(ref C, edgeAB, newTri, out edgeBC, out edgeCA);
399  else if (edgeBC != -1)
400  newTriIndex = m_mesh.ExtrudeTriangleFromEdge(ref A, edgeBC, newTri, out edgeCA, out edgeAB);
401  else
402  newTriIndex = m_mesh.ExtrudeTriangleFromEdge(ref B, edgeCA, newTri, out edgeAB, out edgeBC);
403  }
404  else if (newEdgeCount == 1)
405  {
406  if (edgeAB == -1)
407  newTriIndex = GetTriangleOneNewEdge(ref edgeAB, ref edgeBC, ref edgeCA, newTri);
408  else if (edgeBC == -1)
409  newTriIndex = GetTriangleOneNewEdge(ref edgeBC, ref edgeCA, ref edgeAB, newTri);
410  else
411  newTriIndex = GetTriangleOneNewEdge(ref edgeCA, ref edgeAB, ref edgeBC, newTri);
412  }
413  else
414  {
415  var entryAB = m_mesh.GetEdge(edgeAB);
416  var entryBC = m_mesh.GetEdge(edgeBC);
417  var entryCA = m_mesh.GetEdge(edgeCA);
418  int sharedA = entryCA.TryGetSharedVertex(ref entryAB);
419  int sharedB = entryAB.TryGetSharedVertex(ref entryBC);
420  int sharedC = entryBC.TryGetSharedVertex(ref entryCA);
421 
422  int sharedVertCount = 0;
423  sharedVertCount += sharedA == -1 ? 0 : 1;
424  sharedVertCount += sharedB == -1 ? 0 : 1;
425  sharedVertCount += sharedC == -1 ? 0 : 1;
426 
427  if (sharedVertCount == 3)
428  {
429  newTriIndex = m_mesh.MakeFace(newTri, edgeAB);
430  }
431  else if (sharedVertCount == 2)
432  {
433  if (sharedA == -1)
434  newTriIndex = GetTriangleTwoSharedVertices(edgeAB, edgeBC, ref edgeCA, sharedB, sharedC, newTri);
435  else if (sharedB == -1)
436  newTriIndex = GetTriangleTwoSharedVertices(edgeBC, edgeCA, ref edgeAB, sharedC, sharedA, newTri);
437  else
438  newTriIndex = GetTriangleTwoSharedVertices(edgeCA, edgeAB, ref edgeBC, sharedA, sharedB, newTri);
439  }
440  else if (sharedVertCount == 1)
441  {
442  if (sharedA != -1)
443  newTriIndex = GetTriangleOneSharedVertex(edgeCA, edgeAB, ref edgeBC, sharedA, newTri);
444  else if (sharedB != -1)
445  newTriIndex = GetTriangleOneSharedVertex(edgeAB, edgeBC, ref edgeCA, sharedB, newTri);
446  else
447  newTriIndex = GetTriangleOneSharedVertex(edgeBC, edgeCA, ref edgeAB, sharedC, newTri);
448  }
449  else
450  {
451  int next, prev;
452  newTriIndex = m_mesh.ExtrudeTriangleFromEdge(ref C, edgeAB, newTri, out next, out prev);
453  m_mesh.MergeEdges(prev, edgeCA);
454  m_mesh.MergeEdges(next, edgeBC);
455  }
456  }
457 
458  newTri.Init(this, newTriIndex);
459  return newTri;
460  }
