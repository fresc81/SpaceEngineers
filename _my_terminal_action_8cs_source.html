186  public void RequestParameterCollection(IList<TerminalActionParameter> parameters, Action<bool> callback)
187  {
188  if (parameters == null)
189  {
190  throw new ArgumentException("parameters");
191  }
192  if (callback == null)

194  throw new ArgumentNullException("callback");
195  }
196  var blockCallback = DoUserParameterRequest;
197  var myParameters = ParameterDefinitions;
198  parameters.Clear();
199  // Fill the provided parameters list with available parameters and their default values.
200  foreach (var parameter in myParameters)
201  parameters.Add(parameter);
202 
203  if (blockCallback == null)
204  {
205  // We have no callback, so we simply call back with the default values.
206  // This is not considered a cancel.
207  callback(true);
208  return;
209  }
210 
211  blockCallback(parameters, callback);
212  }
