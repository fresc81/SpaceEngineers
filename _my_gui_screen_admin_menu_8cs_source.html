636  private static void SortEntityList(MyEntitySortOrder selectedOrder, ref List<MyEntityListInfoItem> items)
637  {
638  switch (selectedOrder)
639  {
640  case MyEntitySortOrder.DisplayName:
641  items.Sort((a, b) =>
642  {
643  int res = string.Compare(a.DisplayName, b.DisplayName, StringComparison.CurrentCultureIgnoreCase);
644  if (m_invertOrder)
645  return -res;
646  return res;
647  });
648  break;
649  case MyEntitySortOrder.BlockCount:
650  items.Sort((a, b) =>
651  {
652  int res = b.BlockCount.CompareTo(a.BlockCount);
653  if (m_invertOrder)
654  return -res;
655  return res;
656  });
657  break;
658  case MyEntitySortOrder.Mass:
659  items.Sort((a, b) =>
660  {
661  int res;
662  if (a.Mass == 0)
663  res = -1;
664  else if (b.Mass == 0)
665  res = 1;
666  else
667  res = b.Mass.CompareTo(a.Mass);
668  if (m_invertOrder)
669  return -res;
670  return res;
671  });
672  break;
673  case MyEntitySortOrder.OwnerName:
674  items.Sort((a, b) =>
675  {
676  int res = string.Compare(a.OwnerName, b.OwnerName, StringComparison.CurrentCultureIgnoreCase);
677  if (m_invertOrder)
678  return -res;
679  return res;
680  });
681  break;
682  case MyEntitySortOrder.DistanceFromCenter:
683  items.Sort((a, b) =>
684  {
685  int res = a.Position.LengthSquared().CompareTo(b.Position.LengthSquared());
686  if (m_invertOrder)
687  return -res;
688  return res;
689  });
690  break;
691  case MyEntitySortOrder.Speed:
692  items.Sort((a, b) =>
693  {
694  int res = b.Speed.CompareTo(a.Speed);
695  if (m_invertOrder)
696  return -res;
697  return res;
698  });
699  break;
700  case MyEntitySortOrder.DistanceFromPlayers:
701  items.Sort((a, b) =>
702  {
703  int res = b.DistanceFromPlayers.CompareTo(a.DistanceFromPlayers);
704  if (m_invertOrder)
705  return -res;
706  return res;
707  });
708  break;
709  case MyEntitySortOrder.OwnerLastLogin:
710  items.Sort((a, b) =>
711  {
712  int res = b.OwnerLoginTime.CompareTo(a.OwnerLoginTime);
713  if (m_invertOrder)
