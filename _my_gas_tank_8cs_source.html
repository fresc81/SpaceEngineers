200  public void RefillBottles()
201  {
202  var items = this.GetInventory().GetItems();
203  bool changed = false;
204  float newFilledRatio = FilledRatio;
205  foreach (var item in items)
206  {
207  if (FilledRatio == 0f)
208  break;
209 
210  var gasContainer = item.Content as MyObjectBuilder_GasContainerObject;
211  if (gasContainer == null || gasContainer.GasLevel >= 1f)
212  continue;
213 
214  var physicalItem = MyDefinitionManager.Static.GetPhysicalItemDefinition(gasContainer) as MyOxygenContainerDefinition;
215  Debug.Assert(physicalItem != null);
216 
217  float bottleGasAmount = gasContainer.GasLevel * physicalItem.Capacity;
218  float tankGasAmount = FilledRatio * Capacity;
219 
220  float transferredAmount = Math.Min(physicalItem.Capacity - bottleGasAmount, tankGasAmount);
221  gasContainer.GasLevel = Math.Min((bottleGasAmount + transferredAmount) / physicalItem.Capacity, 1f);
222 
223  newFilledRatio = Math.Max(FilledRatio - transferredAmount / Capacity, 0f);
224 
225  this.GetInventory().UpdateGasAmount();
226  changed = true;
227  }
228 
229  if (changed)
230  ChangeFilledRatio(newFilledRatio, true);
231  }
