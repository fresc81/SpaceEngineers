156  // The part towards the cannon is found by projecting the target's

1329  {

1331  if (m_barrel == null || m_enableIdleRotation == false)
1332  return;

1337

1339  float needRotation = m_randomStandbyRotation;
1340  float needElevation = m_randomStandbyElevation;
1341  float step = m_rotationSpeed * m_rotationInterval_ms;
1342  float diffRot = needRotation - m_rotation;
1343 
1344  bool playAimingSound = false;
1345 
1346  if (diffRot > float.Epsilon)
1347  {
1348  float value = MathHelper.Clamp(step, float.Epsilon, needRotation - m_rotation);
1349  m_rotation += value;
1350  playAimingSound = true;
1351  }
1352  else if (diffRot < -float.Epsilon)
1353  {
1354  float value = MathHelper.Clamp(step, float.Epsilon, Math.Abs(needRotation - m_rotation));
1355  m_rotation -= value;
1356  playAimingSound = true;
1357  }
1358

1360  step = m_elevationSpeed * m_elevationInterval_ms;
1361 
1362  if (needElevation > m_barrel.BarrelElevationMin)
1363  {
1364  canElevate = true;
1365  }
1366  else
1367  {
1368  canElevate = false;
1369  }
1370 
1371  if (canElevate)
1372  {
1373  diffRot = needElevation - m_elevation;
1374 
1375  if (diffRot > float.Epsilon)
1376  {
1377  float value = MathHelper.Clamp(step, float.Epsilon, needElevation - m_elevation);
1378  m_elevation += value;
1379  playAimingSound = true;
1380  }
1381  if (diffRot < -float.Epsilon)
1382  {
1383  float value = MathHelper.Clamp(step, float.Epsilon, Math.Abs(needElevation - m_elevation));
1384  m_elevation -= value;
1385  playAimingSound = true;
1386  }
