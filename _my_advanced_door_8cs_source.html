544  private void UpdateDoorPosition()
545  {
546  if (this.CubeGrid.Physics == null)
547  return;
548 
549  for(int i = 0; i < m_subpartCount; i++)
550  {
551  transMat[i] = Matrix.Identity;
552  rotMat[i] = Matrix.Identity;
553  }
554 
555  for (int i = 0; i < m_sequenceCount; i++)
556  {
557  MyObjectBuilder_AdvancedDoorDefinition.Opening.MoveType moveType = m_openingSequence[i].Move;
558  float opening = m_currentOpening[i];
559 
560  // get id of the subpart this opening sequenz should be applied to
561  int id = m_subpartIDs[i];
562 
563  if (m_subparts.Count == 0 || id < 0)
564  break;
565 
566  if (m_subparts[id] != null && m_subparts[id].Physics != null)
567  {
568  if (moveType == MyObjectBuilder_AdvancedDoorDefinition.Opening.MoveType.Slide)
569  {
570  transMat[id] *=
571  Matrix.CreateTranslation(m_openingSequence[i].SlideDirection * new Vector3(opening));
572  }
573  else if (moveType == MyObjectBuilder_AdvancedDoorDefinition.Opening.MoveType.Rotate)
574  {
575  float invert = m_openingSequence[i].InvertRotation ? -1f : 1f;
576 
577  float rotX = 0f;
578  float rotY = 0f;
579  float rotZ = 0f;
580 
581  if (m_openingSequence[i].RotationAxis == MyObjectBuilder_AdvancedDoorDefinition.Opening.Rotation.X)
582  {
583  rotX = MathHelper.ToRadians(opening * invert);
584  }
585  else if (m_openingSequence[i].RotationAxis == MyObjectBuilder_AdvancedDoorDefinition.Opening.Rotation.Y)
586  {
587  rotY = MathHelper.ToRadians(opening * invert);
588  }
589  else if (m_openingSequence[i].RotationAxis == MyObjectBuilder_AdvancedDoorDefinition.Opening.Rotation.Z)
590  {
591  rotZ = MathHelper.ToRadians(opening * invert);
592  }
593 
594  Vector3 hingePos = (m_openingSequence[i].PivotPosition == null) ? m_hingePosition[id] : (Vector3)m_openingSequence[i].PivotPosition;
595 
596  rotMat[id] *=
597  (Matrix.CreateTranslation(-(hingePos)) *
598  (Matrix.CreateRotationX(rotX) *
599  Matrix.CreateRotationY(rotY) *
600  Matrix.CreateRotationZ(rotZ)) *
601  Matrix.CreateTranslation(hingePos));
602  }
603 
604  if (m_subparts[id].Physics.LinearVelocity != this.CubeGrid.Physics.LinearVelocity)
605  {
606  m_subparts[id].Physics.LinearVelocity = this.CubeGrid.Physics.LinearVelocity;
607  }
608 
609  if (m_subparts[id].Physics.AngularVelocity != this.CubeGrid.Physics.AngularVelocity)
610  {
611  m_subparts[id].Physics.AngularVelocity = this.CubeGrid.Physics.AngularVelocity;
612  }
613  }
614  }
615 
616  // combine matrices and apply to subparts
617  for (int i = 0; i < m_subpartCount; i++)
618  {
619  m_subparts[i].PositionComp.LocalMatrix = rotMat[i] * transMat[i];
620  }
621  }
