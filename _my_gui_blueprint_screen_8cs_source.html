923  public static bool CopyBlueprintPrefabToClipboard(MyObjectBuilder_Definitions prefab, MyGridClipboard clipboard, bool setOwner = true)
924  {
925  if (prefab.ShipBlueprints == null)
926  return false;
927 
928  var cubeGrids = prefab.ShipBlueprints[0].CubeGrids;
929 
930  if (cubeGrids == null || cubeGrids.Length == 0)
931  return false;
932 
933  var localBB = MyCubeGridExtensions.CalculateBoundingSphere(cubeGrids[0]);
934 
935  var posAndOrient = cubeGrids[0].PositionAndOrientation.Value;
936  var worldMatrix = MatrixD.CreateWorld(posAndOrient.Position, posAndOrient.Forward, posAndOrient.Up);
937  var invertedNormalizedWorld = Matrix.Normalize(Matrix.Invert(worldMatrix));
938 
939  var worldBB = localBB.Transform(worldMatrix);
940 
941  var dragVector = Vector3.TransformNormal((Vector3)(Vector3D)posAndOrient.Position - worldBB.Center, invertedNormalizedWorld);
942  var dragDistance = localBB.Radius + 10f;
943 
944  //Reset ownership to local player
945  if (setOwner)
946  {
947  foreach (var gridBuilder in cubeGrids)
948  {
949  foreach (var block in gridBuilder.CubeBlocks)
950  {
951  if (block.Owner != 0)
952  {
953  block.Owner = MySession.Static.LocalPlayerId;
954  }
955  }
956  }
957  }
958 
959  // Blueprint can have old (deprecated) fractured blocks, they have to be converted to fracture components. There is no version in blueprints.
960  if (MyFakes.ENABLE_FRACTURE_COMPONENT)
961  {
962  for (int i = 0; i < cubeGrids.Length; ++i)
963  {
964  cubeGrids[i] = MyFracturedBlock.ConvertFracturedBlocksToComponents(cubeGrids[i]);
965  }
966  }

970  return true;
