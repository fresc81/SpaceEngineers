397  {
398  MyGuiControlBase inputHandledBySomeControl = null;
399 
400  if (m_lastHandlingControl != null && m_lastHandlingControl.Visible)
401  {
402  if (m_lastHandlingControl.HandleInput() != null)
403  {
404  inputHandledBySomeControl = m_lastHandlingControl;
405  }
406  }
407 
408  if (inputHandledBySomeControl == null && m_gridDragAndDropHandlingNow != null)
409  {
410  if (m_gridDragAndDropHandlingNow.HandleInput() != null)
411  {
412  inputHandledBySomeControl = m_gridDragAndDropHandlingNow;
413  }
414  }
415 
416  if (inputHandledBySomeControl == null && m_comboboxHandlingNow != null)
417  {
418  if (m_comboboxHandlingNow.HandleInput() != null)
419  {
420  inputHandledBySomeControl = m_comboboxHandlingNow;
421  }
422  }
423 
424  // Check focused screen first
425  MyGuiControlBase mouseOverControl = null;
426  if (inputHandledBySomeControl == null)
427  {
428  var visibleControls = Controls.GetVisibleControls();
429  for (int i = 0; i < visibleControls.Count; i++)
430  {
431  MyGuiControlBase control = visibleControls[i];
432  if (control != m_comboboxHandlingNow && control != m_gridDragAndDropHandlingNow && control.CheckMouseOver())
433  {
434  mouseOverControl = control;
435  inputHandledBySomeControl = control.HandleInput();
436  break;
437  }
438  }
439  }
440 
441  // If opened combobox didn't capture the input, we will try to handle it in remaining controls
442  if (inputHandledBySomeControl == null)
443  {
444  var visibleControls = Controls.GetVisibleControls();
445  for (int i = visibleControls.Count-1; i >= 0; --i)
446  {
447  MyGuiControlBase control = visibleControls[i];
448  if (control != m_comboboxHandlingNow && control != m_gridDragAndDropHandlingNow && control != mouseOverControl)
449  {
450  inputHandledBySomeControl = control.HandleInput();
451  if (inputHandledBySomeControl != null)
452  break;
453  }
454  }
455  }
456 
457  if (inputHandledBySomeControl == null)
458  {
459  foreach (var element in Elements)
460  {
461  if (!element.Visible || !element.CanHaveFocus)
462  continue;
463 
464  inputHandledBySomeControl = element.HandleInput();
465  if (inputHandledBySomeControl != null)
466  break;
467  }
468  }
469 
470  if (inputHandledBySomeControl != null)

497  public MyGuiControlBase GetNextFocusControl(MyGuiControlBase currentFocusControl, bool forwardMovement)
498  {
499  Debug.Assert(currentFocusControl.Owner == this);
500 
501  var visibleControls = Controls.GetVisibleControls();
502 
503  // Pretend that both Controls and Elements are in single array.
504  int totalCount = visibleControls.Count + Elements.Count;
505 
506  int idxCurrent = visibleControls.IndexOf(currentFocusControl);
507  if (idxCurrent == -1)
508  {
509  idxCurrent = Elements.IndexOf(currentFocusControl);
510  Debug.Assert(idxCurrent != -1);
511  idxCurrent += visibleControls.Count;
512  }
513 
514  int sign = (forwardMovement) ? +1 : -1;
515  for (int i = 1; i < totalCount; ++i)
516  {
517  int idx = idxCurrent + (sign * i);
518  if (forwardMovement)
519  idx %= totalCount;
520  else if (idx < 0)
521  idx += totalCount;
522 
523  if (idx < visibleControls.Count)
524  {
525  if (CanHaveFocusRightNow(visibleControls[idx]))
526  {
527  if (visibleControls[idx] is MyGuiControlParent)
528  return visibleControls[idx].GetFocusControl(forwardMovement);
529  else
530  return visibleControls[idx];
531  }
532  }
533  else
534  {
535  idx -= visibleControls.Count;
536  if (CanHaveFocusRightNow(Elements[idx]))
537  return Elements[idx];
538  }
539  }
540 
541  return null;
542  }

834  else

837  }
838  }
839  else if (State == MyGuiScreenState.CLOSING || State == MyGuiScreenState.HIDING)
840  {

842 
843  if (deltaTime >= GetTransitionClosingTime())
844  {
845  m_transitionAlpha = MyGuiConstants.TRANSITION_ALPHA_MIN;
846 
847  // Closing phase finished, we are now in close state
848  if (State == MyGuiScreenState.CLOSING)

856 
857  OnHide();
858  }
859  }
860  else
861  {
862  m_transitionAlpha = MathHelper.Lerp(MyGuiConstants.TRANSITION_ALPHA_MAX, MyGuiConstants.TRANSITION_ALPHA_MIN, MathHelper.Clamp((float)deltaTime / (float)GetTransitionClosingTime(), 0, 1));
863  }
