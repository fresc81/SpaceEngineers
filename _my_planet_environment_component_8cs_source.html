343  private unsafe void RasterSectorsForPhysics(BoundingBoxD range)
344  {
345  range.InflateToMinimum(EnvironmentDefinition.SectorSize);
346 
347  Vector2I top = new Vector2I(1 << m_clipmaps[0].Depth) - 1;
348 
349  Vector3D* pos = stackalloc Vector3D[8];
350 
351  range.GetCornersUnsafe(pos);
352 
353  // bitmask for faces, 7th bit is simple bit
354  int markedFaces = 0;
355  int firstFace = 0;
356 
357  for (var i = 0; i < 8; ++i)
358  {
359  Vector3D copy = pos[i];
360 
361  int index = MyPlanetCubemapHelper.FindCubeFace(ref copy);
362  firstFace = index;
363  index = 1 << index;
364 
365  if ((markedFaces & ~index) != 0) markedFaces |= 0x40;
366 
367  markedFaces |= index;
368  }
369 
370  // This way we can ensure a single code path.
371  int startFace = 0;
372  int endFace = 5;
373 
374  // If we only encounter one face we narrow it down.
375  if ((markedFaces & 0x40) == 0)
376  {
377  startFace = endFace = firstFace;
378  }
379 
380  for (int face = startFace; face <= endFace; ++face)
381  {
382  if (((1 << face) & markedFaces) == 0)
383  continue;
384 
385  double size = m_clipmaps[face].LeafSize;
386 
387  // Offset
388  var offset = 1 << m_clipmaps[face].Depth - 1;
389 
390  BoundingBox2D bounds = BoundingBox2D.CreateInvalid();
391  for (int i = 0; i < 8; ++i)
392  {
393  Vector3D copy = pos[i];
394 
395  Vector2D normCoords;
396  MyPlanetCubemapHelper.ProjectForFace(ref copy, face, out normCoords);
397  bounds.Include(normCoords);
398  }
399 
400  bounds.Min += 1;
401  bounds.Min *= offset;
402 
403  bounds.Max += 1;
404  bounds.Max *= offset;
405 
406  // Calculate bounds in sectors.
407  var start = new Vector2I((int)bounds.Min.X, (int)bounds.Min.Y);
408  var end = new Vector2I((int)bounds.Max.X, (int)bounds.Max.Y);
409 
410  Vector2I.Max(ref start, ref Vector2I.Zero, out start);
411  Vector2I.Min(ref end, ref top, out end);
412 
413  for (int x = start.X; x <= end.X; ++x)
414  for (int y = start.Y; y <= end.Y; ++y)
415  {
416  EnsurePhysicsSector(x, y, face);
417  }
418  }

955  private unsafe void RasterSectorsForCollision(MyEntity entity)
956  {
957  if (!(entity is MyCubeGrid)) return;
958 
959  BoundingBoxD range = entity.PositionComp.WorldAABB;
960  range.Inflate(8);
961  range.Translate(-PlanetTranslation);
962 
963  Vector2I top = new Vector2I(1 << m_clipmaps[0].Depth) - 1;
964 
965  Vector3D* pos = stackalloc Vector3D[8];
966 
967  range.GetCornersUnsafe(pos);
968 
969  // bitmask for faces, 7th bit is simple bit
970  int markedFaces = 0;
971  int firstFace = 0;
972 
973  for (var i = 0; i < 8; ++i)
974  {
975  Vector3D copy = pos[i];
976 
977  int index = MyPlanetCubemapHelper.FindCubeFace(ref copy);
978  firstFace = index;
979  index = 1 << index;
980 
981  if ((markedFaces & ~index) != 0) markedFaces |= 0x40;
982 
983  markedFaces |= index;
984  }
985 
986  // This way we can ensure a single code path.
987  int startFace = 0;
988  int endFace = 5;
989 
990  // If we only encounter one face we narrow it down.
991  if ((markedFaces & 0x40) == 0)
992  {
993  startFace = endFace = firstFace;
994  }
995 
996  for (int face = startFace; face <= endFace; ++face)
997  {
998  if (((1 << face) & markedFaces) == 0)
999  continue;
1000 
1001  // Offset
1002  var offset = 1 << m_clipmaps[face].Depth - 1;
1003 
1004  BoundingBox2D bounds = BoundingBox2D.CreateInvalid();
1005  for (int i = 0; i < 8; ++i)
1006  {
1007  Vector3D copy = pos[i];
1008 
1009  Vector2D normCoords;
1010  MyPlanetCubemapHelper.ProjectForFace(ref copy, face, out normCoords);
1011  bounds.Include(normCoords);
1012  }
1013 
1014  bounds.Min += 1;
1015  bounds.Min *= offset;
1016 
1017  bounds.Max += 1;
1018  bounds.Max *= offset;
1019 
1020  // Calculate bounds in sectors.
1021  var start = new Vector2I((int)bounds.Min.X, (int)bounds.Min.Y);
1022  var end = new Vector2I((int)bounds.Max.X, (int)bounds.Max.Y);
1023 
1024  Vector2I.Max(ref start, ref Vector2I.Zero, out start);
1025  Vector2I.Min(ref end, ref top, out end);
1026 
1027  for (int x = start.X; x <= end.X; ++x)
1028  for (int y = start.Y; y <= end.Y; ++y)
1029  {
1030  long sect = MyPlanetSectorId.MakeSectorId(x, y, face);
1031 
1032  List<MyOrientedBoundingBoxD> boxes;
1033  if (!m_obstructorsPerSector.TryGetValue(sect, out boxes))
1034  {
1035  boxes = new List<MyOrientedBoundingBoxD>();
1036  m_obstructorsPerSector.Add(sect, boxes);
1037  }
1038 
1039  var bb = entity.PositionComp.LocalAABB;
1040  bb.Inflate(8); // inflate by 8m to increase the likellyhood of overlap with trees' roots.
1041 
1042  boxes.Add(new MyOrientedBoundingBoxD((BoundingBoxD)bb, entity.PositionComp.WorldMatrix));
1043  }
1044  }
1045  }
