631  {//active wait for 2ms headstart for next frame + prediction at the end

642  if (m_currFrame - m_previousFrame > 1)

644  m_vrSystem.GetTimeSinceLastVsync(ref m_timeSinceLastVsync, ref m_currFrame);

647 
648  if (m_currFrame - m_previousFrame > 1)
649  MissedFramesCount++;
650  m_previousFrame = m_currFrame;
651  //Debug.WriteLine(m_currFrame);
652  }
