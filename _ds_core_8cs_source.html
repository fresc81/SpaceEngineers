252  public enum SeekingFlags // AM_SEEKING_SeekingFlags AM_SEEKING_SEEKING_FLAGS
253  {
254  NoPositioning = 0x00, // No change
255  AbsolutePositioning = 0x01, // Position is supplied and is absolute
256  RelativePositioning = 0x02, // Position is supplied and is relative
257  IncrementalPositioning = 0x03, // (Stop) position relative to current, useful for seeking when paused (use +1)
258  SeekToKeyFrame = 0x04, // Just seek to key frame (performance gain)
259  ReturnTime = 0x08, // Plug the media time equivalents back into the supplied LONGLONGs

261  NoFlush = 0x20 // Don't flush
