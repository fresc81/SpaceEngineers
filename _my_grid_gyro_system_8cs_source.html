215 
216  // CH: CAUTION: Don't try to use InertiaTensor, although it might be more intuitive in some cases.
217  // I tried it and it's not an inverse of the InverseInertiaTensor! Only the InverseInertiaTensor seems to be correct!

220

222  Vector3 desiredTorque = desiredAcceleration / invTensorVector;
223 
224  // Calculate the available force for the correction by arbitrarily sum overridden gyros

290  Vector3 desiredAcceleration = (m_overrideTargetVelocity - localAngularVelocity) * VRage.Game.MyEngineConstants.UPDATE_STEPS_PER_SECOND;
291 
292  // The correction is done by overridden gyros and by the remaining power of the controlled gyros
293  // This is not entirely physically correct, but it feels good
294  float correctionForce = m_maxOverrideForce + m_maxGyroForce * (1.0f - control.Length());
295 
296  // This is to ensure that the correction is done uniformly in all axes
297  desiredAcceleration = desiredAcceleration * Vector3.Normalize(invTensorVector);
298

301 
302  // If we are very close to the target velocity, just set it without applying the torque
303  const float minimalBypassVelocity = 0.005f * 0.005f;
304  if (framesToDesiredVelocity < 0.5f && m_overrideTargetVelocity.LengthSquared() < minimalBypassVelocity)
305  {
306  return m_overrideTargetVelocity;
307  }
308 
309  if (!Vector3.IsZero(desiredAcceleration, 0.0001f))
310  {
311  // The smoothing coefficient is here to avoid the slowdown stopping the ship abruptly, which doesn't look good
312  float smoothingCoeff = 1.0f - 0.8f / (float)Math.Exp(0.5f * framesToDesiredVelocity);
