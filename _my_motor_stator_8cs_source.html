385  void TryActivateLimits()
386  {
387  if (float.IsNegativeInfinity(m_minAngle) && float.IsPositiveInfinity(m_maxAngle))
388  {
389  m_currentAngle = MoveUp(m_currentAngle, 0, MathHelper.TwoPi);
390  m_currentAngle = MoveDown(m_currentAngle, MathHelper.TwoPi, MathHelper.TwoPi);
391  SetAngleToPhysics();
392  m_limitsActive = false;
393  }
394  else if (!m_limitsActive)
395  {
396  const float graceDegrees = 5;
397  float minimum = m_minAngle - MathHelper.ToRadians(graceDegrees);
398  float maximum = m_maxAngle + MathHelper.ToRadians(graceDegrees);
399  float angle = m_currentAngle;
400 
401  if (angle < minimum)
402  {
403  // Change angle to same or smallest bigger than m_minAngle
404  angle = MoveUp(angle, minimum, MathHelper.TwoPi);
405  }
406  else if (angle > maximum)
407  {
408  // Change angle to same or largest smaller than m_maxAngle
409  angle = MoveDown(angle, maximum, MathHelper.TwoPi);
410  }
411 
412  if (angle >= minimum && angle <= maximum)
413  {
414  m_limitsActive = true;
415  m_currentAngle = angle;
416  SetAngleToPhysics();
417  return;
418  }
419  }
