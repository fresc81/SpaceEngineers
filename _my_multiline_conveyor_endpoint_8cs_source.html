204

206 
207  Vector3I doorPositionInt = Vector3I.Floor(doorPosition / cubeSize);
208  doorPositionInt = Vector3I.Max(Vector3I.Zero, doorPositionInt);
209  doorPositionInt = Vector3I.Min(definition.Size - Vector3I.One, doorPositionInt);
