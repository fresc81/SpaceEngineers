535  return MyRefuseReason.NoOwner;

539  var fbs = grid.GetFatBlocks<MyRemoteControl>();
540  fbs.MoveNext();
541  if (fbs.Current.OwnerId != MySession.Static.LocalHumanPlayer.Identity.IdentityId ||
542  (MySession.Static.Factions.GetPlayerFaction(fbs.Current.OwnerId)!=null &&
543  !MySession.Static.Factions.GetPlayerFaction(fbs.Current.OwnerId).IsMember(MySession.Static.LocalHumanPlayer.Identity.IdentityId)))

547  }

581  var cubeBlock = item.Entity as MyCubeBlock;
