422  bool CheckFactionStateChange(MyFactionStateChange action, long fromFactionId, long toFactionId, long playerId, long senderId)
423  {
424  Debug.Assert(Sync.IsServer, "Only server can check state of message!");
425  if (!Sync.IsServer)
426  return false;
427 
428  if (!m_factions.ContainsKey(fromFactionId) || !m_factions.ContainsKey(toFactionId))
429  return false;
430 
431  var key = new MyFactionPair(fromFactionId, toFactionId);
432  HashSet<long> tmpSet;
433 
434  switch (action)
435  {
436  case MyFactionStateChange.RemoveFaction: return true;
437 
438  case MyFactionStateChange.SendPeaceRequest: return (!m_factionRequests.TryGetValue(fromFactionId, out tmpSet)) || !tmpSet.Contains(toFactionId);
439  case MyFactionStateChange.CancelPeaceRequest: return (m_factionRequests.TryGetValue(fromFactionId, out tmpSet)) && tmpSet.Contains(toFactionId);
440 
441  case MyFactionStateChange.AcceptPeace: return GetRelationBetweenFactions(fromFactionId, toFactionId) != MyRelationsBetweenFactions.Neutral;
442  case MyFactionStateChange.DeclareWar: return GetRelationBetweenFactions(fromFactionId, toFactionId) != MyRelationsBetweenFactions.Enemies;
443 
444  case MyFactionStateChange.FactionMemberSendJoin: return !m_factions[fromFactionId].IsMember(playerId) && !m_factions[fromFactionId].JoinRequests.ContainsKey(playerId);
445  case MyFactionStateChange.FactionMemberCancelJoin: return !m_factions[fromFactionId].IsMember(playerId) && m_factions[fromFactionId].JoinRequests.ContainsKey(playerId);
446  case MyFactionStateChange.FactionMemberAcceptJoin: return m_factions[fromFactionId].JoinRequests.ContainsKey(playerId);
447  case MyFactionStateChange.FactionMemberKick: return m_factions[fromFactionId].IsMember(playerId);
448  case MyFactionStateChange.FactionMemberPromote: return m_factions[fromFactionId].IsMember(playerId);
449  case MyFactionStateChange.FactionMemberDemote: return m_factions[fromFactionId].IsLeader(playerId);
450  case MyFactionStateChange.FactionMemberLeave: return m_factions[fromFactionId].IsMember(playerId);
451  }
452  return false;
453  }

455  void ApplyFactionStateChange(MyFactionStateChange action, long fromFactionId, long toFactionId, long playerId, long senderId)
456  {
457  switch (action)
458  {
459  case MyFactionStateChange.RemoveFaction:
460  if (m_factions[fromFactionId].IsMember(MySession.Static.LocalPlayerId)) // kick all players from faction
461  m_playerFaction.Remove(playerId);
462 
463  foreach (var faction in m_factions)
464  {
465  if (faction.Key != fromFactionId) // remove faction from neutrals and requests
466  {
467  ClearRequest(fromFactionId, faction.Key);
468  RemoveRelation(fromFactionId, faction.Key);
469  }
470  }
471 
472  MyFaction toRemoveFaction = null;
473  var tag = m_factions.TryGetValue(fromFactionId, out toRemoveFaction);
474  UnregisterFactionTag(toRemoveFaction);
475 
476  m_factions.Remove(fromFactionId);
477  break;
478 
479  case MyFactionStateChange.SendPeaceRequest:
480  HashSet<long> tmp;
481 
482  if (m_factionRequests.TryGetValue(fromFactionId, out tmp))
483  tmp.Add(toFactionId);
484  else
485  {
486  tmp = new HashSet<long>();
487  tmp.Add(toFactionId);
488  m_factionRequests.Add(fromFactionId, tmp);
489  }
490  break;
491 
492  case MyFactionStateChange.CancelPeaceRequest:
493  ClearRequest(fromFactionId, toFactionId);
494  break;
495 
496  case MyFactionStateChange.AcceptPeace:
497  ClearRequest(fromFactionId, toFactionId);
498  ChangeRelation(fromFactionId, toFactionId, MyRelationsBetweenFactions.Neutral);
499  break;
500 
501  case MyFactionStateChange.DeclareWar:
502  ClearRequest(fromFactionId, toFactionId);
503  ChangeRelation(fromFactionId, toFactionId, MyRelationsBetweenFactions.Enemies);
504  break;
505 
506  case MyFactionStateChange.FactionMemberSendJoin: m_factions[fromFactionId].AddJoinRequest(playerId); break;
507  case MyFactionStateChange.FactionMemberCancelJoin: m_factions[fromFactionId].CancelJoinRequest(playerId); break;
508  case MyFactionStateChange.FactionMemberAcceptJoin:
509  bool canAccept = false;
510  MyIdentity identity = MySession.Static.Players.TryGetIdentity(senderId);
511  MyPlayer player = identity != null ? MyPlayer.GetPlayerFromCharacter(identity.Character) : null;
512  if (player != null)
513  canAccept = MySession.Static.IsUserSpaceMaster(player.Client.SteamUserId);
514  if (canAccept && m_factions[fromFactionId].IsEveryoneNpc())

516  m_factions[fromFactionId].AcceptJoin(playerId, canAccept);
517  m_factions[fromFactionId].PromoteMember(playerId);
518  }
519  else
520  m_factions[fromFactionId].AcceptJoin(playerId, canAccept);
521  break;
522  case MyFactionStateChange.FactionMemberLeave:
523  case MyFactionStateChange.FactionMemberKick:
524  m_factions[fromFactionId].KickMember(playerId);
525  if (Sync.IsServer && MySession.Static.Settings.ScenarioEditMode && Sync.Players.IdentityIsNpc(playerId))

527  MyIdentity id = Sync.Players.TryGetIdentity(playerId);
528  id.TransferAllBlocksTo(senderId);
529  }

564  var toFaction = MySession.Static.Factions.TryGetFactionById(toFactionId);

568  {
569  if ((action == MyFactionStateChange.FactionMemberKick ||
570  action == MyFactionStateChange.FactionMemberLeave) && fromFaction.Members.Count() == 1)
571  {
572  action = MyFactionStateChange.RemoveFaction;
573  }
574  else if (action == MyFactionStateChange.FactionMemberSendJoin)
575  {
576  bool canAccept = false;
577  var identity = MySession.Static.Players.TryGetIdentity(senderId);
578  MyPlayer humanPlayer = null;
579  if (identity != null)

581  humanPlayer = MyPlayer.GetPlayerFromCharacter(identity.Character);
582  if (humanPlayer != null)
583  canAccept = MySession.Static.IsUserSpaceMaster(humanPlayer.Client.SteamUserId);
584  }
585  if (toFaction.AutoAcceptMember)
586  {
587  canAccept = true;
588  if (!toFaction.AcceptHumans)
589  {
590  // Check, whether the requesting player is human or bot
591  //I think this way the faction can accept dead character
592  if (humanPlayer != null && humanPlayer.Identity.IdentityId == playerId)
593  {
594  // You are a human. We dont like human!
595  canAccept = false;
596  action = MyFactionStateChange.FactionMemberCancelJoin;
597  }
598  }
599  }
600  if (canAccept)
601  {
602  action = MyFactionStateChange.FactionMemberAcceptJoin;
603  //senderId = 0; // no need to check who accepted this // no longer true because of SM ability to add NPC (replacement for ScenarioEditMode)
604  }
605  }
606  else if (action == MyFactionStateChange.SendPeaceRequest && toFaction.AutoAcceptPeace)
607  {
608  action = MyFactionStateChange.AcceptPeace;
609  senderId = 0; // no need to check who accepted this
610  }
611
