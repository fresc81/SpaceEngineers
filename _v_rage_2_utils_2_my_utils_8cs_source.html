542  public static Vector3 GetClosestPointOnLine(ref Vector3 linePointA, ref Vector3 linePointB, ref Vector3D point, out float dist)
543  {
544  // Create the vector from end point vA to our point vPoint.
545  Vector3 vector1 = point - linePointA;
546 
547  // Create a normalized direction vector from end point vA to end point vB
548  Vector3 vector2 = MyUtils.Normalize(linePointB - linePointA);
549 
550  // Use the distance formula to find the distance of the line segment (or magnitude)
551  float d = Vector3.Distance(linePointA, linePointB);
552 
553  // Using the dot product, we project the vVector1 onto the vector vVector2.
554  // This essentially gives us the distance from our projected vector from vA.
555  float t = Vector3.Dot(vector2, vector1);
556 
557  dist = t;
558 
559  // If our projected distance from vA, "t", is less than or equal to 0, it must
560  // be closest to the end point vA. We want to return this end point.
561  if (t <= 0) return linePointA;
562 
563  // If our projected distance from vA, "t", is greater than or equal to the magnitude
564  // or distance of the line segment, it must be closest to the end point vB. So, return vB.
