675  private void SwitchWeapon(ref IMyHandheldGunObject<MyDeviceBase> weapon, ref MyVRWeaponInfo weaponInfo, List<MyVRWeaponInfo> weapons, ControllerRole role, Vector2? touchpadPos)
676  {
677  int currentIndex = -1;
678 
679  if (weapon != null)
680  {
681  for (int i = 0; i < weapons.Count; ++i)
682  {
683  var info = weapons[i];
684  if (info.DefinitionId == weapon.DefinitionId)
685  {
686  currentIndex = i;
687  break;
688  }
689  }
690  }
691 
692  int nextIndex = 0;
693  if (touchpadPos != null)
694  {
695  Vector2 pos = touchpadPos.Value;
696  if (touchpadPos != Vector2.Zero)
697  {
698  pos.Normalize();
699  float anglePerWeaponSector = 360f / weapons.Count;
700  float dot = Vector2.Dot(Vector2.UnitX, pos);
701  float angle = (float)Math.Acos(Math.Abs(dot));
702  if (pos.Y >= 0)
703  {
704  if (dot < 0)
705  angle = 180 - angle;
706  }
707  else
708  {
709  if (dot < 0)
710  angle = 180 + angle;
711  else
712  angle = 360 - angle;
713  }
714 
715  nextIndex = (int)Math.Floor(angle / anglePerWeaponSector);
716  Debug.Assert(nextIndex < weapons.Count);
717  }
718  }
719 
720  if (nextIndex == currentIndex)
721  return;
722 
723  if (weapon != null)
724  {
725  weapon.OnControlReleased();
726  MyEntity weaponEntity = (MyEntity)weapon;
727  weaponEntity.Close();
728  weapon = null;
729  }
730 
731  {
732  weaponInfo = weapons[nextIndex];
733  weapon = CreateWeapon(weaponInfo.DefinitionId, weaponInfo.Reference);
734  weapon.OnControlAcquired(null);
735  MyEntity weaponEntity = (MyEntity)weapon;
736  }
737 
738  var gunBase = weapon.GunBase as MyGunBase;
739  Matrix holdingMatrix = gunBase != null ? gunBase.m_holdingDummyMatrix : Matrix.Identity;
740  MyOpenVR.LMUAdd(holdingMatrix, m_worldMatrixOriginal, role, weaponInfo.Reference);
741  }
