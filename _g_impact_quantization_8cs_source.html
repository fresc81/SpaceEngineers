54  public static void CalcQuantizationParameters(
55  out IndexedVector3 outMinBound,
56  out IndexedVector3 outMaxBound,
57  out IndexedVector3 bvhQuantization,
58  ref IndexedVector3 srcMinBound, ref IndexedVector3 srcMaxBound,
59  float quantizationMargin)
60  {
61  //enlarge the AABB to avoid division by zero when initializing the quantization values
62  IndexedVector3 clampValue = new IndexedVector3(quantizationMargin);
63  outMinBound = srcMinBound - clampValue;
64  outMaxBound = srcMaxBound + clampValue;
65  IndexedVector3 aabbSize = outMaxBound - outMinBound;
66  bvhQuantization = new IndexedVector3(65535.0f) / aabbSize;
67  }

69  public static void QuantizeClamp(
70  out UShortVector3 output,
71  ref IndexedVector3 point,
72  ref IndexedVector3 min_bound,
73  ref IndexedVector3 max_bound,
74  ref IndexedVector3 bvhQuantization)
75  {
76 
77  IndexedVector3 clampedPoint = point;
78  MathUtil.VectorMax(ref min_bound, ref clampedPoint);
79  MathUtil.VectorMin(ref max_bound, ref clampedPoint);
80 
81  IndexedVector3 v = (clampedPoint - min_bound) * bvhQuantization;
82  output = new UShortVector3();
83  output[0] = (ushort)(v.X + 0.5f);
84  output[1] = (ushort)(v.Y + 0.5f);
85  output[2] = (ushort)(v.Z + 0.5f);
86  }
