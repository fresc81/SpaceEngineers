249  void SmoothTransition(MyLodStrategyInfo strategyInfo, float timeDeltaSeconds, float distance, ref int currentLod, ref int targetLod, ref float transition)
250  {
251  if (transition == 0) // the transition has not started
252  {
253  //Check the suitable lod
254  targetLod = GetTheBestLodWithHisteresis(strategyInfo, distance, currentLod);
255 
256  //Lod is fine, therefore no transition
257  if (currentLod == targetLod)
258  {
259  targetLod = -1;
260  return;
261  }
262 
263  //Lod is not found, init transition:
264  m_transitionStartedAtDistance = distance;
265 
266  //And swap target and current lod
267  int swappedLod = targetLod;
268  targetLod = currentLod;
269  currentLod = swappedLod;
270  }
271 
272  // The transition should start
273  float deltaTransitionTime = timeDeltaSeconds/MAX_TRANSITION_TIME_IN_SECS;
274  float deltaTransitionDistance = Math.Abs(m_transitionStartedAtDistance-distance) / m_lodTransitionDistances[Math.Min(currentLod, targetLod)];
275  float deltaTransition = Math.Max(deltaTransitionDistance, deltaTransitionTime);
276  deltaTransition = Math.Min(deltaTransition, MAX_TRANSITION_PER_FRAME);
277  transition = transition + deltaTransition;
278  if (transition >= 1.0f)
279  {
280  transition = 0.0f;
281  targetLod = -1;
282  }
