105  int lines = 1;
106  for (int i = 0; i < text.Length; i++)
107  {
108  char c = text[i];
109 
110  // New line
111  if (c == NEW_LINE)
112  {
113  lines++;

117  }

123

125  if (KernEnabled)
126  {

128  cLast = c;
129  }
130

132  pxWidth += ginfo.pxAdvanceWidth;
133 
134  // Spacing

136

139  }
140 
141  return new Vector2(maxPxWidth * scale, lines * LineHeight * scale);
