244  private static void ExtractAction(ActionCollection actions, MethodInfo methodInfo)
245  {
246  var btActionAttribute = methodInfo.GetCustomAttribute<MyBehaviorTreeActionAttribute>();
247  if (btActionAttribute == null)
248  return;
249  switch (btActionAttribute.ActionType)
250  {
251  case MyBehaviorTreeActionType.INIT:
252  actions.AddInitAction(btActionAttribute.ActionName, (x) => methodInfo.Invoke(x.BotActions, null));
253  break;
254  case MyBehaviorTreeActionType.BODY:
255  actions.AddAction(btActionAttribute.ActionName, methodInfo, btActionAttribute.ReturnsRunning, (x, y) => (MyBehaviorTreeState)methodInfo.Invoke(x.BotActions, y));
256  break;
257  case MyBehaviorTreeActionType.POST:
258  actions.AddPostAction(btActionAttribute.ActionName, (x) => methodInfo.Invoke(x.BotActions, null));
