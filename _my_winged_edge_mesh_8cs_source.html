1335  public void RemoveFace(int faceIndex)
1336  {

1338

1349  //
1350  // edge v1
1351  // o-------------o
1352  // | /
1353  // | removed /
1354  // ... face / nextEdge
1355  // /
1356  // /
1357  // ...--o v2
1358  //
1359  //
1360  // According to neighboring faces of the current and next edge, we have four possible scenarios:
1361  // A) Both current and next edge have empty face as neighbor:
1362  // In this case, the shared vertex should be removed, because it means that the vertex has only two faces
1363  // B) Current edge has empty face, next edge non-empty:
1364  // In this case, we are not creating a new empty face, we are just extending the old one.
1365  // C) Current edge has non-empty face, next edge empty:
1366  // Analogous to B - we are just extending an existing empty face.
1367  // D) Both edges have non-empty neighbors:
1368  // This seemingly easy situation has one quirk - there could be another empty face along the shared vertex triangle fan
1369  // In this case, we have to split the vertex into two (otherwise, we'd have a vertex with two empty faces in its fan
1370  // and that is forbidden).
1371  do
1372  {
1373  int nextEdge = edgeEntry.FaceSucc(faceIndex);
1374  v1 = v2;
1375  // If the next edge is the first edge, which could have been deallocated, we use the saved first entry
1376  EdgeTableEntry nextEntry = nextEdge == firstEdge && firstDeallocated ? firstEntry : GetEdgeEntry(nextEdge);
1377  v2 = nextEntry.OtherVertex(v2);

1380  {
1381  if (edge == firstEdge)
