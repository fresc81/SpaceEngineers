296  public static void InitializeConstantBuffer(int? randomSeed = null)
297  {
298  MyRandom random;
299  if (randomSeed.HasValue)
300  random = new MyRandom(randomSeed.Value);
301  else
302  random = new MyRandom();
303 
304  const int JITTERSIZE = 4 * 4;
305  var jitters = new Vector4[JITTERSIZE];
306  for (int i = 0; i < JITTERSIZE; i++)
307  {
308  float angle = 2.0f * (float)Math.PI * random.NextFloat() / NUM_DIRECTIONS;
309  jitters[i].X = (float)Math.Cos(angle);
310  jitters[i].Y = (float)Math.Sin(angle);
311  jitters[i].Z = random.NextFloat();
312  jitters[i].W = random.NextFloat();
313  }
314  PerPassConstantBuffer data;
315  for (uint sliceIndex = 0; sliceIndex < NUM_SLICES; ++sliceIndex)
316  {
317  data.Offset.X = (float)(sliceIndex % 4) + 0.5f;
318  data.Offset.Y = (float)(sliceIndex / 4) + 0.5f;
319  data.Jitter = jitters[sliceIndex];
320  data.SliceIndexFloat = (float)sliceIndex;
321  data.SliceIndexInt = sliceIndex;
322 
323  var buffer = m_perPassCBs[sliceIndex];
324  if (buffer == null)
325  {
326  buffer = MyManagers.Buffers.CreateConstantBuffer("MyHBAO::passCB " + sliceIndex, PERPASSCONSTANTBUFFERSIZE, usage: ResourceUsage.Dynamic);
327  m_perPassCBs[sliceIndex] = buffer;
328  }
329 
330  var mapping = MyMapping.MapDiscard(buffer);
331  mapping.WriteAndPosition(ref data);
332  mapping.Unmap();
333  }

376  static float METERS_TO_VIEW_SPACE_UNITS = 1.0f;
377  static GlobalConstantBuffer InitConstantBuffer(MyViewport viewport)
378  {
379  GlobalConstantBuffer m_Data = new GlobalConstantBuffer();
380  Matrix m = MyRender11.Environment.Matrices.Projection;
381 
382  // ProjectionMatrixInfo
383  // In matrices generated with D3DXMatrixPerspectiveFovRH
384  // A = zf/(zn-zf)
385  // B = zn*zf/(zn-zf)
386  // C = -1
387  float A = m.M33;
388  float B = m.M43;
389 
390  // Rely on INFs to be generated in case of any divisions by zero
391  float zNear = B / A;
392  float zFar = B / (A + 1);
393  // Some matrices may use negative m00 or m11 terms to flip X/Y axises
394  float tanHalfFovX = 1 / Math.Abs(m.M11);
395  float tanHalfFovY = 1 / Math.Abs(m.M22);
396 
397  // SetDepthLinearizationConstants
398  const float EPSILON = 1e-6f;
399  float inverseZNear = Math.Max(1 / zNear, EPSILON);
400  float inverseZFar = Math.Max(1 / zFar, EPSILON);
401  m_Data.LinearizeDepthA = inverseZFar - inverseZNear;
402  m_Data.LinearizeDepthB = inverseZNear;
403 
404  // SetViewportConstants
405  m_Data.InverseDepthRangeA = 1f;
406  m_Data.InverseDepthRangeB = 0f;
407 
408  m_Data.InputViewportTopLeft.X = viewport.OffsetX;
409  m_Data.InputViewportTopLeft.Y = viewport.OffsetY;
410 
411  // SetProjectionConstants
412  m_Data.UVToViewA.X = 2 * tanHalfFovX;
413  m_Data.UVToViewA.Y = -2 * tanHalfFovY;
414  m_Data.UVToViewB.X = -1 * tanHalfFovX;
415  m_Data.UVToViewB.Y = 1 * tanHalfFovY;
416 
417  // SetResolutionConstants
418  m_Data.InvFullResolution.X = 1.0f / viewport.Width;
419  m_Data.InvFullResolution.Y = 1.0f / viewport.Height;
420  m_Data.InvQuarterResolution.X = 1.0f / DivUp((int)viewport.Width, 4);
421  m_Data.InvQuarterResolution.Y = 1.0f / DivUp((int)viewport.Height, 4);
422 
423  // SetNormalData
424  m_Data.NormalMatrix = MyRender11.Environment.Matrices.ViewAt0;
425  m_Data.NormalDecodeScale = 2;
426  m_Data.NormalDecodeBias = -1;
427 
428  // SetAORadiusConstants
429  float radiusInMeters = Math.Max(Params.Radius, EPSILON);
430  float r = radiusInMeters * METERS_TO_VIEW_SPACE_UNITS;
431  m_Data.R2 = r * r;
432  m_Data.NegInvR2 = -1 / m_Data.R2;
433 
434  m_Data.RadiusToScreen = r * 0.5f / tanHalfFovY * viewport.Height;
435 
436  float backgroundViewDepth = Math.Max(Params.BackgroundViewDepth, EPSILON);
437  if (Params.AdaptToFOV)
438  {
439  // use larger background view depth for low FOV values (less then 30 degrees)
440  float factor = Math.Min(1.0f, MyRender11.Environment.Matrices.FovH / MathHelper.ToRadians(30));
441  backgroundViewDepth = MathHelper.Lerp(6000, backgroundViewDepth, factor);
442  }
443  m_Data.BackgroundAORadiusPixels = m_Data.RadiusToScreen / backgroundViewDepth;
444 
445  float foregroundViewDepth = Math.Max(Params.ForegroundViewDepth, EPSILON);
446  m_Data.ForegroundAORadiusPixels = m_Data.RadiusToScreen / foregroundViewDepth;
447 
448  // SetBlurConstants
449  float BaseSharpness = Math.Max(Params.BlurSharpness, 0);
450  BaseSharpness /= METERS_TO_VIEW_SPACE_UNITS;
451 
452  if (Params.BlurSharpnessFunctionEnable)
453  {
454  m_Data.BlurViewDepth0 = Math.Max(Params.BlurSharpnessFunctionForegroundViewDepth, 0);
455  m_Data.BlurViewDepth1 = Math.Max(Params.BlurSharpnessFunctionBackgroundViewDepth, m_Data.BlurViewDepth0 + EPSILON);
456  m_Data.BlurSharpness0 = BaseSharpness * Math.Max(Params.BlurSharpnessFunctionForegroundScale, 0);
457  m_Data.BlurSharpness1 = BaseSharpness;
458  }
459  else
460  {
461  m_Data.BlurSharpness0 = BaseSharpness;
462  m_Data.BlurSharpness1 = BaseSharpness;
463  m_Data.BlurViewDepth0 = 0;
464  m_Data.BlurViewDepth1 = 1;
465  }
466 
467  // SetDepthThresholdConstants
468  if (Params.DepthThresholdEnable)
469  {
470  m_Data.ViewDepthThresholdNegInv = -1 / Math.Max(Params.DepthThreshold, EPSILON);
471  m_Data.ViewDepthThresholdSharpness = Math.Max(Params.DepthThresholdSharpness, 0);
472  }
473  else
474  {
475  m_Data.ViewDepthThresholdNegInv = 0;
476  m_Data.ViewDepthThresholdSharpness = 1;
477  }
478 
479  // SetAOParameters
480  m_Data.PowExponent = Math.Min(Math.Max(Params.PowerExponent, 1), 8);
481  m_Data.NDotVBias = Math.Min(Math.Max(Params.Bias, 0.0f), 0.5f);
482 
483  float aoAmountScaleFactor = 1 / (1 - m_Data.NDotVBias);
484  m_Data.SmallScaleAOAmount = Math.Min(Math.Max(Params.SmallScaleAO, 0), 4) * aoAmountScaleFactor * 2;
485  m_Data.LargeScaleAOAmount = Math.Min(Math.Max(Params.LargeScaleAO, 0), 4) * aoAmountScaleFactor;
486 
487  return m_Data;
488  }
