225  MyStatsUpdater.Timestamps.Update(ref currentTimestamp);
226  long timestampForUpdate = m_prevTimestampsUpdate + 1000 * 1000 / updatesPerSecond;

233  string group = "Timings";

235  MyStatsDisplay.WritePersistent(group, "CPU per frame (us)", (int)frame, page);
