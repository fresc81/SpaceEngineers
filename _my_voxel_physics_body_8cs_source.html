702  {
703  // mk:NOTE This method must be thread safe. Called from worker threads.
704 
705  if (mesh == null || mesh.TrianglesCount == 0 || mesh.VerticesCount == 0)
706  return HkShape.Empty;

709  s_indexList.Capacity = mesh.TrianglesCount * 3;

723  for (int i = 0; i < mesh.VerticesCount; i++)
724  {
725  s_vertexList.Add(positions[i] * mesh.PositionScale + positionOffset);
726  }
