316  public override MyGuiControlBase HandleInput()
317  {
318  MyGuiControlBase captureInput = base.HandleInput();
319 
320  if (captureInput == null)
321  {
322  if (!m_activateOnMouseRelease)
323  {
324  if (IsMouseOver && MyInput.Static.IsNewPrimaryButtonPressed())
325  m_readyToClick = true;
326  if (!IsMouseOver && MyInput.Static.IsNewPrimaryButtonReleased())
327  m_readyToClick = false;
328  }
329  else
330  m_readyToClick = true;
331 
332  if ((IsMouseOver && (MyInput.Static.IsNewPrimaryButtonReleased()) && m_readyToClick ||
333  (HasFocus && (MyInput.Static.IsNewKeyPressed(MyKeys.Enter) ||
334  MyInput.Static.IsNewKeyPressed(MyKeys.Space)))))
335  {
336  if (Enabled)
337  {
338  MyGuiSoundManager.PlaySound(m_cueEnum);
339  if (ButtonClicked != null)
340  ButtonClicked(this);
341  }
342 
343  captureInput = this;
344  m_readyToClick = false;
345  return captureInput;
346  }
347  if (IsMouseOver && MyInput.Static.IsPrimaryButtonPressed())
348  captureInput = this;//to be first in queue when button is released
349  }
350  return captureInput;
