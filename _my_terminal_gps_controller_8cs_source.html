397  {//takes current right side values of name, description and coordinates, compares them against record with previous hash and synces if necessary
398  if (m_previousHash != null && (m_needsSyncName || m_needsSyncDesc || m_needsSyncX || m_needsSyncY || m_needsSyncZ))
399  {
400  if (MySession.Static.Gpss.ExistsForPlayer(MySession.Static.LocalPlayerId))
401  {
402  if (IsNameOk(m_panelInsName.Text) && IsCoordOk(m_xCoord.Text) && IsCoordOk(m_yCoord.Text) && IsCoordOk(m_zCoord.Text))
403  {
404  Dictionary<int, MyGps> insList;
405  insList = MySession.Static.Gpss[MySession.Static.LocalPlayerId];
406  MyGps ins;
407  if (insList.TryGetValue((int)m_previousHash, out ins))
408  {
409  if (m_needsSyncName)
410  ins.Name = m_panelInsName.Text;
411  if (m_needsSyncDesc)
412  ins.Description = m_panelInsDesc.Text;
413  StringBuilder str = new StringBuilder();
414  if (m_needsSyncX)
415  {
416  m_xCoord.GetText(str);
417  ins.Coords.X = Math.Round(double.Parse(str.ToString(), System.Globalization.CultureInfo.InvariantCulture),2);
418  }
419  str.Clear();
420  if (m_needsSyncY)
421  {
422  m_yCoord.GetText(str);
423  ins.Coords.Y = Math.Round(double.Parse(str.ToString(), System.Globalization.CultureInfo.InvariantCulture),2);
424  }
425  str.Clear();
426  if (m_needsSyncZ)
427  {
428  m_zCoord.GetText(str);
429  ins.Coords.Z = Math.Round(double.Parse(str.ToString(), System.Globalization.CultureInfo.InvariantCulture), 2);
430  }
431  m_syncedGps = ins;
432  MySession.Static.Gpss.SendModifyGps(MySession.Static.LocalPlayerId, ins);
