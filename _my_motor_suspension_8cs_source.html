782  private void Accelerate(float force, bool forward)

787  return;

793  var wheelRadius = TopBlock.BlockDefinition.Size.X * TopGrid.GridSizeHalf;
794  var surfaceVelSq = TopGrid.Physics.AngularVelocity.LengthSquared();
795  var speedLimitSq = (SpeedLimit * (1 / 3.6f)) / wheelRadius;
796  speedLimitSq *= speedLimitSq;

821  }

823  lin = TopGrid.Physics.AngularVelocity.Length() * wheelRadius; // linear velocity at tire surface
