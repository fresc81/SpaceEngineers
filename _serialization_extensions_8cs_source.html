64  public static Quaternion ReadQuaternionNorm(this BitStream stream)
65  {
66  Quaternion value;
67  bool cwNeg = stream.ReadBool();
68  bool cxNeg = stream.ReadBool();
69  bool cyNeg = stream.ReadBool();
70  bool czNeg = stream.ReadBool();
71  ushort cx = stream.ReadUInt16();
72  ushort cy = stream.ReadUInt16();
73  ushort cz = stream.ReadUInt16();
74 
75  // Calculate w from x,y,z
76  value.X = (float)(cx / 65535.0);
77  value.Y = (float)(cy / 65535.0);
78  value.Z = (float)(cz / 65535.0);
79  if (cxNeg) value.X = -value.X;
80  if (cyNeg) value.Y = -value.Y;
81  if (czNeg) value.Z = -value.Z;
82  float difference = 1.0f - value.X * value.X - value.Y * value.Y - value.Z * value.Z;
83  if (difference < 0.0f)
84  difference = 0.0f;
85  value.W = (float)(Math.Sqrt(difference));
86  if (cwNeg)
87  value.W = -value.W;
88  return value;
89  }
