766  private MyCueId CheckDistanceSounds(MyCueId soundId)
767  {
768  if (soundId.IsNull == false)
769  {
770  if (m_lastSoundData != null && m_lastSoundData.DistantSounds != null && m_lastSoundData.DistantSounds.Count > 0)
771  {
772  float distanceToSoundSquered = Vector3.DistanceSquared(MySector.MainCamera.Position, this.SourcePosition);
773  int bestSoundIndex = -1;
774  m_usesDistanceSounds = true;
775  m_secondaryEnabled = false;
776  float dist, crossfadeDist;
777  for (int i = 0; i < m_lastSoundData.DistantSounds.Count; i++)
778  {
779  dist = m_lastSoundData.DistantSounds[i].distance * m_lastSoundData.DistantSounds[i].distance;
780  if (distanceToSoundSquered > dist)
781  bestSoundIndex = i;
782  else
783  {
784  crossfadeDist = m_lastSoundData.DistantSounds[i].distanceCrossfade >= 0f ? m_lastSoundData.DistantSounds[i].distanceCrossfade * m_lastSoundData.DistantSounds[i].distanceCrossfade : float.MaxValue;
785  if (distanceToSoundSquered > crossfadeDist)
786  {
787  m_secondaryVolumeRatio = (distanceToSoundSquered - crossfadeDist) / (dist - crossfadeDist);
788  m_secondaryEnabled = true;
789  MySoundPair secondarySoundPair = new MySoundPair(m_lastSoundData.DistantSounds[i].sound);
790  if (secondarySoundPair != MySoundPair.Empty)
791  {
792  m_secondaryCueEnum = SelectCue(secondarySoundPair);
793  }
794  else
795  m_secondaryCueEnum = new MyCueId(MyStringHash.GetOrCompute(m_lastSoundData.DistantSounds[bestSoundIndex].sound));
796  }
797  else
798  break;
799  }
800  }
801  if (bestSoundIndex >= 0)
802  {
803  MySoundPair soundPair = new MySoundPair(m_lastSoundData.DistantSounds[bestSoundIndex].sound);
804  if (soundPair != MySoundPair.Empty)
805  {
806  m_soundPair = soundPair;
807  soundId = SelectCue(m_soundPair);
808  } else
809  soundId = new MyCueId(MyStringHash.GetOrCompute(m_lastSoundData.DistantSounds[bestSoundIndex].sound));
810  }
811  else
812  {
813  m_soundPair = m_closeSoundSoundPair;
814  }
815  }
816  else
817  {
818  m_usesDistanceSounds = false;
819  }
820  }
821  if (m_secondaryEnabled == false)
822  m_secondaryCueEnum = myEmptyCueId;
823  return soundId;
824  }

830  private void PlaySoundInternal(bool skipIntro = false, bool skipToEnd = false, bool force2D = false, bool alwaysHearOnRealistic = false)
831  {
832  Force2D = force2D;
833  m_alwaysHearOnRealistic = alwaysHearOnRealistic;

835  Loop = MyAudio.Static.IsLoopable(SoundId) && !skipToEnd && CanPlayLoopSounds;
836  if (!SoundId.IsNull)

839  skipIntro = true;
840  if (m_playing2D && CheckForSynchronizedSounds())
841  Sound = MyAudio.Static.PlaySound(SoundId, this, MySoundDimensions.D2, skipIntro, skipToEnd);
842  else if (CanHearSound() && CheckForSynchronizedSounds()) //Start 3D sound only if can be heard
843  Sound = MyAudio.Static.PlaySound(SoundId, this, MySoundDimensions.D3, skipIntro, skipToEnd);
