491  private bool ProcessTriangleForHierarchy(MyNavigationTriangle triangle)
492  {
493  // The triangle parent can be wrong when we have multiple navmeshes connected via external edges
494  if (triangle.Parent != m_mesh)
495  {
496  return false;
497  }
498 
499  // Previously unvisited triangle will be assigned the current relative component index
500  if (triangle.ComponentIndex == -1)
501  {
502  m_navmeshComponents.AddComponentTriangle(triangle, triangle.Center);
503  m_tmpComponentTriangles.Add(triangle);
504  triangle.ComponentIndex = m_currentComponentMarker;
505  return true;
506  }
507  else if (triangle.ComponentIndex == m_currentComponentMarker)
508  {
509  // We can safely ignore this triangle (it has already been processed);
510  return true;
511  }
512  else
513  {
514  ulong cellIndex;
515  if (m_navmeshComponents.GetComponentCell(triangle.ComponentIndex, out cellIndex))
516  {
517  MyCellCoord cellCoord = new MyCellCoord();
518  cellCoord.SetUnpack(cellIndex);
519  Vector3I diff = cellCoord.CoordInLod - m_currentCell;
520  if (diff.RectangularLength() != 1)
521  {
522  // CH: TODO: Connection of components over cell edges or vertices. I currently silently ignore that...
523  return false;
524  }
525 
526  ConnectionInfo connection = new ConnectionInfo();
527  connection.Direction = Base6Directions.GetDirection(diff);
528  connection.ComponentIndex = triangle.ComponentIndex;
529

531  if (!m_currentCellConnections[m_currentComponentRel].Contains(connection))
