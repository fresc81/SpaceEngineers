108  HkShape CreateShape(MyModel model, HkShapeType shapeType)
109  {

124

132  case HkShapeType.Sphere:

336  private void GeneratePositions(BoundingBoxD boundingBox, List<Vector3D> positionBuffer)
337  {
338  positionBuffer.Clear();
339 
340  Vector3D minMax = boundingBox.Max - boundingBox.Min;
341  var product = minMax.X * minMax.Y * minMax.Z;
342 
343  var a3 = MyDebrisConstants.APPROX_NUMBER_OF_DEBRIS_OBJECTS_PER_MODEL_EXPLOSION / product;

346 
347  Vector3D minMaxScaled = minMax * a;

351  int maxZ = (int)Math.Ceiling(minMaxScaled.Z);
352 
353  Vector3D offset = new Vector3D(minMax.X / maxX, minMax.Y / maxY, minMax.Z / maxZ);
354 
355  Vector3D origin = boundingBox.Min + 0.5 * offset;

357  for (int x = 0; x < maxX; x++)
