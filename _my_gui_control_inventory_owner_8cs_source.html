385  public static void FormatItemAmount(Type typeId, double amount, StringBuilder text)
386  {
387  try
388  {
389  if (typeId == typeof(MyObjectBuilder_Ore) ||
390  typeId == typeof(MyObjectBuilder_Ingot))
391  {
392 
393  if (amount < 0.01)
394  text.Append(amount.ToString("<0.01", CultureInfo.InvariantCulture));
395  else if (amount < 10)
396  text.Append(amount.ToString("0.##", CultureInfo.InvariantCulture));
397  else if (amount < 100)
398  text.Append(amount.ToString("0.#", CultureInfo.InvariantCulture));
399  else if (amount < 1000)
400  text.Append(amount.ToString("0.", CultureInfo.InvariantCulture));
401  else if (amount < 10000)
402  text.Append((amount / 1000.0).ToString("0.##k", CultureInfo.InvariantCulture));
403  else if (amount < 100000)
404  text.Append((amount / 1000.0).ToString("0.#k", CultureInfo.InvariantCulture));
405  else
406  text.Append((amount / 1000.0).ToString("#,##0.k", CultureInfo.InvariantCulture));
407  }
408  else if (typeId == typeof(MyObjectBuilder_PhysicalGunObject))
409  {
410  Debug.Assert(amount == 1, "There should only be one gun in a single slot. This is safe to ignore.");
411  }
412  else if (typeId == typeof(MyObjectBuilder_GasContainerObject) || typeId.BaseType == typeof(MyObjectBuilder_GasContainerObject))
413  {
414  int integerPart = (int)amount;
415  text.Append(integerPart.ToString()+"%");
416  }
417  else
418  {
419  int integerPart = (int)amount;
420  var decimalPart = amount - integerPart;
421  if (decimalPart > 0)
422  text.Append('~'); // used for half empty magazines and such
423  text.Append(integerPart.ToString("#,##0.x", CultureInfo.InvariantCulture));
424  }
425  }
426  catch (System.OverflowException)
427  {
428  text.Append("ERROR");
429  }
430  }
