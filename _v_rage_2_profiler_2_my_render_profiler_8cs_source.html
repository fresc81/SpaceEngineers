255  public static List<MyProfilerBlock> GetSortedChildren(int frameToSortBy)
256  {
257  List<MyProfilerBlock> sortedChildren = new List<MyProfilerBlock>(m_selectedProfiler.SelectedRootChildren);
258 
259  switch (m_sortingOrder)
260  {
261  case RenderProfilerSortingOrder.Id:
262  sortedChildren.Sort(delegate(MyProfilerBlock a, MyProfilerBlock b)
263  {
264  return a.Id.CompareTo(b.Id);
265  });
266  break;
267  case RenderProfilerSortingOrder.MillisecondsLastFrame:
268  sortedChildren.Sort(delegate(MyProfilerBlock a, MyProfilerBlock b)
269  {
270  // Sorts by milliseconds, in case of equal performance, sorts by ID instead
271  int comparisonResult = b.Miliseconds[frameToSortBy].CompareTo(a.Miliseconds[frameToSortBy]);
272  if (comparisonResult != 0) return comparisonResult;
273  return a.Id.CompareTo(b.Id);
274  });
275  break;
276  case RenderProfilerSortingOrder.MillisecondsAverage:
277  sortedChildren.Sort(delegate(MyProfilerBlock a, MyProfilerBlock b)
278  {
279  // Sorts by average milliseconds, in case of equal performance, sorts by ID instead
280  int comparisonResult = b.averageMiliseconds.CompareTo(a.averageMiliseconds);
281  if (comparisonResult != 0) return comparisonResult;
282  return a.Id.CompareTo(b.Id);
283  });
284  break;
285  }
286 
287  return sortedChildren;
288  }

654  static void FindMax(float[] data, int lower, int upper, int lastValidFrame, ref float max, ref int maxIndex)
655  {
656  Debug.Assert(lower <= upper, "Lower must be smaller or equal to Upper");
657  int windowEnd = (lastValidFrame + 1 + MyProfiler.UPDATE_WINDOW) % MyProfiler.MAX_FRAMES;
658 
659  if (lastValidFrame > windowEnd)
660  {
661  FindMax(data, Math.Max(lower, windowEnd), Math.Min(lastValidFrame, upper), ref max, ref maxIndex);
662  }
663  else
664  {
665  FindMax(data, lower, Math.Min(lastValidFrame, upper), ref max, ref maxIndex);

667  }
