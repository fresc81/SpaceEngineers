554  protected virtual void InsertQueueItem(int idx, MyBlueprintDefinitionBase blueprint, MyFixedPoint amount)
555  {
556  Debug.Assert(idx <= m_queue.Count);
557  Debug.Assert(CanUseBlueprint(blueprint));
558  Debug.Assert(amount > 0);
559 
560  QueueItem item = new QueueItem();
561  item.Amount = amount;
562  item.Blueprint = blueprint;
563 
564  if (CanUseBlueprint(item.Blueprint))
565  {
566  if (m_queue.IsValidIndex(idx) && m_queue[idx].Blueprint == item.Blueprint)
567  {
568  // Increase amount if there is same kind of item at this index.
569  item.Amount += m_queue[idx].Amount;
570  item.ItemId = m_queue[idx].ItemId;
571  m_queue[idx] = item;
572  }
573  else if (m_queue.Count > 0 && (idx >= m_queue.Count || idx == -1) && m_queue[m_queue.Count - 1].Blueprint == item.Blueprint)
574  {
575  // Add to the last item in the queue if it is the same.
576  item.Amount += m_queue[m_queue.Count - 1].Amount;
577  item.ItemId = m_queue[m_queue.Count - 1].ItemId;
578  m_queue[m_queue.Count - 1] = item;
579  }
580  else
581  {
582  if (idx == -1)
583  idx = m_queue.Count;
584 
585  if (idx > m_queue.Count)
586  {
587  MyLog.Default.WriteLine("Production block.InsertQueueItem: Index out of bounds, desync!");
588  idx = m_queue.Count;
589  }
590 
591  // Reset timer when adding first item. Otherwise we might produce it faster than possible.
592  if (m_queue.Count == 0)
593  m_lastUpdateTime = MySandboxGame.TotalGamePlayTimeInMilliseconds;
594 
595  // Put new item into the queue with given amount and type.
596  item.ItemId = NextItemId;
597  m_queue.Insert(idx, item);
598  }

600  UpdatePower();
601 
602  OnQueueChanged();
603  }
