506  private int GetProceduralAsteroidSeed(MyGuiControlTextbox textbox)
507  {
508  int seed = 12345;

512  String text = textbox.Text;
513 
514  HashAlgorithm algorithm = SHA1.Create();
515  byte[] bytes = algorithm.ComputeHash(Encoding.UTF8.GetBytes(text));
516  //hash is too big to store it in int, so we take only first four bytes - it's more than fine, though
517  int shift = 0;
518  for (int i = 0; i < 4 && i < bytes.Length; ++i)
519  {
520  seed |= bytes[i] << shift;

522  }
523  }
