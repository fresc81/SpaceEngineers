91  protected override bool Activate(HashSet<MySlimBlock> targets)
92  {

101  float coefficient = (MyShipGrinderConstants.GRINDER_COOLDOWN_IN_MILISECONDS * 0.001f) / targets.Count;
102  foreach (var block in targets)
103  {
104  if ((MySession.Static.IsScenario || MySession.Static.Settings.ScenarioEditMode) && !block.CubeGrid.BlocksDestructionEnabled)
105  continue;
106 
107  m_otherGrid = block.CubeGrid;
108 
109  bool tmp2 = m_otherGrid.Physics == null || !m_otherGrid.Physics.Enabled;
110  if (tmp2)
111  {
112  successTargets--;
113  continue;
114  }
115 
116  float damage = MySession.Static.GrinderSpeedMultiplier * MyShipGrinderConstants.GRINDER_AMOUNT_PER_SECOND * coefficient;
117  MyDamageInformation damageInfo = new MyDamageInformation(false, damage, MyDamageType.Grind, EntityId);
118 
119  if (block.UseDamageSystem)
120  MyDamageSystem.Static.RaiseBeforeDamageApplied(block, ref damageInfo);
121 
122  if (block.CubeGrid.Editable)
123  {
124  block.DecreaseMountLevel(damageInfo.Amount, this.GetInventory());
125  block.MoveItemsFromConstructionStockpile(this.GetInventory());
126  }
127 
128  if (block.UseDamageSystem)
129  MyDamageSystem.Static.RaiseAfterDamageApplied(block, damageInfo);
130 
131  if (block.IsFullyDismounted)
132  {
133  if (block.FatBlock != null && block.FatBlock.HasInventory)
134  {
135  EmptyBlockInventories(block.FatBlock);
136  }
137 
138  if(block.UseDamageSystem)
139  MyDamageSystem.Static.RaiseDestroyed(block, damageInfo);

141  block.SpawnConstructionStockpile();
142  block.CubeGrid.RazeBlock(block.Min);
143  }
144  }
145  if (successTargets > 0)
146  SetBuildingMusic(200);
147  }
148  m_wantsToShake = successTargets != 0;

312  private Matrix GetEffectMatrix(int effectNum)
313  {
314  Matrix retval = WorldMatrix;
315  retval.Translation += retval.Forward * 1.5f;
316  retval.Translation += retval.Up * (effectNum == 1 ? 0.25f : -0.25f);
317 
318  float zRotation = effectNum == 1 ? MyUtils.GetRandomFloat(0.3f, 0.7f) : MyUtils.GetRandomFloat(-0.7f, -0.3f);
319  float yRotation = (float)Math.PI * -MyUtils.GetRandomFloat(0.47f, 0.53f);
320 
321  retval = Matrix.CreateRotationZ(zRotation) * Matrix.CreateRotationY(yRotation) * retval;
