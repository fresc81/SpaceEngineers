822  private Matrix CalculateParticleRotation(MatrixD mat)
823  {
824  Vector3 angle = Angle;
825  Vector3 angleVar = AngleVar;
826  if (angleVar.LengthSquared() > 0)
827  {
828  angle = new Vector3(
829  MyUtils.GetRandomFloat(angle.X - angleVar.X, angle.X + angleVar.X),
830  MyUtils.GetRandomFloat(angle.Y - angleVar.Y, angle.Y + angleVar.Y),
831  MyUtils.GetRandomFloat(angle.Z - angleVar.Z, angle.Z + angleVar.Z));
832  }
833  if (angle.LengthSquared() > 0)
834  {
835  return Matrix.CreateFromAxisAngle(mat.Right, MathHelper.ToRadians(angle.X)) *
836  Matrix.CreateFromAxisAngle(mat.Up, MathHelper.ToRadians(angle.Y)) *
837  Matrix.CreateFromAxisAngle(mat.Forward, MathHelper.ToRadians(angle.Z));
838  }
839  else return mat;
840  }
