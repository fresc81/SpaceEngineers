428  ResolveTypes(methodGenerator, createdTypes);
429 
430  Dictionary<long, Label> labels = new Dictionary<long, Label>();
431  foreach (VRage.Compiler.IlReader.IlInstruction instr in instructions)
432  {
433  labels[instr.Offset] = methodGenerator.DefineLabel();
434  }
435  methodGenerator.Emit(OpCodes.Call, methodToInjectMethodCheck);
436  methodGenerator.Emit(OpCodes.Call, methodToInject);
437  // get the operation code of the current instruction
438  foreach (var instruction in instructions)
439  {
440  methodGenerator.MarkLabel(labels[instruction.Offset]);
441  var code = instruction.OpCode;
442 
443  if (code == OpCodes.Switch)
444  {
445  methodGenerator.Emit(OpCodes.Call, methodToInject);
446  var op = (instruction.Operand as int[]).Select(off => labels[off]).ToArray();
447  methodGenerator.Emit(OpCodes.Switch, op);
448  }
449  else if (code.FlowControl == FlowControl.Branch || code.FlowControl == FlowControl.Cond_Branch)

470  break;

481  }
482  else
483  {
484  methodGenerator.Emit(code, type);
485  }
486  }
487  catch

489 
490  }
491  break;

493  case OperandType.InlineSig:
494  case OperandType.InlineI:
