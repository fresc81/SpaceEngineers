610  private void CreateBot(int playerNumber, MyObjectBuilder_Bot botBuilder)
611  {
612  Debug.Assert(BotFactory != null, "Bot factory is not set! Cannot create a new bot!");
613  if (BotFactory == null) return;
614 
615  var newPlayer = Sync.Clients.LocalClient.GetPlayer(playerNumber);
616  if (newPlayer == null) return;
617 
618  var isBotSpawned = m_agentsToSpawn.ContainsKey(playerNumber);
619  var isLoading = botBuilder != null;
620  var createdByPlayer = false;
621  MyBotDefinition botDefinition = null;
622  AgentSpawnData spawnData = default(AgentSpawnData);
623 
624  // We have to get the bot object builder and bot definition somehow
625  // Either, the bot is being spawned on this computer and the definition was saved in the spawn data
626  // or the bot is just being created from the object builder (MP bot creation, etc.), so the definition is there
627  if (isBotSpawned)
628  {
629  spawnData = m_agentsToSpawn[playerNumber];
630  createdByPlayer = spawnData.CreatedByPlayer;
631  botDefinition = spawnData.AgentDefinition;
632  m_agentsToSpawn.Remove(playerNumber);
633  }
634  else
635  {
636  if (botBuilder == null || botBuilder.BotDefId.TypeId.IsNull)
637  {
638  MyPlayer missingBotPlayer = null;
639  if (Sync.Players.TryGetPlayerById(new MyPlayer.PlayerId(Sync.MyId, playerNumber), out missingBotPlayer))
640  {
641  Sync.Players.RemovePlayer(missingBotPlayer);
642  }
643  return;
644  }
645 
646  MyDefinitionManager.Static.TryGetBotDefinition(botBuilder.BotDefId, out botDefinition);
647  Debug.Assert(botDefinition != null, "Bot definition could not be found.");
648  if (botDefinition == null)
649  return;
650  }
651 
652  if ((newPlayer.Character == null || !newPlayer.Character.IsDead)
653  && BotFactory.CanCreateBotOfType(botDefinition.BehaviorType, isLoading)
654  || createdByPlayer)
655  {
656  IMyBot bot = null;
657  if (isBotSpawned)
658  bot = BotFactory.CreateBot(newPlayer, botBuilder, spawnData.AgentDefinition);
659  else
660  bot = BotFactory.CreateBot(newPlayer, botBuilder, botDefinition);
661 
662  if (bot == null)
663  {
664  MyLog.Default.WriteLine("Could not create a bot for player " + newPlayer + "!");
665  }
666  else
667  {
668  m_botCollection.AddBot(playerNumber, bot);
669  if (isBotSpawned && bot is IMyEntityBot)
670  (bot as IMyEntityBot).Spawn(spawnData.SpawnPosition, createdByPlayer);
671 
672  if (BotCreatedEvent != null)
673  {
674  BotCreatedEvent(playerNumber, bot.BotDefinition);
675  }
676  }
677  }
678  else
679  {
680  // hack for removing uncontrolled bot players or saved dead characters
681  var player = Sync.Players.GetPlayerById(new MyPlayer.PlayerId(Sync.MyId, playerNumber));
682  Sync.Players.RemovePlayer(player);
683  }
684  }
