424  void CheckReplanningOfPath(Vector3D targetPos, Vector3D navigationTarget)
425  {
426  // replans a path if target of navigation is too far from target position
427  // current navigation target differs from actual target position
428  Vector3D directionToTarget = targetPos - Bot.Navigation.PositionAndOrientation.Translation;
429  Vector3D directionToNavigationTarget = navigationTarget - Bot.Navigation.PositionAndOrientation.Translation;
430  double toTargetDist = directionToTarget.Length();
431  double toNavigationTargetDist = directionToNavigationTarget.Length();
432  if (toTargetDist != 0 && toNavigationTargetDist != 0)
433  {
434  double cosFi = directionToTarget.Dot(directionToNavigationTarget) / (toTargetDist * toNavigationTargetDist);
435  double angle = Math.Acos(cosFi);
436  double actualToPlanned = toTargetDist / toNavigationTargetDist;
437  if (angle > Math.PI / 5 // angle to target is too different
438  || actualToPlanned < 0.8 // this could be ommited probably
439  || (actualToPlanned>1 && toNavigationTargetDist<2) ) // bot is approaching navigation target already
440  {
441  // change of navigation target
442  AiTargetBase.GotoTarget();
443  AiTargetBase.AimAtTarget();
444  }
445  }
