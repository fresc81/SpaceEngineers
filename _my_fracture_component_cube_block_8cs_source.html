357  public static HkdBreakableShape AddMountForShape(HkdBreakableShape shape, Matrix transform, ref BoundingBox blockBB, float gridSize,
358  List<MyCubeBlockDefinition.MountPoint> outMountPoints)
359  {
360  Vector4 min;
361  Vector4 max;
362  shape.GetShape().GetLocalAABB(0.01f, out min, out max);//.Transform(CubeGrid.PositionComp.WorldMatrix);
363  var bb = new BoundingBox(new Vector3(min), new Vector3(max));
364  bb = bb.Transform(transform);
365  bb.Min /= gridSize; //normalize for mount point
366  bb.Max /= gridSize;
367 
368  bb.Inflate(0.04f);//add tolerance (fracture shapes are smaller than block)
369  bb.Min += blockBB.HalfExtents;
370  bb.Max += blockBB.HalfExtents;
371 
372  if (blockBB.Contains(bb) == ContainmentType.Intersects)
373  {
374  bb.Inflate(-0.04f);
375  foreach (var directionEnum in Base6Directions.EnumDirections)
376  {
377  int dirEnum = (int)directionEnum;
378  Vector3 direction = Base6Directions.Directions[dirEnum];
379  Vector3 absDir = Vector3.Abs(direction);
380  var mp = new MyCubeBlockDefinition.MountPoint();
381  mp.Start = bb.Min;
382  mp.End = bb.Max;
383  mp.Enabled = true;
384  var start = mp.Start * absDir / (blockBB.HalfExtents * 2) - absDir * 0.04f;
385  var end = mp.End * absDir / (blockBB.HalfExtents * 2) + absDir * 0.04f;
386  bool add = false;
387  bool one = false;
388  if (start.Max() < 1 && end.Max() > 1 && direction.Max() > 0)
389  {
390  add = true;
391  one = true;
392  }
393  else if (start.Min() < 0 && end.Max() > 0 && direction.Min() < 0)
394  {
395  add = true;
396  }
397  if (!add)
398  {
399  continue;
400  }
401  mp.Start -= mp.Start * absDir - absDir * 0.04f;
402  mp.End -= mp.End * absDir + absDir * 0.04f;
403  if (one)
404  {
405  mp.Start += absDir * blockBB.HalfExtents * 2;
406  mp.End += absDir * blockBB.HalfExtents * 2;
407  }
408  mp.Start -= blockBB.HalfExtents - Vector3.One / 2;
409  mp.End -= blockBB.HalfExtents - Vector3.One / 2;
410  mp.Normal = new Vector3I(direction);
411  outMountPoints.Add(mp);
412  }
413  }
414  return shape;
