852  out MyMeshPartInfo1[] parts, out MyMeshSectionInfo1[] sections,
853  out MyLODDescriptor[] lodDescriptors, float rescale)
854  {
855  parts = null;
856  sections = null;
857  lodDescriptors = null;
858 
859  var importer = new MyModelImporter();
860  var file = lodMeshInfo.FileName;
861  if (file == null)
862  {
863  return true;
864  }
865  var fsPath = Path.IsPathRooted(file) ? file : Path.Combine(MyFileSystem.ContentPath, file);
866 
867  if (!MyFileSystem.FileExists(fsPath))
868  {
869  MyRender11.Log.WriteLine(String.Format("Mesh asset {0} missing", file));
870  return false;
871  }
872

875  meshData.DoImport(importer, fsPath);
876

878  lodDescriptors = (MyLODDescriptor[])meshData.Lods.Clone();

881  {
882  var positions = meshData.Positions;
883 
884  if (rescale != 1.0f)
885  for (int i = 0; i < positions.Length; i++)
886  positions[i] = VF_Packer.PackPosition(VF_Packer.UnpackPosition(positions[i]) * rescale);
887 
888  var verticesNum = positions.Length;
889 
890  lodMeshInfo.VerticesNum = verticesNum;
891 
892  Debug.Assert(positions.Length > 0);
893  if (positions.Length == 0)

896  importer.Clear();

898  }
899  }
900 
901  // Alter tangents
902  {
903  var normals = meshData.Normals;
904  var tangents = meshData.Tangents;
905  var bitangents = meshData.Bitangents;
906 
907  var storedTangents = new Byte4[meshData.VerticesNum];
908  meshData.StoredTangents = storedTangents;
909 
910  if (tangents.Length > 0 && bitangents.Length > 0)
911  {
912  // calculate tangents used by run-time
913  for (int i = 0; i < meshData.VerticesNum; i++)
914  {
915  var N = VF_Packer.UnpackNormal(normals[i].PackedValue);
916  var T = VF_Packer.UnpackNormal(tangents[i].PackedValue);
917  var B = VF_Packer.UnpackNormal(bitangents[i].PackedValue);
918 
919  var tanW = new Vector4(T.X, T.Y, T.Z, 0);
920  tanW.W = T.Cross(N).Dot(B) < 0 ? -1 : 1;
921 
922  storedTangents[i] = VF_Packer.PackTangentSignB4(ref tanW);
923  }
924  }
925  }
926

928  {
929  var patternScale = meshData.PatternScale;
930  var texcoords = meshData.Texcoords;
931 
932  if (patternScale != 1f && texcoords.Length > 0)
933  for (int i = 0; i < texcoords.Length; ++i)
934  texcoords[i] = new HalfVector2(texcoords[i].ToVector2() / patternScale);
935  }
936 
937  // Mesh part info gathering and merging of parts
938  CreatePartInfos(file, GetContentPath(file), meshData, ref lodMeshInfo, out parts, out sections);
939 
940  // Fill raw mesh data
941  MyMeshRawData rawData;
942  FillMeshData(meshData, ref lodMeshInfo, out rawData);
943 
944  // Set output mesh info
945  lodMeshInfo.Data = rawData;
946  lodMeshInfo.BoundingBox = meshData.BoundindBox;
947  lodMeshInfo.BoundingSphere = meshData.BoundingSphere;
948 
949  var diagonal = meshData.BoundindBox.Size.Length();
950  lodMeshInfo.TriangleDensity = lodMeshInfo.TrianglesNum / (diagonal * diagonal);
951 
952  importer.Clear();
953  return true;
954  }
