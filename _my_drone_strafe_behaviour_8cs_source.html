504  //find new target
505  private bool FindNewTarget()
506  {
507  //create new target list with players
508  List<DroneTarget> possibleTargets = new List<DroneTarget>();
509  if (PlayerPriority > 0)
510  {
511  foreach (var player in MySession.Static.Players.GetOnlinePlayers())
512  {
513  var controlled = player.Controller.ControlledEntity;
514  if (controlled == null)
515  continue;
516  if (controlled is MyCharacter && ((MyCharacter)controlled).IsDead)
517  continue;
518 
519  Vector3D position = controlled.Entity.WorldMatrix.Translation;
520  double distSq = Vector3D.DistanceSquared(m_remoteControl.PositionComp.GetPosition(), position);
521  if (distSq < m_maxPlayerDistanceSq)
522  possibleTargets.Add(new DroneTarget((MyEntity)controlled, PlayerPriority));
523  }
524  }
525  for (int i = 0; i < m_targetsList.Count;i++ )
526  {
527  if (IsValidTarget(m_targetsList[i].Target))
528  possibleTargets.Add(m_targetsList[i]);
529  }
530  m_targetsFiltered.Clear();
531  m_targetsFiltered = possibleTargets;
532  if (possibleTargets.Count == 0)
533  return false;
534 
535  bool completeRandom = m_prioritizationStyle == TargetPrioritization.Random;
536  switch (m_prioritizationStyle)
537  {
538  default:
539  case TargetPrioritization.HightestPriorityFirst:
540  possibleTargets.Sort();
541  m_currentTarget = possibleTargets[0].Target;
542  return true;
543 
544  case TargetPrioritization.ClosestFirst:
545  double bestDist = double.MaxValue;
546  foreach (DroneTarget target in possibleTargets)
547  {
548  double distSq = Vector3D.DistanceSquared(m_remoteControl.PositionComp.GetPosition(), target.Target.PositionComp.GetPosition());
549  if (distSq < bestDist)
550  {
551  bestDist = distSq;
552  m_currentTarget = target.Target;
553  }
554  }
555  return true;
556 
557  case TargetPrioritization.Random:
558  case TargetPrioritization.PriorityRandom:
559  int sum = 0;
560  foreach (var pair in possibleTargets)
561  {
562  sum += completeRandom ? 1 : Math.Max(0, pair.Priority);
563  }
564  int randomPick = MyUtils.GetRandomInt(0, sum+1);
565  foreach (var pair in possibleTargets)
566  {
567  int p = completeRandom ? 1 : Math.Max(0, pair.Priority);
568  if (randomPick <= p)
569  {
570  m_currentTarget = pair.Target;
571  break;
572  }
573  else
574  {
575  randomPick -= p;
576  }
577  }
578  return true;
579  }
580 
581  return false;
582  }
