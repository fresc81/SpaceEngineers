721  {

723  {
724  case WireType.Fixed32:
725  if(available < 4) Ensure(4, true);
726  available -= 4;
727  ioIndex += 4;
728  position += 4;
729  return;
730  case WireType.Fixed64:
731  if (available < 8) Ensure(8, true);
732  available -= 8;
733  ioIndex += 8;
734  position += 8;
735  return;
736  case WireType.String:
737  int len = (int)ReadUInt32Variant(false);
738  if (len <= available)
739  { // just jump it!
740  available -= len;
741  ioIndex += len;
742  position += len;
743  return;
744  }
745  // everything remaining in the buffer is garbage
746  position += len; // assumes success, but if it fails we're screwed anyway
747  len -= available; // discount anything we've got to-hand
748  ioIndex = available = 0; // note that we have no data in the buffer
749  if (isFixedLength)
750  {
751  if (len > dataRemaining) throw EoF(this);
752  // else assume we're going to be OK
753  dataRemaining -= len;
754  }
755  ProtoReader.Seek(source, len, ioBuffer);
756  return;
757  case WireType.Variant:
758  case WireType.SignedVariant:
759  ReadUInt64Variant(); // and drop it
760  return;
761  case WireType.StartGroup:
762  int originalFieldNumber = this.fieldNumber;
763  depth++; // need to satisfy the sanity-checks in ReadFieldHeader
764  while (ReadFieldHeader() > 0) { SkipField(); }
765  depth--;
766  if (wireType == WireType.EndGroup && fieldNumber == originalFieldNumber)
767  { // we expect to exit in a similar state to how we entered
768  wireType = ProtoBuf.WireType.None;
769  return;
770  }

772  case WireType.None: // treat as explicit errorr
773  case WireType.EndGroup: // treat as explicit error

861  public static byte[] AppendBytes(byte[] value, ProtoReader reader)
862  {
863  switch (reader.wireType)
864  {
865  case WireType.String:
866  int len = (int)reader.ReadUInt32Variant(false);
867  reader.wireType = WireType.None;
868  if (len == 0) return value == null ? EmptyBlob : value;
869  int offset;
870  if (value == null || value.Length == 0)
871  {
872  offset = 0;
873  value = new byte[len];
874  }
875  else
876  {
877  offset = value.Length;
878  byte[] tmp = new byte[value.Length + len];
879  Helpers.BlockCopy(value, 0, tmp, 0, value.Length);
880  value = tmp;
881  }
882  // value is now sized with the final length, and (if necessary)
883  // contains the old data up to "offset"
884  reader.position += len; // assume success
885  while (len > reader.available)
886  {
887  if (reader.available > 0)
888  {
889  // copy what we *do* have
890  Helpers.BlockCopy(reader.ioBuffer, reader.ioIndex, value, offset, reader.available);
891  len -= reader.available;
892  offset += reader.available;
893  reader.ioIndex = reader.available = 0; // we've drained the buffer
894  }
895  // now refill the buffer (without overflowing it)
896  int count = len > reader.ioBuffer.Length ? reader.ioBuffer.Length : len;
897  if (count > 0) reader.Ensure(count, true);
898  }
899  // at this point, we know that len <= available
900  if (len > 0)
901  { // still need data, but we have enough buffered
902  Helpers.BlockCopy(reader.ioBuffer, reader.ioIndex, value, offset, len);
903  reader.ioIndex += len;
904  reader.available -= len;
905  }
906  return value;
907  default:
908  throw reader.CreateWireTypeException();
909  }
