372  private bool CallScriptInternal(string message)
373  {
374 #if XB1 // XB1_ALLINONEASSEMBLY

376  return false;
377 #else // !XB1
378  Assembly ass;
379  if (IlCompiler.Buffer.Length > 0)
380  {
381  if (IlCompiler.Compile(new string[] { IlCompiler.Buffer.ToString() }, out ass,true))

384  if (!string.IsNullOrEmpty(message))

386  return true;
387  IlCompiler.Buffer.Clear();

389  else

409  return false;
410  }

412  var type = ass.GetType(parts[2]);
413  if (type == null)
414  {
415  string availableScripts = "";
416  var types = ass.GetTypes();
417  foreach (var scriptType in types)
418  {
419  availableScripts += scriptType.FullName + "\n";
420  }
421  MyAPIGateway.Utilities.ShowMissionScreen("Class not found", "", "Available classes:", availableScripts);
422  return false;
423  }

425  if (method == null)
426  {
427  string availableScripts = "";

429  foreach (var scriptType in types)
430  {
431  availableScripts += scriptType.Name + "\n";
432  }
433  MyAPIGateway.Utilities.ShowMissionScreen("Method not found", "", "Available methods:", availableScripts);
434  return false;
435  }
436  var paramInfos = method.GetParameters();
437  List<object> parameters = new List<object>();
438  for (int i = 4; i < paramInfos.Length + 4 && i < parts.Length; i++)
439  {
440  var paramType = paramInfos[i - 4].ParameterType;
441  var parseMet = paramType.GetMethod("TryParse", new Type[] { typeof(System.String), paramType.MakeByRefType() });

445  var args = new object[] { parts[i], output };

447  parameters.Add(args[1]);
448  }
449  else

451  }
