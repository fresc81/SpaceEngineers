352  private void RefreshPresetCombo(MyRenderSettings1 renderSettings)
353  {
354  int presetIdx;
355  for (presetIdx = 0; presetIdx < m_presets.Length; ++presetIdx)
356  {
357  var preset = m_presets[presetIdx];
358  if (preset.AnisotropicFiltering == renderSettings.AnisotropicFiltering &&
359  preset.AntialiasingMode == renderSettings.AntialiasingMode &&
360  preset.FoliageDetails == renderSettings.FoliageDetails &&
361  preset.ShadowQuality == renderSettings.ShadowQuality &&
362  preset.TextureQuality == renderSettings.TextureQuality)
363  break;
364  }
365  m_comboGraphicsPresets.SelectItemByKey(presetIdx, false);
366  }
