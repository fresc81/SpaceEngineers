409  public bool CleanCachedChunks()
410  {
411  int count = m_chunksbyAge.Count;
412  int discarded = 0;
413 
414  // This procedure is thread safe enough because even if a chunk
415  // is discarded when in use the data is still valid.
416  for (int i = 0; i < count && discarded < MaxChunksToDiscard; ++i)
417  {
418  var coord = m_chunksbyAge.Dequeue();
419 
420  var ck = m_cachedChunks[coord];
421 
422  // this allows us to skip locking on most chunks
423  if (ck.Dirty == 0 && ck.HitCount <= MaximumHitsForDiscard)
424  using (ck.Lock.AcquireSharedUsing())
425  {
426  if (ck.Dirty == 0 && ck.HitCount <= MaximumHitsForDiscard)
427  {
428  using (m_cacheLock.AcquireExclusiveUsing())
429  {
430  m_cachedChunks.Remove(coord);
431  m_cacheMap.RemoveProxy(ck.TreeProxy);
432  }
433  }
434  else
435  {
436  // not removed, requeue, refresh age
437  m_chunksbyAge.Enqueue(coord);
438  ck.HitCount = 0; // reset hit count for next time.
439  }
440  }
441  else
442  {
443  m_chunksbyAge.Enqueue(coord);
444  ck.HitCount = 0; // reset hit count for next time.
445  }
446  }
447 
448  return m_cachedChunks.Count == 0;
