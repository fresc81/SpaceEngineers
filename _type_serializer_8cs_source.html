552  }

575  ctx.LoadValue(group.First);
576  Compiler.CodeLabel processThisField = ctx.DefineLabel();

580  }
581  else

583  ctx.LoadValue(fieldNumber);
584  ctx.LoadValue(group.First);

586  Compiler.CodeLabel[] jmp = new Compiler.CodeLabel[groupItemCount];
587  for (int i = 0; i < groupItemCount; i++) {
588  jmp[i] = ctx.DefineLabel();
589  }
590  ctx.Switch(jmp);
591  // write the default...
592  ctx.Branch(tryNextField, false);
593  for (int i = 0; i < groupItemCount; i++)
594  {
595  WriteFieldHandler(ctx, expected, loc, jmp[i], @continue, (IProtoSerializer)group.Items[i]);
596  }
597  }
598  ctx.MarkLabel(tryNextField);
599  }
600 
601  EmitCreateIfNull(ctx, loc);
602  ctx.LoadReaderWriter();
603  if (isExtensible)
604  {
605  ctx.LoadValue(loc);
606  ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("AppendExtensionData"));
607  }
608  else
609  {
610  ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("SkipField"));
611  }
612

614  ctx.EmitBasicRead("ReadFieldHeader", ctx.MapType(typeof(int)));
615  ctx.CopyValue();
616  ctx.StoreValue(fieldNumber);
617  ctx.LoadValue(0);
618  ctx.BranchIfGreater(processField, false);

622  EmitCallbackIfNeeded(ctx, loc, TypeModel.CallbackType.AfterDeserialize);
