326  private Vector3I CubeToCell(ref Vector3I cube)
327  {
328  // We have 20 bits per coord for the cell coordinate. If we want signed coords, we have to convert to unsigned by
329  // adding 2^19. The range will then be <0, 2^20-1> after shift and <-2^19, 2^19-1> before shift. Anything else will
330  // be reported as overflow by MyCellCoord
331  Vector3D cubeD = cube;
332  cubeD = cubeD / m_cellSize;
333  Vector3I retval;
334 
335  Vector3I.Floor(ref cubeD, out retval);
336 
337  retval += CELL_COORD_SHIFT;
338 
339  return retval;
340  }
