562  public bool CheckAngleLimits(Vector3D directionNormalized)
563  {
564  if (m_angleLimitCosine == 0)
565  m_angleLimitCosine = Math.Cos(MathHelper.ToRadians(BlockDefinition.RaycastConeLimit));
566 
567  var projTargetFront = VectorProjection(directionNormalized, this.WorldMatrix.Forward);

569  var projTargetUp = VectorProjection(directionNormalized, this.WorldMatrix.Up);
570  var projTargetFrontLeft = projTargetFront + projTargetLeft;
571  var projTargetFrontUp = projTargetFront + projTargetUp;

574  var pitchCheck = projTargetFrontUp.Dot(this.WorldMatrix.Forward);
575 
576  return !(yawCheck < m_angleLimitCosine) && !(pitchCheck < m_angleLimitCosine);

627  public MyDetectedEntityInfo Raycast(double distance, Vector3D direction)
628  {
629  if (Vector3D.IsZero(direction))
630  throw new ArgumentOutOfRangeException("direction", "Direction cannot be 0,0,0");
631 
632  //mods can disable raycast on a block by setting the distance limit to 0 (-1 means infinite)
633  if (distance <= 0 || (BlockDefinition.RaycastDistanceLimit > -1 && distance > BlockDefinition.RaycastDistanceLimit))
634  return new MyDetectedEntityInfo();
635 
636  if (AvailableScanRange < distance || !this.CheckIsWorking())
637  return new MyDetectedEntityInfo();
638 
639  AvailableScanRange -= distance;
640 
641  var startPos = this.WorldMatrix.Translation;
642  var targetPos = startPos + direction * distance;
643 
644  //try a physics raycast first
645  //very accurate, but very slow
646  List<MyPhysics.HitInfo> hits = new List<MyPhysics.HitInfo>();
647  MyPhysics.CastRay(startPos, targetPos, hits);
648 
649  foreach (var hit in hits)
650  {
651  var entity = (MyEntity)hit.HkHitInfo.GetHitEntity();
652  if (entity == this)
653  continue;
654 
655  m_lastRay = new RaycastInfo() { Distance = distance, Start = startPos, End = targetPos, Hit = hit.Position };
656  return MyDetectedEntityInfoHelper.Create(entity, this.OwnerId, hit.Position);
657  }
658 
659  //long-distance planet scanning
660  //fastest way is to intersect planet bounding boxes then treat the planet as a sphere
661  LineD line = new LineD(startPos, targetPos);
662  var voxels = new List<MyLineSegmentOverlapResult<MyVoxelBase>>();
663  MyGamePruningStructure.GetVoxelMapsOverlappingRay(ref line, voxels);
664 
665  foreach (var result in voxels)
666  {
667  var planet = result.Element as MyPlanet;
668  if (planet == null)
669  continue;
670 
671  double distCenter = Vector3D.DistanceSquared(this.PositionComp.GetPosition(), planet.PositionComp.GetPosition());
672  var gravComp = planet.Components.Get<MyGravityProviderComponent>();
673  if (gravComp == null)
674  continue;
675 
676  if (!gravComp.IsPositionInRange(startPos) && distCenter > planet.MaximumRadius * planet.MaximumRadius)
677  {
678  var boundingSphere = new BoundingSphereD(planet.PositionComp.GetPosition(), planet.MaximumRadius);
679  var rayd = new RayD(startPos, direction);
680  var intersection = boundingSphere.Intersects(rayd);
681 
682  if (!intersection.HasValue)
683  continue;
684 
685  if (distance < intersection.Value)
686  continue;
687 
688  var hitPos = startPos + direction * intersection.Value;
689  m_lastRay = new RaycastInfo() {Distance = distance, Start = startPos, End = targetPos, Hit = hitPos};
690  return MyDetectedEntityInfoHelper.Create(result.Element, this.OwnerId, hitPos);
691  }
692 
693  //if the camera is inside gravity, query voxel storage
694  if (planet.RootVoxel.Storage == null)
695  continue;
696  var start = Vector3D.Transform(line.From, planet.PositionComp.WorldMatrixInvScaled);
697  start += planet.SizeInMetresHalf;
698  var end = Vector3D.Transform(line.To, planet.PositionComp.WorldMatrixInvScaled);
699  end += planet.SizeInMetresHalf;
700 
701  var voxRay = new LineD(start, end);
702 
703  double startOffset;
704  double endOffset;
705  if (!planet.RootVoxel.Storage.DataProvider.Intersect(ref voxRay, out startOffset, out endOffset))
706  continue;
707 
708  var from = voxRay.From;
709  voxRay.From = from + voxRay.Direction * voxRay.Length * startOffset;
710  voxRay.To = from + voxRay.Direction * voxRay.Length * endOffset;
711 
712  start = voxRay.From - planet.SizeInMetresHalf;
