673  void findViews(int species, Vector3 cDIR, out Vector3I vv, out Vector3 rr)
674  {
675  Vector3 VDIR = new Vector3(cDIR.X, Math.Max(-cDIR.Y, 0.01f), cDIR.Z);
676  float a = Math.Abs(VDIR.X) > Math.Abs(VDIR.Z) ? -VDIR.Z / VDIR.X : -VDIR.X / -VDIR.Z;
677  float nxx = N * (1.0f - a) * (float)Math.Acos(MathHelper.Clamp(VDIR.Y, -1, 1)) / 3.141592657f; // uniform sampling in theta
678  float nyy = N * (1.0f + a) * (float)Math.Acos(MathHelper.Clamp(VDIR.Y, -1, 1)) / 3.141592657f;
679  int i = (int)Math.Floor(nxx);
680  int j = (int)Math.Floor(nyy);
681  float ti = nxx - i;
682  float tj = nyy - j;
683  float alpha = 1.0f - ti - tj;
684  bool b = alpha > 0.0;
685  Vector3I ii = new Vector3I(b ? i : i + 1, i + 1, i);
686  Vector3I jj = new Vector3I(b ? j : j + 1, j, j + 1);
687  rr = new Vector3(Math.Abs(alpha), b ? ti : 1.0 - tj, b ? tj : 1.0 - ti);
688  if (Math.Abs(VDIR.Z) >= Math.Abs(VDIR.X))
689  {
690  Vector3I tmp = ii;
691  ii = -jj;
692  jj = tmp;
693  }
694  if (Math.Abs(VDIR.X + -VDIR.Z) > 0.00001f)
695  {
696  ii *= (int)Math.Sign(VDIR.X + -VDIR.Z);
697  jj *= (int)Math.Sign(VDIR.X + -VDIR.Z);
698  }
699  vv = new Vector3I(species * NT) + new Vector3I(viewNumber(ii.X, jj.X), viewNumber(ii.Y, jj.Y), viewNumber(ii.Z, jj.Z));
700  }
