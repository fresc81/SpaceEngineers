427  public override bool TransferItemsFrom(MyInventoryBase sourceInventory, IMyInventoryItem item, MyFixedPoint amount)
428  {
429  if (sourceInventory == null)
430  {
431  System.Diagnostics.Debug.Fail("Source inventory is null!");
432  return false;
433  }
434  MyInventoryBase destinationInventory = this;
435  if (destinationInventory == null)
436  {
437  System.Diagnostics.Debug.Fail("Destionation inventory is null!");
438  return false;
439  }
440  if (item == null)
441  {
442  System.Diagnostics.Debug.Fail("Item is null!");
443  return false;
444  }
445  if (amount == 0)
446  {
447  return true;
448  }
449 
450  bool transfered = false;
451  if ((destinationInventory.ItemsCanBeAdded(amount, item) || destinationInventory == sourceInventory) && sourceInventory.ItemsCanBeRemoved(amount, item))
452  {
453  if (Sync.IsServer)
454  {
455  if (destinationInventory != sourceInventory)
456  {
457  // try to add first and then remove to ensure this items don't disappear
458  if (destinationInventory.Add(item, amount))
459  {
460  if (sourceInventory.Remove(item, amount))
461  {
462  // successfull transaction
463  return true;
464  }
465  else
466  {
467  // This can happend, that it can't be removed due to some lock, then we need to revert the add.
468  destinationInventory.Remove(item, amount);
469  }
470  }
471  }
472  else
473  {
474  // same inventory transfer = splitting amount, need to remove first and add second
475  if (sourceInventory.Remove(item, amount) && destinationInventory.Add(item, amount))
476  {
477  return true;
478  }
479  else
480  {
481  System.Diagnostics.Debug.Fail("Error! Unsuccesfull splitting!");
482  }
483  }
484  }
485  else
486  {
487  Debug.Assert(sourceInventory != null);
488  MyInventoryTransferEventContent eventParams = new MyInventoryTransferEventContent();
489  eventParams.Amount = amount;
490  eventParams.ItemId = item.ItemId;
491  eventParams.SourceOwnerId = sourceInventory.Entity.EntityId;
492  eventParams.SourceInventoryId = sourceInventory.InventoryId;
493  eventParams.DestinationOwnerId = destinationInventory.Entity.EntityId;
494  eventParams.DestinationInventoryId = destinationInventory.InventoryId;
495  MyMultiplayer.RaiseStaticEvent(s => InventoryBaseTransferItem_Implementation, eventParams);
496  }
497  }
498 
499  return transfered;
500  }
